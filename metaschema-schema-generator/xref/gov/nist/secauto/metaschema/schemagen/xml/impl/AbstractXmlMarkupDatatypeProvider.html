6   package gov.nist.secauto.metaschema.schemagen.xml.impl;
7   
8   import gov.nist.secauto.metaschema.core.model.IModule;
9   import gov.nist.secauto.metaschema.core.util.CollectionUtil;
10  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
11  
12  import org.eclipse.jdt.annotation.Owning;
13  import org.jdom2.Element;
14  
15  import java.io.InputStream;
16  import java.util.List;
17  import java.util.Map;
18  import java.util.stream.Collectors;
19  
20  import edu.umd.cs.findbugs.annotations.NonNull;
21  
22  public abstract class AbstractXmlMarkupDatatypeProvider
23      extends AbstractXmlDatatypeProvider {
24  
25    @SuppressWarnings("null")
26    @Owning
27    @Override
28    protected InputStream getSchemaResource() {
29      return IModule.class.getResourceAsStream(getSchemaResourcePath());
30    }
31  
32    /**
33     * Get the absolute classpath of the schema resource.
34     *
35     * @return the resource path
36     */
37    @NonNull
38    protected abstract String getSchemaResourcePath();
39  
40    @Override
41    protected List<Element> queryElements(JDom2XmlSchemaLoader loader) {
42      return loader.getContent(
43          "/xs:schema/*",
44          CollectionUtil.singletonMap("xs", JDom2XmlSchemaLoader.NS_XML_SCHEMA));
45    }
46  
47    @NonNull
48    protected abstract String getDataTypeName();
49  
50    @Override
51    protected Map<String, IDatatypeContent> handleResults(@NonNull List<Element> items) {
52      String dataTypeName = getDataTypeName();
53      return CollectionUtil.singletonMap(
54          dataTypeName,
55          new JDom2DatatypeContent(
56              dataTypeName,
57              ObjectUtils.notNull(items.stream()
58                  .filter(element -> !"include".equals(element.getName()))
59                  .collect(Collectors.toList())),
60              CollectionUtil.emptyList()));
61    }
62  
63  }

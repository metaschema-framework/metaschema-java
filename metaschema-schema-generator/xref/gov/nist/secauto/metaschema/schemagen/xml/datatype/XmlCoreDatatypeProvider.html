6   package gov.nist.secauto.metaschema.schemagen.xml.datatype;
7   
8   import gov.nist.secauto.metaschema.core.model.IModule;
9   import gov.nist.secauto.metaschema.core.util.CollectionUtil;
10  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
11  import gov.nist.secauto.metaschema.schemagen.datatype.IDatatypeContent;
12  
13  import org.jdom2.Attribute;
14  import org.jdom2.Element;
15  import org.jdom2.filter.Filters;
16  import org.jdom2.xpath.XPathExpression;
17  import org.jdom2.xpath.XPathFactory;
18  
19  import java.io.InputStream;
20  import java.util.LinkedHashMap;
21  import java.util.List;
22  import java.util.Map;
23  import java.util.function.Function;
24  import java.util.stream.Collectors;
25  
26  import edu.umd.cs.findbugs.annotations.NonNull;
27  
28  public class XmlCoreDatatypeProvider
29      extends AbstractXmlDatatypeProvider {
30  
31    @Override
32    protected InputStream getSchemaResource() {
33      return IModule.class.getResourceAsStream("/schema/xml/metaschema-datatypes.xsd");
34    }
35  
36    @Override
37    protected List<Element> queryElements(JDom2XmlSchemaLoader loader) {
38      return loader.getContent(
39          "/xs:schema/xs:simpleType",
40          CollectionUtil.singletonMap("xs", JDom2XmlSchemaLoader.NS_XML_SCHEMA));
41    }
42  
43    @NonNull
44    private static List<String> analyzeDependencies(@NonNull Element element) {
45      XPathExpression<Attribute> xpath = XPathFactory.instance().compile(".//@base", Filters.attribute());
46      return ObjectUtils.notNull(xpath.evaluate(element).stream()
47          .map(attr -> attr.getValue())
48          .filter(type -> !type.startsWith("xs:"))
49          .distinct()
50          .collect(Collectors.toList()));
51    }
52  
53    @Override
54    protected @NonNull Map<String, IDatatypeContent> handleResults(
55        @NonNull List<Element> items) {
56      return ObjectUtils.notNull(items.stream()
57          .map(element -> {
58            return (IDatatypeContent) new JDom2DatatypeContent(
59                ObjectUtils.requireNonNull(element.getAttributeValue("name")),
60                CollectionUtil.singletonList(element),
61                analyzeDependencies(element));
62          }).collect(Collectors.toMap(content -> content.getTypeName(), Function.identity(), (e1, e2) -> e2,
63              LinkedHashMap::new)));
64    }
65  }

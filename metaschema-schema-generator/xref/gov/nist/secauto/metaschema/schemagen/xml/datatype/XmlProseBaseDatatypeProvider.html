6   package gov.nist.secauto.metaschema.schemagen.xml.datatype;
7   
8   import gov.nist.secauto.metaschema.core.model.IModule;
9   import gov.nist.secauto.metaschema.core.util.CollectionUtil;
10  import gov.nist.secauto.metaschema.schemagen.datatype.IDatatypeContent;
11  
12  import org.jdom2.Element;
13  
14  import java.io.InputStream;
15  import java.util.List;
16  import java.util.Map;
17  
18  import edu.umd.cs.findbugs.annotations.NonNull;
19  
20  public class XmlProseBaseDatatypeProvider
21      extends AbstractXmlDatatypeProvider {
22    private static final String DATATYPE_NAME = "ProseBase";
23  
24    @Override
25    protected InputStream getSchemaResource() {
26      return IModule.class
27          .getResourceAsStream("/schema/xml/metaschema-prose-base.xsd");
28    }
29  
30    @Override
31    protected List<Element> queryElements(JDom2XmlSchemaLoader loader) {
32      return loader.getContent(
33          "/xs:schema/*",
34          CollectionUtil.singletonMap("xs", JDom2XmlSchemaLoader.NS_XML_SCHEMA));
35    }
36  
37    @Override
38    protected @NonNull Map<String, IDatatypeContent> handleResults(@NonNull List<Element> items) {
39      return CollectionUtil.singletonMap(
40          DATATYPE_NAME,
41          new JDom2DatatypeContent(
42              DATATYPE_NAME,
43              items,
44              CollectionUtil.emptyList()));
45    }
46  }

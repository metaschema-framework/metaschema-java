6   package gov.nist.secauto.metaschema.schemagen.xml.schematype;
7   
8   import gov.nist.secauto.metaschema.core.datatype.markup.MarkupLine;
9   import gov.nist.secauto.metaschema.core.model.IValuedDefinition;
10  import gov.nist.secauto.metaschema.core.model.constraint.IAllowedValue;
11  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
12  import gov.nist.secauto.metaschema.schemagen.AbstractGenerationState.AllowedValueCollection;
13  import gov.nist.secauto.metaschema.schemagen.SchemaGenerationException;
14  import gov.nist.secauto.metaschema.schemagen.xml.XmlSchemaGenerator;
15  import gov.nist.secauto.metaschema.schemagen.xml.impl.XmlGenerationState;
16  
17  import org.codehaus.stax2.XMLStreamWriter2;
18  
19  import javax.xml.namespace.QName;
20  import javax.xml.stream.XMLStreamException;
21  
22  import edu.umd.cs.findbugs.annotations.NonNull;
23  
24  public class XmlSimpleTypeDataTypeRestriction
25      extends AbstractXmlSimpleType {
26    @NonNull
27    private final AllowedValueCollection allowedValuesCollection;
28  
29    public XmlSimpleTypeDataTypeRestriction(
30        @NonNull QName qname,
31        @NonNull IValuedDefinition definition,
32        @NonNull AllowedValueCollection allowedValuesCollection) {
33      super(qname, definition);
34      this.allowedValuesCollection = allowedValuesCollection;
35    }
36  
37    protected AllowedValueCollection getAllowedValuesCollection() {
38      return allowedValuesCollection;
39    }
40  
41    @Override
42    public boolean isInline(XmlGenerationState state) {
43      return true;
44    }
45  
46    @Override
47    public boolean isGeneratedType(XmlGenerationState state) {
48      return true;
49    }
50  
51    @Override
52    public void generate(XmlGenerationState state) {
53      try {
54        state.writeStartElement(XmlSchemaGenerator.PREFIX_XML_SCHEMA, "simpleType", XmlSchemaGenerator.NS_XML_SCHEMA);
55  
56        if (!isInline(state)) {
57          state.writeAttribute("name", ObjectUtils.notNull(getQName().getLocalPart()));
58        }
59  
60        state.writeStartElement(XmlSchemaGenerator.PREFIX_XML_SCHEMA, "restriction", XmlSchemaGenerator.NS_XML_SCHEMA);
61        state.writeAttribute("base", state.getSimpleType(getDataTypeAdapter()).getTypeReference());
62  
63        for (IAllowedValue allowedValue : getAllowedValuesCollection().getValues()) {
64          state.writeStartElement(XmlSchemaGenerator.PREFIX_XML_SCHEMA, "enumeration", XmlSchemaGenerator.NS_XML_SCHEMA);
65          state.writeAttribute("value", allowedValue.getValue());
66  
67          MarkupLine description = allowedValue.getDescription();
68          if (!description.isEmpty()) {
69            generateDescriptionAnnotation(
70                description,
71                ObjectUtils.notNull(getQName().getNamespaceURI()),
72                state);
73            // LOGGER.info(String.format("Field:%s:%s: %s",
74            // definition.getContainingMetaschema().getLocation(),
75            // definition.getName(), allowedValue.getValue()));
76          }
77          state.writeEndElement(); // xs:enumeration
78        }
79  
80        state.writeEndElement(); // xs:restriction
81        state.writeEndElement(); // xs:simpleType
82      } catch (XMLStreamException ex) {
83        throw new SchemaGenerationException(ex);
84      }
85    }
86  
87    public static void generateDescriptionAnnotation(
88        @NonNull MarkupLine description,
89        @NonNull String xmlNS,
90        @NonNull XmlGenerationState state) throws XMLStreamException {
91      XMLStreamWriter2 writer = state.getXMLStreamWriter();
92      writer.writeStartElement(XmlSchemaGenerator.PREFIX_XML_SCHEMA, "annotation", XmlSchemaGenerator.NS_XML_SCHEMA);
93      writer.writeStartElement(XmlSchemaGenerator.PREFIX_XML_SCHEMA, "documentation", XmlSchemaGenerator.NS_XML_SCHEMA);
94  
95      // write description
96      writer.writeStartElement(xmlNS, "p");
97  
98      description.writeXHtml(xmlNS, writer);
99  
100     writer.writeEndElement(); // p
101 
102     writer.writeEndElement(); // xs:documentation
103     writer.writeEndElement(); // xs:annotation
104   }
105 }

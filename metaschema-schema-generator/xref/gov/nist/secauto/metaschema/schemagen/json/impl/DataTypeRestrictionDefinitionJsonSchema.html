6   package gov.nist.secauto.metaschema.schemagen.json.impl;
7   
8   import com.fasterxml.jackson.databind.node.ArrayNode;
9   import com.fasterxml.jackson.databind.node.JsonNodeFactory;
10  import com.fasterxml.jackson.databind.node.ObjectNode;
11  
12  import gov.nist.secauto.metaschema.core.datatype.IDataTypeAdapter;
13  import gov.nist.secauto.metaschema.core.model.IValuedDefinition;
14  import gov.nist.secauto.metaschema.core.model.constraint.IAllowedValue;
15  import gov.nist.secauto.metaschema.core.util.CollectionUtil;
16  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
17  import gov.nist.secauto.metaschema.schemagen.AbstractGenerationState.AllowedValueCollection;
18  import gov.nist.secauto.metaschema.schemagen.json.IDataTypeJsonSchema;
19  import gov.nist.secauto.metaschema.schemagen.json.IDefinitionJsonSchema;
20  import gov.nist.secauto.metaschema.schemagen.json.IJsonGenerationState;
21  
22  import java.math.BigDecimal;
23  import java.math.BigInteger;
24  import java.math.MathContext;
25  import java.util.Map;
26  
27  import edu.umd.cs.findbugs.annotations.NonNull;
28  
29  public class DataTypeRestrictionDefinitionJsonSchema
30      extends AbstractDefineableJsonSchema
31      implements IDataTypeJsonSchema, IDefinitionJsonSchema<IValuedDefinition> {
32    @NonNull
33    private final IValuedDefinition definition;
34    @NonNull
35    private final AllowedValueCollection allowedValuesCollection;
36  
37    public DataTypeRestrictionDefinitionJsonSchema(
38        @NonNull IValuedDefinition definition,
39        @NonNull AllowedValueCollection allowedValuesCollection) {
40      this.definition = definition;
41      CollectionUtil.requireNonEmpty(allowedValuesCollection.getValues());
42      this.allowedValuesCollection = allowedValuesCollection;
43    }
44  
45    @Override
46    public IKey getKey() {
47      return IKey.of(definition, null, null, null);
48    }
49  
50    @Override
51    @NonNull
52    public IValuedDefinition getDefinition() {
53      return definition;
54    }
55  
56    @Override
57    public IDataTypeAdapter<?> getDataTypeAdapter() {
58      return getDefinition().getJavaTypeAdapter();
59    }
60  
61    @NonNull
62    protected AllowedValueCollection getAllowedValuesCollection() {
63      return allowedValuesCollection;
64    }
65  
66    @Override
67    public boolean isInline(IJsonGenerationState state) {
68      // // inline if the definition is inline
69      // return state.isInline(definition);
70      // always inline
71      return true;
72    }
73  
74    @Override
75    protected String generateDefinitionName(IJsonGenerationState state) {
76      return state.getTypeNameForDefinition(definition, "Value");
77    }
78  
79    @Override
80    public void generateInlineSchema(ObjectNode obj, IJsonGenerationState state) {
81      // generate a restriction on the built-in type for the enumerated values
82      ArrayNode enumArray = JsonNodeFactory.instance.arrayNode();
83  
84      AllowedValueCollection allowedValuesCollection = getAllowedValuesCollection();
85      for (IAllowedValue allowedValue : allowedValuesCollection.getValues()) {
86        switch (getDefinition().getJavaTypeAdapter().getJsonRawType()) {
87        case STRING:
88          enumArray.add(allowedValue.getValue());
89          break;
90        case BOOLEAN:
91          enumArray.add(Boolean.parseBoolean(allowedValue.getValue()));
92          break;
93        case INTEGER:
94          enumArray.add(new BigInteger(allowedValue.getValue())); // NOPMD unavoidable
95          break;
96        case NUMBER:
97          enumArray.add(new BigDecimal(allowedValue.getValue(), MathContext.DECIMAL64)); // NOPMD unavoidable
98          break;
99        default:
100         throw new UnsupportedOperationException(getDefinition().getJavaTypeAdapter().getJsonRawType().toString());
101       }
102     }
103     // get schema for the built-in type
104     IDataTypeJsonSchema dataTypeSchema = state.getSchema(getDefinition().getJavaTypeAdapter());
105 
106     // if other values are allowed, we need to make a union of the restriction type
107     // and the base
108     // built-in type
109     ArrayNode ofArray;
110     if (allowedValuesCollection.isClosed()) {
111       // this restriction is allOf, since both must match
112       ofArray = obj.putArray("allOf");
113     } else {
114       // this restriction is anyOf, since any can match
115       ofArray = obj.putArray("anyOf");
116     }
117 
118     // add the data type reference
119     dataTypeSchema.generateSchemaOrRef(ObjectUtils.notNull(ofArray.addObject()), state);
120 
121     // add the enumeration
122     ofArray.addObject()
123         .set("enum", enumArray);
124   }
125 
126   @Override
127   public void gatherDefinitions(
128       @NonNull Map<IKey, IDefinitionJsonSchema<?>> gatheredDefinitions,
129       @NonNull IJsonGenerationState state) {
130     // ensure the base datatype is registered
131     state.getSchema(getDataTypeAdapter());
132 
133     // Generate a definition if the restricted definition is not inline
134     IKey key = getKey();
135     gatheredDefinitions.put(key, state.getSchema(getKey()));
136   }
137 }

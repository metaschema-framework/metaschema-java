6   package gov.nist.secauto.metaschema.core.model;
7   
8   import gov.nist.secauto.metaschema.core.datatype.markup.MarkupLine;
9   import gov.nist.secauto.metaschema.core.qname.IEnhancedQName;
10  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
11  
12  import java.util.Locale;
13  
14  import edu.umd.cs.findbugs.annotations.NonNull;
15  import edu.umd.cs.findbugs.annotations.Nullable;
16  
17  /**
18   * A trait indicating that the implementation is a localized definition that is
19   * declared in-line as an instance.
20   *
21   * @param <DEFINITION>
22   *          the associated definition Java type
23   * @param <INSTANCE>
24   *          the associated instance Java type
25   */
26  public interface IFeatureDefinitionInstanceInlined<
27      DEFINITION extends IDefinition,
28      INSTANCE extends INamedInstance>
29      extends IDefinition, INamedInstance {
30  
31    @Override
32    default IEnhancedQName getDefinitionQName() {
33      return getReferencedDefinitionQName();
34    }
35  
36    @Override
37    default DEFINITION getDefinition() {
38      return ObjectUtils.asType(this);
39    }
40  
41    @Override
42    default boolean isInline() {
43      // has to be inline
44      return true;
45    }
46  
47    @Override
48    @NonNull
49    default INSTANCE getInlineInstance() {
50      return ObjectUtils.asType(this);
51    }
52  
53    @Override
54    default String getEffectiveFormalName() {
55      return getFormalName();
56    }
57  
58    @Override
59    default MarkupLine getEffectiveDescription() {
60      return getDescription();
61    }
62  
63    @Override
64    default String getEffectiveName() {
65      // don't use use-name
66      return getName();
67    }
68  
69    @Override
70    default Integer getEffectiveIndex() {
71      return getIndex();
72    }
73  
74    @Override
75    @Nullable
76    default Object getEffectiveDefaultValue() {
77      // This is an inline instance that is both a definition and an instance. Don't
78      // delegate to the definition, since this would be redundant.
79      return getDefaultValue();
80    }
81  
82    /**
83     * Generates a "coordinate" string for the provided inline definition instance.
84     *
85     * A coordinate consists of the element's:
86     * <ul>
87     * <li>containing Metaschema module's short name</li>
88     * <li>model type</li>
89     * <li>definition name</li>
90     * <li>hash code</li>
91     * </ul>
92     *
93     * @return the coordinate
94     */
95    @SuppressWarnings("null")
96    @Override
97    default String toCoordinates() {
98      IModule module = getContainingModule();
99      return String.format("%s-inline-definition:%s:%s/%s@%d",
100         getModelType().toString().toLowerCase(Locale.ROOT),
101         module.getShortName(),
102         getContainingDefinition().getName(),
103         getName(),
104         hashCode());
105   }
106 }

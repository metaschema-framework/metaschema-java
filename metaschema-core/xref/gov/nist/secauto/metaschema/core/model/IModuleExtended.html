6   package gov.nist.secauto.metaschema.core.model;
7   
8   import gov.nist.secauto.metaschema.core.util.ObjectUtils;
9   
10  import java.util.Collection;
11  import java.util.List;
12  import java.util.function.Predicate;
13  import java.util.stream.Collectors;
14  import java.util.stream.Stream;
15  
16  import javax.xml.namespace.QName;
17  
18  import edu.umd.cs.findbugs.annotations.NonNull;
19  import edu.umd.cs.findbugs.annotations.Nullable;

21  /**
22   * The API for accessing information about a given Metaschema module.
23   * <p>
24   * A Metaschem module may import another Metaschema module. This import graph
25   * can be accessed using {@link #getImportedModules()}.
26   * <p>
27   * Global scoped Metaschema definitions can be accessed using
28   * {@link #getScopedAssemblyDefinitionByName(QName)},
29   * {@link #getScopedFieldDefinitionByName(QName)}, and
30   * {@link #getScopedFlagDefinitionByName(QName)}. These methods take into
31   * consideration the import order to provide the global definitions that are in
32   * scope within the given Metschema module.
33   * <p>
34   * Global scoped definitions exported by this Metaschema module, available for
35   * use by importing Metaschema modules, can be accessed using
36   * {@link #getExportedAssemblyDefinitions()},
37   * {@link #getExportedFieldDefinitions()}, and
38   * {@link #getExportedFlagDefinitions()}.
39   * <p>
40   * Global scoped definitions defined directly within the given Metaschema module
41   * can be accessed using {@link #getAssemblyDefinitions()},
42   * {@link #getFieldDefinitions()}, and {@link #getFlagDefinitions()}, along with
43   * similarly named access methods.
44   *
45   * @param <M>
46   *          the imported module Java type
47   * @param <D>
48   *          the model definition Java type
49   * @param <FL>
50   *          the flag definition Java type
51   * @param <FI>
52   *          the field definition Java type
53   * @param <A>
54   *          the assembly definition Java type
55   */
56  public interface IModuleExtended<
57      M extends IModuleExtended<M, D, FL, FI, A>,
58      D extends IModelDefinition,
59      FL extends IFlagDefinition,
60      FI extends IFieldDefinition,
61      A extends IAssemblyDefinition> extends IModule {
62  
63    /**
64     * Get a filter that will match all definitions that are not locally defined.
65     *
66     * @param <DEF>
67     *          the type of definition
68     * @return a predicate implementing the filter
69     */
70    static <DEF extends IDefinition> Predicate<DEF> allNonLocalDefinitions() {
71      return definition -> {
72        return ModuleScopeEnum.INHERITED.equals(definition.getModuleScope())
73            || ModelType.ASSEMBLY.equals(definition.getModelType()) && ((IAssemblyDefinition) definition).isRoot();
74      };
75    }
76  
77    /**
78     * Get a filter that will match all definitions that are root assemblies.
79     *
80     * @param <DEF>
81     *          the type of definition
82     * @return a predicate implementing the filter
83     */
84    static <DEF extends IDefinition> Predicate<DEF> allRootAssemblyDefinitions() {
85      return definition -> {
86        return ModelType.ASSEMBLY.equals(definition.getModelType()) && ((IAssemblyDefinition) definition).isRoot();
87      };
88    }
89  
90    @Override
91    @NonNull
92    List<? extends M> getImportedModules();
93  
94    @Override
95    @Nullable
96    M getImportedModuleByShortName(String name);
97  
98    @Override
99    @NonNull
100   Collection<FL> getFlagDefinitions();
101 
102   @Override
103   @Nullable
104   FL getFlagDefinitionByName(@NonNull QName name);
105 
106   @Override
107   @NonNull
108   Collection<A> getAssemblyDefinitions();
109 
110   @Override
111   @Nullable
112   A getAssemblyDefinitionByName(@NonNull QName name);
113 
114   @Override
115   @NonNull
116   Collection<FI> getFieldDefinitions();
117 
118   @Override
119   @Nullable
120   FI getFieldDefinitionByName(@NonNull QName name);
121 
122   @Override
123   @SuppressWarnings("unchecked")
124   @NonNull
125   default List<D> getAssemblyAndFieldDefinitions() {
126     return ObjectUtils.notNull(
127         Stream.concat(
128             (Stream<D>) getAssemblyDefinitions().stream(),
129             (Stream<D>) getFieldDefinitions().stream())
130             .collect(Collectors.toList()));
131   }
132 
133   @Override
134   @Nullable
135   default A getScopedAssemblyDefinitionByName(@NonNull QName name) {
136     // first try local/global top-level definitions from current metaschema module
137     A retval = getAssemblyDefinitionByName(name);
138     if (retval == null) {
139       // try global definitions from imported Metaschema modules
140       retval = getExportedAssemblyDefinitionByName(name);
141     }
142     return retval;
143   }
144 
145   @Override
146   @Nullable
147   default FI getScopedFieldDefinitionByName(@NonNull QName name) {
148     // first try local/global top-level definitions from current metaschema module
149     FI retval = getFieldDefinitionByName(name);
150     if (retval == null) {
151       // try global definitions from imported metaschema modules
152       retval = getExportedFieldDefinitionByName(name);
153     }
154     return retval;
155   }
156 
157   @Override
158   @Nullable
159   default FL getScopedFlagDefinitionByName(@NonNull QName name) {
160     // first try local/global top-level definitions from current metaschema module
161     FL retval = getFlagDefinitionByName(name);
162     if (retval == null) {
163       // try global definitions from imported metaschema modules
164       retval = getExportedFlagDefinitionByName(name);
165     }
166     return retval;
167   }
168 
169   @Override
170   @NonNull
171   default Collection<? extends A> getExportedRootAssemblyDefinitions() {
172     return ObjectUtils.notNull(getExportedAssemblyDefinitions().stream()
173         .filter(allRootAssemblyDefinitions())
174         .collect(Collectors.toList()));
175   }
176 
177   @Override
178   @NonNull
179   default Collection<? extends A> getRootAssemblyDefinitions() {
180     return ObjectUtils.notNull(getAssemblyDefinitions().stream()
181         .filter(allRootAssemblyDefinitions())
182         .collect(Collectors.toList()));
183   }
184 
185   @Override
186   @NonNull
187   Collection<? extends FL> getExportedFlagDefinitions();
188 
189   @Override
190   @Nullable
191   FL getExportedFlagDefinitionByName(QName name);
192 
193   @Override
194   @NonNull
195   Collection<? extends FI> getExportedFieldDefinitions();
196 
197   @Override
198   @Nullable
199   FI getExportedFieldDefinitionByName(QName name);
200 
201   @Override
202   @NonNull
203   Collection<? extends A> getExportedAssemblyDefinitions();
204 
205   @Override
206   @Nullable
207   A getExportedAssemblyDefinitionByName(QName name);
208 }

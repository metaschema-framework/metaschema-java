6   package gov.nist.secauto.metaschema.core.model;
7   
8   import gov.nist.secauto.metaschema.core.model.util.ModuleUtils;
9   
10  import edu.umd.cs.findbugs.annotations.NonNull;
11  
12  /**
13   * A base class for a field definition defined globally within a Metaschema
14   * module.
15   *
16   * @param <MODULE>
17   *          the Java type of the containing module
18   * @param <INSTANCE>
19   *          the expected Java type of an instance of this definition
20   * @param <FLAG>
21   *          the expected Java type of flag children
22   */
23  public abstract class AbstractGlobalFieldDefinition<
24      MODULE extends IModule,
25      INSTANCE extends IFieldInstance,
26      FLAG extends IFlagInstance>
27      extends AbstractGlobalDefinition<MODULE, INSTANCE>
28      implements IFieldDefinition, IFeatureContainerFlag<FLAG> {
29  
30    /**
31     * Construct a new global field definition.
32     *
33     * @param module
34     *          the parent module containing this definition
35     */
36    protected AbstractGlobalFieldDefinition(@NonNull MODULE module) {
37      super(module, name -> ModuleUtils.parseModelName(module, name));
38    }
39  }

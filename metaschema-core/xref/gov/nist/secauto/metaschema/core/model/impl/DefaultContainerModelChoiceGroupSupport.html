7   
8   import gov.nist.secauto.metaschema.core.model.AbstractContainerModelSupport;
9   import gov.nist.secauto.metaschema.core.model.IAssemblyInstance;
10  import gov.nist.secauto.metaschema.core.model.IFieldInstance;
11  import gov.nist.secauto.metaschema.core.model.INamedModelInstance;
12  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
13  
14  import java.util.Collection;
15  import java.util.LinkedHashMap;
16  import java.util.Map;
17  
18  import edu.umd.cs.findbugs.annotations.NonNull;
19  
20  /**
21   * Supports choice group model instance operations on assembly model instances.
22   * <p>
23   * This implementation uses underlying {@link LinkedHashMap} instances to
24   * preserve ordering.
25   *
26   * @param <NMI>
27   *          the named model instance Java type
28   * @param <FI>
29   *          the field instance Java type
30   * @param <AI>
31   *          the assembly instance Java type
32   */
33  public class DefaultContainerModelChoiceGroupSupport<
34      NMI extends INamedModelInstance,
35      FI extends IFieldInstance,
36      AI extends IAssemblyInstance>
37      extends AbstractContainerModelSupport<NMI, NMI, FI, AI> {
38  
39    /**
40     * Construct an new container using the provided collections.
41     *
42     * @param namedModelInstances
43     *          a collection of named model instances
44     * @param fieldInstances
45     *          a collection of field instances
46     * @param assemblyInstances
47     *          a collection of assembly instances
48     */
49    public DefaultContainerModelChoiceGroupSupport(
50        @NonNull Map<Integer, NMI> namedModelInstances,
51        @NonNull Map<Integer, FI> fieldInstances,
52        @NonNull Map<Integer, AI> assemblyInstances) {
53      super(namedModelInstances, fieldInstances, assemblyInstances);
54    }
55  
56    @Override
57    public Collection<NMI> getModelInstances() {
58      return ObjectUtils.notNull(getNamedModelInstanceMap().values());
59    }
60  }

24  /**
25   * Represents an expect constraint.
26   * <p>
27   * Requires that the associated test evaluates to {@link IBooleanItem#TRUE}
28   * against the target.
29   */
30  public final class DefaultExpectConstraint
31      extends AbstractConfigurableMessageConstraint
32      implements IExpectConstraint {
33    @NonNull
34    private final Lazy<MetapathExpression> testMetapath;
35  
36    /**
37     * Construct a new expect constraint.
38     *
39     * @param id
40     *          the optional identifier for the constraint
41     * @param formalName
42     *          the constraint's formal name or {@code null} if not provided
43     * @param description
44     *          the constraint's semantic description or {@code null} if not
45     *          provided
46     * @param source
47     *          information about the constraint source
48     * @param level
49     *          the significance of a violation of this constraint
50     * @param target
51     *          the Metapath expression identifying the nodes the constraint targets
52     * @param properties
53     *          a collection of associated properties
54     * @param test
55     *          a Metapath expression that is evaluated against the target node to
56     *          determine if the constraint passes
57     * @param message
58     *          an optional message to emit when the constraint is violated
59     * @param remarks
60     *          optional remarks describing the intent of the constraint
61     */
62    @SuppressWarnings("PMD.ExcessiveParameterList")
63    public DefaultExpectConstraint(
64        @Nullable String id,
65        @Nullable String formalName,
66        @Nullable MarkupLine description,
67        @NonNull ISource source,
68        @NonNull Level level,
69        @NonNull String target,
70        @NonNull Map<IAttributable.Key, Set<String>> properties,
71        @NonNull String test,
72        @Nullable String message,
73        @Nullable MarkupMultiline remarks) {
74      super(id, formalName, description, source, level, target, properties, message, remarks);
75      this.testMetapath = ObjectUtils.notNull(
76          Lazy.lazy(() -> MetapathExpression.compile(
77              test,
78              source.getStaticContext())));
79    }
80  
81    /**
82     * Get the compiled Metapath expression for the test.
83     *
84     * @return the compiled Metapath expression
85     */
86    @NonNull
87    public MetapathExpression getTestMetapath() {
88      return ObjectUtils.notNull(testMetapath.get());
89    }
90  
91    @Override
92    public String getTest() {
93      return getTestMetapath().getPath();
94    }
95  }

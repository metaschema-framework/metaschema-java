6   package gov.nist.secauto.metaschema.core.model.xml.impl;
7   
8   import gov.nist.secauto.metaschema.core.model.IAssemblyInstanceGrouped;
9   import gov.nist.secauto.metaschema.core.model.IContainerModelSupport;
10  import gov.nist.secauto.metaschema.core.model.IFieldInstanceGrouped;
11  import gov.nist.secauto.metaschema.core.model.INamedModelInstanceGrouped;
12  import gov.nist.secauto.metaschema.core.util.CollectionUtil;
13  
14  import java.util.Collection;
15  import java.util.LinkedHashMap;
16  import java.util.Map;
17  
18  import javax.xml.namespace.QName;
19  
20  import edu.umd.cs.findbugs.annotations.NonNull;
21  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
22  
23  /**
24   * Supports grouped model instance operations on assembly model instances.
25   * <p>
26   * This implementation uses underlying {@link LinkedHashMap} instances to
27   * preserve ordering.
28   * <p>
29   * Since a choice group only contains named model instances (i.e., fields,
30   * assemblies), model instance operations are supported by the map returned by
31   * {@link #getNamedModelInstanceMap()}.
32   *
33   * @param <NMI>
34   *          the named model instance Java type
35   * @param <FI>
36   *          the field instance Java type
37   * @param <AI>
38   *          the assembly instance Java type
39   */
40  public class DefaultGroupedModelContainerSupport<
41      NMI extends INamedModelInstanceGrouped,
42      FI extends IFieldInstanceGrouped,
43      AI extends IAssemblyInstanceGrouped>
44      implements IContainerModelSupport<NMI, NMI, FI, AI> {
45  
46    @NonNull
47    private final Map<QName, NMI> namedModelInstances;
48    @NonNull
49    private final Map<QName, FI> fieldInstances;
50    @NonNull
51    private final Map<QName, AI> assemblyInstances;
52  
53    /**
54     * Construct an empty, mutable container.
55     */
56    @SuppressWarnings("PMD.UseConcurrentHashMap")
57    public DefaultGroupedModelContainerSupport() {
58      this(
59          new LinkedHashMap<>(),
60          new LinkedHashMap<>(),
61          new LinkedHashMap<>());
62    }
63  
64    /**
65     * Construct an immutable container from a collection of named model instances.
66     *
67     * @param instances
68     *          the collection of named model instances to add to the new container.
69     * @param fieldClass
70     *          the Java type for field instances
71     * @param assemblyClass
72     *          the Java type for assembly instances
73     */
74    @SuppressWarnings({ "PMD.UseConcurrentHashMap" })
75    @SuppressFBWarnings(value = "CT_CONSTRUCTOR_THROW", justification = "Use of final fields")
76    private DefaultGroupedModelContainerSupport(
77        @NonNull Collection<NMI> instances,
78        @NonNull Class<FI> fieldClass,
79        @NonNull Class<AI> assemblyClass) {
80      assert !fieldClass.isAssignableFrom(assemblyClass) : String.format(
81          "The field class '%s' must not be assignment compatible to the assembly class '%s'.",
82          fieldClass.getName(),
83          assemblyClass.getName());
84  
85      Map<QName, NMI> namedModelInstances = new LinkedHashMap<>();
86      Map<QName, FI> fieldInstances = new LinkedHashMap<>();
87      Map<QName, AI> assemblyInstances = new LinkedHashMap<>();
88      for (NMI instance : instances) {
89        QName key = instance.getXmlQName();
90        namedModelInstances.put(key, instance);
91  
92        if (fieldClass.isInstance(instance)) {
93          fieldInstances.put(key, fieldClass.cast(instance));
94        } else if (assemblyClass.isInstance(instance)) {
95          assemblyInstances.put(key, assemblyClass.cast(instance));
96        }
97      }
98  
99      this.namedModelInstances = namedModelInstances.isEmpty()
100         ? CollectionUtil.emptyMap()
101         : CollectionUtil.unmodifiableMap(namedModelInstances);
102     this.fieldInstances = fieldInstances.isEmpty()
103         ? CollectionUtil.emptyMap()
104         : CollectionUtil.unmodifiableMap(fieldInstances);
105     this.assemblyInstances = assemblyInstances.isEmpty()
106         ? CollectionUtil.emptyMap()
107         : CollectionUtil.unmodifiableMap(assemblyInstances);
108   }
109 
110   /**
111    * Construct an new container using the provided collections.
112    *
113    * @param namedModelInstances
114    *          a collection of named model instances
115    * @param fieldInstances
116    *          a collection of field instances
117    * @param assemblyInstances
118    *          a collection of assembly instances
119    */
120   protected DefaultGroupedModelContainerSupport(
121       @NonNull Map<QName, NMI> namedModelInstances,
122       @NonNull Map<QName, FI> fieldInstances,
123       @NonNull Map<QName, AI> assemblyInstances) {
124     this.namedModelInstances = namedModelInstances;
125     this.fieldInstances = fieldInstances;
126     this.assemblyInstances = assemblyInstances;
127   }
128 
129   @SuppressWarnings("null")
130   @Override
131   public Collection<NMI> getModelInstances() {
132     return namedModelInstances.values();
133   }
134 
135   @Override
136   public Map<QName, NMI> getNamedModelInstanceMap() {
137     return namedModelInstances;
138   }
139 
140   @Override
141   public Map<QName, FI> getFieldInstanceMap() {
142     return fieldInstances;
143   }
144 
145   @Override
146   public Map<QName, AI> getAssemblyInstanceMap() {
147     return assemblyInstances;
148   }
149 }

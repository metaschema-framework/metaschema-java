33  /**
34   * Supports validating an XML resource using an XML schema.
35   */
36  public class XmlSchemaContentValidator
37      extends AbstractContentValidator {
38    private final Schema schema;
39  
40    @SuppressWarnings({ "resource", "PMD.UseTryWithResources" })
41    @NonNull
42    private static Schema toSchema(@NonNull List<? extends Source> schemaSources) throws IOException {
43      SchemaFactory schemafactory = SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI);
44      // schemafactory.setResourceResolver(new ClasspathResourceResolver());
45      // TODO verify source input streams are closed
46      try {
47        return ObjectUtils.notNull(schemaSources.isEmpty()
48            ? schemafactory.newSchema()
49            : schemafactory.newSchema(schemaSources.toArray(new Source[0])));
50      } catch (SAXException ex) {
51        throw new IOException(ex);
52      } finally {
53        // Close all source input streams
54        for (Source source : schemaSources) {
55          if (source instanceof StreamSource) {
56            StreamSource streamSource = (StreamSource) source;
57            if (streamSource.getInputStream() != null) {
58              streamSource.getInputStream().close();
59            }
60          }
61        }
62      }
63    }
64  
65    /**
66     * Construct a new XML schema validator using the provided XML schema sources.
67     *
68     * @param schemaSources
69     *          the XML schemas to use for validation
70     * @throws IOException
71     *           if an error occurred while parsing the provided XML schemas
72     */
73    public XmlSchemaContentValidator(@Owning @NonNull List<? extends Source> schemaSources) throws IOException {
74      this(toSchema(ObjectUtils.requireNonNull(schemaSources, "schemaSources")));
75    }
76  
77    /**
78     * Construct a new XML schema validator using the provided pre-parsed XML
79     * schema(s).
80     *
81     * @param schema
82     *          the pre-parsed XML schema(s) to use for validation
83     */
84    protected XmlSchemaContentValidator(@NonNull Schema schema) {
85      this.schema = ObjectUtils.requireNonNull(schema, "schema");
86    }
87  
88    private Schema getSchema() {
89      return schema;
90    }
91  
92    @Override
93    public IValidationResult validate(InputStream is, URI documentUri) throws IOException {
94      Source xmlSource = new StreamSource(is, documentUri.toASCIIString());
95  
96      Validator validator = getSchema().newValidator();
97      XmlValidationErrorHandler errorHandler = new XmlValidationErrorHandler(documentUri);
98      validator.setErrorHandler(errorHandler);
99      try {
100       validator.validate(xmlSource);
101     } catch (SAXParseException ex) {
102       String location = ex.getLineNumber() > -1 && ex.getColumnNumber() > -1
103           ? String.format("at %d:%d", ex.getLineNumber(), ex.getColumnNumber())
104           : "";
105       throw new IOException(
106           String.format("Unexpected failure during validation of '%s'%s. %s",
107               documentUri,
108               location,
109               ex.getLocalizedMessage()),
110           ex);
111     } catch (SAXException ex) {
112       throw new IOException(
113           String.format("Unexpected failure during validation of '%s'. %s",
114               documentUri,
115               ex.getLocalizedMessage()),
116           ex);
117     }
118     return errorHandler;
119   }

121   /**
122    * Records an identified individual validation result found during XML schema
123    * validation.
124    */
125   public static class XmlValidationFinding implements IValidationFinding, IResourceLocation {
126     @NonNull
127     private final URI documentUri;
128     @NonNull
129     private final SAXParseException exception;
130     @NonNull
131     private final Level severity;
132 
133     /**
134      * Construct a new XML schema validation finding, which represents an issue
135      * identified during XML schema validation.
136      *
137      * @param severity
138      *          the finding significance
139      * @param exception
140      *          the XML schema validation exception generated during schema
141      *          validation representing the issue
142      * @param resourceUri
143      *          the resource the issue was found in
144      */
145     public XmlValidationFinding(
146         @NonNull Level severity,
147         @NonNull SAXParseException exception,
148         @NonNull URI resourceUri) {
149       this.severity = ObjectUtils.requireNonNull(severity, "severity");
150       this.exception = ObjectUtils.requireNonNull(exception, "exception");
151       this.documentUri = ObjectUtils.requireNonNull(resourceUri, "documentUri");
152     }
153 
154     @Override
155     public String getIdentifier() {
156       // always null
157       return null;
158     }
159 
160     @Override
161     public Kind getKind() {
162       return Level.WARNING.equals(getSeverity()) ? Kind.PASS : Kind.FAIL;
163     }
164 
165     @Override
166     public Level getSeverity() {
167       return severity;
168     }
169 
170     @Override
171     public URI getDocumentUri() {
172       String systemId = getCause().getSystemId();
173       return systemId == null ? documentUri : URI.create(systemId);
174     }
175 
176     @Override
177     public int getLine() {
178       return getCause().getLineNumber();
179     }
180 
181     @Override
182     public int getColumn() {
183       return getCause().getColumnNumber();
184     }
185 
186     @Override
187     public long getCharOffset() {
188       // not known
189       return -1;
190     }
191 
192     @Override
193     public long getByteOffset() {
194       // not known
195       return -1;
196     }
197 
198     @Override
199     public IResourceLocation getLocation() {
200       return this;
201     }
202 
203     @Override
204     public String getPathKind() {
205       // not known
206       return null;
207     }
208 
209     @Override
210     public String getPath() {
211       // not known
212       return null;
213     }
214 
215     @Override
216     public String getMessage() {
217       return getCause().getLocalizedMessage();
218     }
219 
220     @NonNull
221     @Override
222     public SAXParseException getCause() {
223       return exception;
224     }
225   }
226 
227   private static class XmlValidationErrorHandler implements ErrorHandler, IValidationResult {
228     @NonNull
229     private final URI documentUri;
230     @NonNull
231     private final List<XmlValidationFinding> findings = new LinkedList<>();
232     @NonNull
233     private Level highestSeverity = Level.INFORMATIONAL;
234 
235     public XmlValidationErrorHandler(@NonNull URI documentUri) {
236       this.documentUri = ObjectUtils.requireNonNull(documentUri, "documentUri");
237     }
238 
239     @NonNull
240     public URI getDocumentUri() {
241       return documentUri;
242     }
243 
244     private void adjustHighestSeverity(@NonNull Level severity) {
245       if (highestSeverity.ordinal() < severity.ordinal()) {
246         highestSeverity = severity;
247       }
248     }
249 
250     @SuppressWarnings("null")
251     @Override
252     public void warning(SAXParseException ex) throws SAXException {
253       findings.add(new XmlValidationFinding(Level.WARNING, ex, getDocumentUri()));
254       adjustHighestSeverity(Level.WARNING);
255     }
256 
257     @SuppressWarnings("null")
258     @Override
259     public void error(SAXParseException ex) throws SAXException {
260       findings.add(new XmlValidationFinding(Level.ERROR, ex, getDocumentUri()));
261       adjustHighestSeverity(Level.CRITICAL);
262     }
263 
264     @SuppressWarnings("null")
265     @Override
266     public void fatalError(SAXParseException ex) throws SAXException {
267       findings.add(new XmlValidationFinding(Level.CRITICAL, ex, getDocumentUri()));
268       adjustHighestSeverity(Level.CRITICAL);
269     }
270 
271     @SuppressWarnings("null")
272     @Override
273     @NonNull
274     public List<XmlValidationFinding> getFindings() {
275       return Collections.unmodifiableList(findings);
276     }
277 
278     @Override
279     public Level getHighestSeverity() {
280       return highestSeverity;
281     }
282   }

6   package gov.nist.secauto.metaschema.core.model.impl;
7   
8   import gov.nist.secauto.metaschema.core.model.IContainerFlagSupport;
9   import gov.nist.secauto.metaschema.core.model.IFlagInstance;
10  
11  import java.util.Map;
12  
13  import edu.umd.cs.findbugs.annotations.NonNull;
14  import edu.umd.cs.findbugs.annotations.Nullable;
15  
16  /**
17   * A flag container.
18   *
19   * @param <FI>
20   *          the Java type of the flags supported by the container
21   */
22  public class DefaultContainerFlagSupport<FI extends IFlagInstance> implements IContainerFlagSupport<FI> {
23    @NonNull
24    private final Map<Integer, FI> instances;
25    @Nullable
26    private final FI jsonKey;
27  
28    /**
29     * Construct a new flag container using the provided flag instances.
30     *
31     * @param instances
32     *          a collection of flag instances
33     * @param jsonKey
34     *          the JSON key flag instance or {@code null} if no JSON key is
35     *          configured
36     */
37    public DefaultContainerFlagSupport(
38        @NonNull Map<Integer, FI> instances,
39        @Nullable FI jsonKey) {
40      this.instances = instances;
41      this.jsonKey = jsonKey;
42    }
43  
44    @Override
45    public Map<Integer, FI> getFlagInstanceMap() {
46      return instances;
47    }
48  
49    /**
50     * Retrieves the flag instance to use as as the property name for the containing
51     * object in JSON who's value will be the object containing the flag.
52     *
53     * @return the flag instance if a JSON key is configured, or {@code null}
54     *         otherwise
55     */
56    @Override
57    public FI getJsonKeyFlagInstance() {
58      return jsonKey;
59    }
60  }

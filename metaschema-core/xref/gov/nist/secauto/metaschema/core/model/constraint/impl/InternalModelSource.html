6   package gov.nist.secauto.metaschema.core.model.constraint.impl;
7   
8   import gov.nist.secauto.metaschema.core.metapath.StaticContext;
9   import gov.nist.secauto.metaschema.core.model.IModule;
10  import gov.nist.secauto.metaschema.core.model.ISource;
11  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
12  
13  import java.net.URI;
14  import java.util.HashMap;
15  import java.util.Map;
16  import java.util.concurrent.locks.Lock;
17  import java.util.concurrent.locks.ReentrantLock;
18  
19  import edu.umd.cs.findbugs.annotations.NonNull;
20  
21  /**
22   * Implements a
23   * {@link gov.nist.secauto.metaschema.core.model.ISource.SourceType#MODEL}
24   * source with no associated resource.
25   */
26  public final class InternalModelSource implements ISource {
27    @NonNull
28    private static final Map<IModule, InternalModelSource> sources = new HashMap<>(); // NOPMD - intentional
29    @NonNull
30    private static final Lock SOURCE_LOCK = new ReentrantLock();
31    @NonNull
32    private final IModule module;
33  
34    /**
35     * Get a new instance of an external source associated with a resource
36     * {@code location}.
37     *
38     * @param module
39     *          the Metaschema module containing a constraint
40     * @return the source
41     */
42    @NonNull
43    public static ISource instance(@NonNull IModule module) {
44      SOURCE_LOCK.lock();
45      try {
46        return ObjectUtils.notNull(sources.computeIfAbsent(
47            module,
48            uri -> new InternalModelSource(module)));
49      } finally {
50        SOURCE_LOCK.unlock();
51      }
52    }
53  
54    private InternalModelSource(@NonNull IModule module) {
55      this.module = module;
56    }
57  
58    @Override
59    public SourceType getSourceType() {
60      return SourceType.MODEL;
61    }
62  
63    @Override
64    public URI getSource() {
65      return module.getLocation();
66    }
67  
68    @Override
69    public String toString() {
70      return "internal:" + getSource();
71    }
72  
73    @Override
74    public StaticContext getStaticContext() {
75      return module.getModuleStaticContext();
76    }
77  }

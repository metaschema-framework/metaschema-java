6   package gov.nist.secauto.metaschema.core.model;
7   
8   import gov.nist.secauto.metaschema.core.model.util.ModuleUtils;
9   
10  import edu.umd.cs.findbugs.annotations.NonNull;
11  
12  /**
13   * A base class for an assembly definition defined globally within a Metaschema
14   * module.
15   *
16   * @param <MODULE>
17   *          the Java type of the containing module
18   * @param <INSTANCE>
19   *          the expected Java type of an instance of this definition
20   * @param <FLAG>
21   *          the expected Java type of flag children
22   * @param <MODEL>
23   *          the expected Java type of model children
24   * @param <NAMED_MODEL>
25   *          the expected Java type of named model children
26   * @param <FIELD>
27   *          the expected Java type of field children
28   * @param <ASSEMBLY>
29   *          the expected Java type of assembly children
30   * @param <CHOICE>
31   *          the expected Java type of choice children
32   * @param <CHOICE_GROUP>
33   *          the expected Java type of choice group children
34   */
35  public abstract class AbstractGlobalAssemblyDefinition<
36      MODULE extends IModule,
37      INSTANCE extends IAssemblyInstance,
38      FLAG extends IFlagInstance,
39      MODEL extends IModelInstanceAbsolute,
40      NAMED_MODEL extends INamedModelInstanceAbsolute,
41      FIELD extends IFieldInstanceAbsolute,
42      ASSEMBLY extends IAssemblyInstanceAbsolute,
43      CHOICE extends IChoiceInstance,
44      CHOICE_GROUP extends IChoiceGroupInstance>
45      extends AbstractGlobalDefinition<MODULE, INSTANCE>
46      implements IAssemblyDefinition, IFeatureContainerFlag<FLAG>, IFeatureContainerModelAssembly<
47          MODEL,
48          NAMED_MODEL,
49          FIELD,
50          ASSEMBLY,
51          CHOICE,
52          CHOICE_GROUP> {
53  
54    /**
55     * Construct a new global assembly definition.
56     *
57     * @param module
58     *          the parent module containing this definition
59     */
60    protected AbstractGlobalAssemblyDefinition(@NonNull MODULE module) {
61      super(module, name -> ModuleUtils.parseModelName(module, name));
62    }
63  }

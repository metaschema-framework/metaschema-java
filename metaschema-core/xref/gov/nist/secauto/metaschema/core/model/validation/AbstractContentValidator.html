6   package gov.nist.secauto.metaschema.core.model.validation;
7   
8   import gov.nist.secauto.metaschema.core.model.AbstractResourceResolver;
9   import gov.nist.secauto.metaschema.core.util.ObjectUtils;
10  
11  import java.io.BufferedInputStream;
12  import java.io.IOException;
13  import java.io.InputStream;
14  import java.net.URI;
15  import java.net.URL;
16  
17  public abstract class AbstractContentValidator
18      extends AbstractResourceResolver
19      implements IContentValidator {
20  
21    @Override
22    public IValidationResult validate(URI uri) throws IOException {
23      URI resourceUri = resolve(uri);
24      URL resource = resourceUri.toURL();
25  
26      try (InputStream is = new BufferedInputStream(ObjectUtils.notNull(resource.openStream()))) {
27        return validate(is, resourceUri);
28      }
29    }
30  }

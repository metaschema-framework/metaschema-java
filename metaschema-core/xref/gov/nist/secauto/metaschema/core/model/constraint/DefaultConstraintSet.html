6   package gov.nist.secauto.metaschema.core.model.constraint;
7   
8   import gov.nist.secauto.metaschema.core.model.IModule;
9   import gov.nist.secauto.metaschema.core.model.ISource;
10  import gov.nist.secauto.metaschema.core.qname.IEnhancedQName;
11  import gov.nist.secauto.metaschema.core.util.CollectionUtil;
12  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
13  
14  import java.util.Collections;
15  import java.util.List;
16  import java.util.Map;
17  import java.util.Set;
18  import java.util.stream.Collectors;
19  
20  import javax.xml.namespace.QName;
21  
22  import edu.umd.cs.findbugs.annotations.NonNull;
23  
24  /**
25   * The default implementation of a constraint set sourced from an external
26   * constraint resource.
27   */
28  public class DefaultConstraintSet implements IConstraintSet {
29    @NonNull
30    private final ISource source;
31    @NonNull
32    private final Set<IConstraintSet> importedConstraintSets;
33    @NonNull
34    private final Map<IEnhancedQName, List<IScopedContraints>> scopedContraints;
35  
36    /**
37     * Construct a new constraint set.
38     *
39     * @param source
40     *          the resource the constraint was provided from
41     * @param scopedContraints
42     *          a set of constraints qualified by a scope path
43     * @param importedConstraintSets
44     *          constraint sets imported by this constraint set
45     */
46    @SuppressWarnings("null")
47    public DefaultConstraintSet(
48        @NonNull ISource source,
49        @NonNull List<IScopedContraints> scopedContraints,
50        @NonNull Set<IConstraintSet> importedConstraintSets) {
51      this.source = source;
52      this.scopedContraints = scopedContraints.stream()
53          .collect(
54              Collectors.collectingAndThen(
55                  Collectors.groupingBy(
56                      scope -> IEnhancedQName.of(scope.getModuleNamespace().toString(), scope.getModuleShortName()),
57                      Collectors.toUnmodifiableList()),
58                  Collections::unmodifiableMap));
59      this.importedConstraintSets = CollectionUtil.unmodifiableSet(importedConstraintSets);
60    }
61  
62    /**
63     * Get the resource the constraint was provided from.
64     *
65     * @return the resource
66     */
67    @Override
68    public ISource getSource() {
69      return source;
70    }
71  
72    /**
73     * Get the set of Metaschema scoped constraints to apply by a {@link QName}
74     * formed from the Metaschema namespace and short name.
75     *
76     * @return the mapping of QName to scoped constraints
77     */
78    @NonNull
79    public Map<IEnhancedQName, List<IScopedContraints>> getScopedContraints() {
80      return scopedContraints;
81    }
82  
83    @Override
84    public Set<IConstraintSet> getImportedConstraintSets() {
85      return importedConstraintSets;
86    }
87  
88    @Override
89    public Iterable<ITargetedConstraints> getTargetedConstraintsForModule(@NonNull IModule module) {
90      IEnhancedQName qname = module.getQName();
91  
92      Map<IEnhancedQName, List<IScopedContraints>> map = getScopedContraints();
93      List<IScopedContraints> scopes = map.getOrDefault(qname, CollectionUtil.emptyList());
94      return CollectionUtil.toIterable(ObjectUtils.notNull(scopes.stream()
95          .flatMap(scoped -> scoped.getTargetedContraints().stream())));
96    }
97  
98  }

13  public interface IFeatureContainerModelAssembly<

15      NMI extends INamedModelInstanceAbsolute,
16      FI extends IFieldInstanceAbsolute,

18      CI extends IChoiceInstance,
19      CGI extends IChoiceGroupInstance>

21      IFeatureContainerModelAbsolute<MI, NMI, FI, AI> {
22    /**
23     * Get the model container implementation instance.
24     *
25     * @return the model container instance
26     */
27    @Override
28    @NonNull
29    IContainerModelAssemblySupport<MI, NMI, FI, AI, CI, CGI> getModelContainer();
30  
31    @Override
32    default List<CI> getChoiceInstances() {
33      return getModelContainer().getChoiceInstances();
34    }
35  
36    @Override
37    default CGI getChoiceGroupInstanceByName(String name) {

40  
41    @Override
42    default Map<String, CGI> getChoiceGroupInstances() {
43      return getModelContainer().getChoiceGroupInstanceMap();
44    }

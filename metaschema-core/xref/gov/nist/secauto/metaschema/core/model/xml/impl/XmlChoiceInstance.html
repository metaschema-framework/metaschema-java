35  class XmlChoiceInstance
36      extends AbstractChoiceInstance<
37          IAssemblyDefinition,
38          IModelInstanceAbsolute,
39          INamedModelInstanceAbsolute,
40          IFieldInstanceAbsolute,
41          IAssemblyInstanceAbsolute> {
42    @NonNull
43    private final ChoiceType xmlChoice;
44    @NonNull
45    private final Lazy<XmlModelContainer> modelContainer;
46  
47    /**
48     * Constructs a mutually exclusive choice between two possible objects.
49     *
50     * @param xmlObject
51     *          the XML for the choice definition bound to Java objects
52     * @param parent
53     *          the parent container, either a choice or assembly
54     */
55    public XmlChoiceInstance(
56        @NonNull ChoiceType xmlObject,
57        @NonNull IAssemblyDefinition parent) {
58      super(parent);
59      this.xmlChoice = xmlObject;
60      this.modelContainer = ObjectUtils.notNull(Lazy.lazy(() -> new XmlModelContainer(xmlObject, this)));
61    }
62  
63    @Override
64    public XmlModelContainer getModelContainer() {
65      return ObjectUtils.notNull(modelContainer.get());
66    }
67  
68    // ----------------------------------------
69    // - Start XmlBeans driven code - CPD-OFF -
70    // ----------------------------------------
71  
72    /**
73     * Get the underlying XML data.
74     *
75     * @return the underlying XML data
76     */
77    @NonNull
78    protected ChoiceType getXmlObject() {
79      return xmlChoice;
80    }
81  
82    @Override
83    public MarkupMultiline getRemarks() {
84      // remarks not supported
85      return null;
86    }

92    @SuppressWarnings("PMD.UseConcurrentHashMap")
93    @NonNull
94    private static final XmlObjectParser<Pair<IChoiceInstance,
95        XmlModelContainer>> XML_MODEL_PARSER
96            = new XmlObjectParser<>(ObjectUtils.notNull(
97                Map.ofEntries(Map.entry(new QName(IModule.XML_NAMESPACE, "assembly"), XmlChoiceInstance::handleAssembly),
98                    Map.entry(new QName(IModule.XML_NAMESPACE, "define-assembly"),
99                        XmlChoiceInstance::handleDefineAssembly),
100                   Map.entry(new QName(IModule.XML_NAMESPACE, "field"), XmlChoiceInstance::handleField),
101                   Map.entry(new QName(IModule.XML_NAMESPACE, "define-field"), XmlChoiceInstance::handleDefineField)))) {
102 
103             @Override
104             protected Handler<Pair<IChoiceInstance, XmlModelContainer>> identifyHandler(XmlCursor cursor,
105                 XmlObject obj) {
106               Handler<Pair<IChoiceInstance, XmlModelContainer>> retval;
107               if (obj instanceof FieldReferenceType) {
108                 retval = XmlChoiceInstance::handleField;
109               } else if (obj instanceof InlineFieldDefinitionType) {
110                 retval = XmlChoiceInstance::handleDefineField;
111               } else if (obj instanceof AssemblyReferenceType) {
112                 retval = XmlChoiceInstance::handleAssembly;
113               } else if (obj instanceof InlineAssemblyDefinitionType) {
114                 retval = XmlChoiceInstance::handleDefineAssembly;
115               } else {
116                 retval = super.identifyHandler(cursor, obj);
117               }
118               return retval;
119             }
120           };
121 
122   private static void handleField( // NOPMD false positive
123       @NonNull XmlObject obj,
124       Pair<IChoiceInstance, XmlModelContainer> state) {
125     IFieldInstanceAbsolute instance = new XmlFieldInstance(
126         (FieldReferenceType) obj,
127         ObjectUtils.notNull(state.getLeft()));
128     ObjectUtils.notNull(state.getRight()).append(instance);
129   }
130 
131   private static void handleDefineField( // NOPMD false positive
132       @NonNull XmlObject obj,
133       Pair<IChoiceInstance, XmlModelContainer> state) {
134     IFieldInstanceAbsolute instance = new XmlInlineFieldDefinition(
135         (InlineFieldDefinitionType) obj,
136         ObjectUtils.notNull(state.getLeft()));
137     ObjectUtils.notNull(state.getRight()).append(instance);
138   }
139 
140   private static void handleAssembly( // NOPMD false positive
141       @NonNull XmlObject obj,
142       Pair<IChoiceInstance, XmlModelContainer> state) {
143     IAssemblyInstanceAbsolute instance = new XmlAssemblyInstance(
144         (AssemblyReferenceType) obj,
145         ObjectUtils.notNull(state.getLeft()));
146     ObjectUtils.notNull(state.getRight()).append(instance);
147   }
148 
149   private static void handleDefineAssembly( // NOPMD false positive
150       @NonNull XmlObject obj,
151       Pair<IChoiceInstance, XmlModelContainer> state) {
152     IAssemblyInstanceAbsolute instance = new XmlInlineAssemblyDefinition(
153         (InlineAssemblyDefinitionType) obj,
154         ObjectUtils.notNull(state.getLeft()));
155     ObjectUtils.notNull(state.getRight()).append(instance);
156   }
157 
158   private static class XmlModelContainer
159       extends DefaultContainerModelSupport<
160           IModelInstanceAbsolute,
161           INamedModelInstanceAbsolute,
162           IFieldInstanceAbsolute,
163           IAssemblyInstanceAbsolute> {
164 
165     /**
166      * Parse a choice group XMLBeans object.
167      *
168      * @param xmlObject
169      *          the XMLBeans object
170      * @param parent
171      *          the parent Metaschema node, either an assembly definition or choice
172      */
173     public XmlModelContainer(
174         @NonNull ChoiceType xmlObject,
175         @NonNull IChoiceInstance parent) {
176       XML_MODEL_PARSER.parse(xmlObject, Pair.of(parent, this));
177     }
178 
179     public void append(@NonNull IFieldInstanceAbsolute instance) {
180       QName key = instance.getXmlQName();
181       getFieldInstanceMap().put(key, instance);
182       getNamedModelInstanceMap().put(key, instance);
183       getModelInstances().add(instance);
184     }
185 
186     public void append(@NonNull IAssemblyInstanceAbsolute instance) {
187       QName key = instance.getXmlQName();
188       getAssemblyInstanceMap().put(key, instance);
189       getNamedModelInstanceMap().put(key, instance);
190       getModelInstances().add(instance);
191     }
192   }

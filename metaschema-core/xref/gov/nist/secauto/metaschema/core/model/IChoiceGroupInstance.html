6   package gov.nist.secauto.metaschema.core.model;
7   
8   import gov.nist.secauto.metaschema.core.datatype.markup.MarkupMultiline;
9   
10  import java.util.Locale;
11  
12  import edu.umd.cs.findbugs.annotations.NonNull;
13  import edu.umd.cs.findbugs.annotations.Nullable;
14  
15  /**
16   * An Metaschema model instance representing a grouped set of objects consisting
17   * of heterogeneous object types.
18   */
19  public interface IChoiceGroupInstance
20      extends IModelInstanceAbsolute, IContainerModelGrouped {
21  
22    /**
23     * The default max-occurs value for a choice group. {@code -1} represents an
24     * unbounded occurance.
25     */
26    int DEFAULT_CHOICE_GROUP_GROUP_AS_MAX_OCCURS = -1;
27  
28    /**
29     * The default JSON property value used to identify the specific type of the
30     * object.
31     */
32    @NonNull
33    String DEFAULT_JSON_DISCRIMINATOR_PROPERTY_NAME = "object-type";
34  
35    /**
36     * {@inheritDoc}
37     *
38     * @see #DEFAULT_CHOICE_GROUP_GROUP_AS_MAX_OCCURS
39     */
40    @Override
41    default int getMaxOccurs() {
42      return DEFAULT_CHOICE_GROUP_GROUP_AS_MAX_OCCURS;
43    }
44  
45    /**
46     * Provides the Metaschema model type of "CHOICE".
47     *
48     * @return the model type
49     */
50    @Override
51    default ModelType getModelType() {
52      return ModelType.CHOICE_GROUP;
53    }
54  
55    /**
56     * Get the JSON property to use to discriminate between JSON objects.
57     *
58     * @return the discriminator property
59     * @see #DEFAULT_JSON_DISCRIMINATOR_PROPERTY_NAME
60     */
61    @NonNull
62    String getJsonDiscriminatorProperty();
63  
64    @Override
65    default boolean isEffectiveValueWrappedInXml() {
66      return true;
67    }
68  
69    /**
70     * Get the effective name of the JSON key flag, if a JSON key is configured.
71     * <p>
72     * This name is expected to be in the same namespace as the containing model
73     * element (i.e. choice group, assembly, field).
74     *
75     * @return the name of the JSON key flag if configured, or {@code null}
76     *         otherwise
77     */
78    @Nullable
79    String getJsonKeyFlagInstanceName();
80  
81    /**
82     * Get the named model instance for the provided choice group item.
83     *
84     * @param item
85     *          the item to get the instance for
86     * @return the named model instance for the provided choice group item
87     */
88    @NonNull
89    default INamedModelInstanceGrouped getItemInstance(@NonNull Object item) {
90      throw new UnsupportedOperationException("no value");
91    }
92  
93    @Override
94    default MarkupMultiline getRemarks() {
95      // no remarks
96      return null;
97    }
98  
99    @SuppressWarnings("null")
100   @Override
101   default String toCoordinates() {
102     return String.format("%s-instance:%s:%s/%s@%d",
103         getModelType().toString().toLowerCase(Locale.ROOT),
104         getContainingDefinition().getContainingModule().getShortName(),
105         getContainingDefinition().getName(),
106         getGroupAsName(),
107         hashCode());
108   }
109 }

6   package gov.nist.secauto.metaschema.core.model;
7   
8   import gov.nist.secauto.metaschema.core.util.ObjectUtils;
9   
10  import java.io.IOException;
11  import java.net.URI;
12  import java.util.ArrayList;
13  import java.util.Collection;
14  import java.util.Collections;
15  import java.util.Deque;
16  import java.util.LinkedHashMap;
17  import java.util.List;
18  import java.util.Map;
19  
20  import edu.umd.cs.findbugs.annotations.NonNull;
21  
22  /**
23   * Provides methods to load a Metaschema expressed in XML.
24   * <p>
25   * Loaded Metaschema instances are cached to avoid the need to load them for
26   * every use. Any Metaschema imported is also loaded and cached automatically.
27   *
28   * @param <T>
29   *          the Java type of the module binding
30   * @param <M>
31   *          the Java type of the Metaschema module loaded by this loader
32   */
33  public abstract class AbstractModuleLoader<T, M extends IModuleExtended<M, ?, ?, ?, ?>>
34      extends AbstractLoader<M>
35      implements IModuleLoader<M> {
36    /**
37     * Construct a new Metaschema module loader, which use the provided module post
38     * processors when loading a module.
39     */
40    protected AbstractModuleLoader() {
41      // only allow construction by extending classes
42    }
43  
44    /**
45     * Parse the {@code resource} based on the provided {@code xmlObject}.
46     *
47     * @param resource
48     *          the URI of the resource being parsed
49     * @param binding
50     *          the XML beans object to parse
51     * @param importedModules
52     *          previously parsed Metaschema modules imported by the provided
53     *          {@code resource}
54     * @return the parsed resource as a Metaschema module
55     * @throws MetaschemaException
56     *           if an error occurred while parsing the XML beans object
57     */
58    @NonNull
59    protected abstract M newModule(
60        @NonNull URI resource,
61        @NonNull T binding,
62        @NonNull List<? extends M> importedModules) throws MetaschemaException;
63  
64    /**
65     * Get the list of Metaschema module URIs associated with the provided binding.
66     *
67     * @param binding
68     *          the Metaschema module binding declaring the imports
69     * @return the list of Metaschema module URIs
70     */
71    @NonNull
72    protected abstract List<URI> getImports(@NonNull T binding);
73  
74    @Override
75    protected M parseResource(@NonNull URI resource, @NonNull Deque<URI> visitedResources)
76        throws IOException {
77      // parse this Metaschema module
78      T binding = parseModule(resource);
79  
80      // now check if this Metaschema imports other metaschema
81      List<URI> imports = getImports(binding);
82      @NonNull Map<URI, M> importedModules;
83      if (imports.isEmpty()) {
84        importedModules = ObjectUtils.notNull(Collections.emptyMap());
85      } else {
86        try {
87          importedModules = new LinkedHashMap<>();
88          for (URI importedResource : imports) {
89            URI resolvedResource = ObjectUtils.notNull(resource.resolve(importedResource));
90            importedModules.put(resolvedResource, loadInternal(resolvedResource, visitedResources));
91          }
92        } catch (MetaschemaException ex) {
93          throw new IOException(ex);
94        }
95      }
96  
97      // now create this metaschema
98      Collection<M> values = importedModules.values();
99      try {
100       return newModule(resource, binding, new ArrayList<>(values));
101     } catch (MetaschemaException ex) {
102       throw new IOException(ex);
103     }
104   }
105 
106   /**
107    * Parse the provided XML resource as a Metaschema module.
108    *
109    * @param resource
110    *          the resource to parse
111    * @return the XMLBeans representation of the Metaschema module
112    * @throws IOException
113    *           if a parsing error occurred
114    */
115   @NonNull
116   protected abstract T parseModule(@NonNull URI resource) throws IOException;
117 }

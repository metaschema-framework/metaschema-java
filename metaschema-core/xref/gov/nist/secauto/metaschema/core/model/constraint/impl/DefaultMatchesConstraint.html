6   package gov.nist.secauto.metaschema.core.model.constraint.impl;
7   
8   import gov.nist.secauto.metaschema.core.datatype.IDataTypeAdapter;
9   import gov.nist.secauto.metaschema.core.datatype.markup.MarkupLine;
10  import gov.nist.secauto.metaschema.core.datatype.markup.MarkupMultiline;
11  import gov.nist.secauto.metaschema.core.model.IAttributable;
12  import gov.nist.secauto.metaschema.core.model.ISource;
13  import gov.nist.secauto.metaschema.core.model.constraint.IMatchesConstraint;
14  
15  import java.util.Map;
16  import java.util.Set;
17  import java.util.regex.Pattern;
18  
19  import edu.umd.cs.findbugs.annotations.NonNull;
20  import edu.umd.cs.findbugs.annotations.Nullable;
21  
22  /**
23   * Represents a matches constraint.
24   * <p>
25   * Enforces a value pattern and/or data type.
26   */
27  public final class DefaultMatchesConstraint
28      extends AbstractConfigurableMessageConstraint
29      implements IMatchesConstraint {
30    private final Pattern pattern;
31    private final IDataTypeAdapter<?> dataType;
32  
33    /**
34     * Construct a new matches constraint.
35     *
36     * @param id
37     *          the optional identifier for the constraint
38     * @param formalName
39     *          the constraint's formal name or {@code null} if not provided
40     * @param description
41     *          the constraint's semantic description or {@code null} if not
42     *          provided
43     * @param source
44     *          information about the constraint source
45     * @param level
46     *          the significance of a violation of this constraint
47     * @param target
48     *          the Metapath expression identifying the nodes the constraint targets
49     * @param properties
50     *          a collection of associated properties
51     * @param pattern
52     *          the value pattern to match or {@code null} if there is no match
53     *          pattern
54     * @param dataType
55     *          the value data type to match or {@code null} if there is no match
56     *          data type
57     * @param message
58     *          an optional message to emit when the constraint is violated
59     * @param remarks
60     *          optional remarks describing the intent of the constraint
61     */
62    @SuppressWarnings("PMD.ExcessiveParameterList")
63    public DefaultMatchesConstraint(
64        @Nullable String id,
65        @Nullable String formalName,
66        @Nullable MarkupLine description,
67        @NonNull ISource source,
68        @NonNull Level level,
69        @NonNull String target,
70        @NonNull Map<IAttributable.Key, Set<String>> properties,
71        @Nullable Pattern pattern,
72        @Nullable IDataTypeAdapter<?> dataType,
73        @Nullable String message,
74        @Nullable MarkupMultiline remarks) {
75      super(id, formalName, description, source, level, target, properties, message, remarks);
76      if (pattern == null && dataType == null) {
77        throw new IllegalArgumentException("a pattern or data type must be provided");
78      }
79      this.pattern = pattern;
80      this.dataType = dataType;
81    }
82  
83    @Override
84    public Pattern getPattern() {
85      return pattern;
86    }
87  
88    @Override
89    public IDataTypeAdapter<?> getDataType() {
90      return dataType;
91    }
92  
93  }

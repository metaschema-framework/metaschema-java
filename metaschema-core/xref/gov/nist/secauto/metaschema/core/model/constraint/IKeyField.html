6   package gov.nist.secauto.metaschema.core.model.constraint;
7   
8   import gov.nist.secauto.metaschema.core.datatype.markup.MarkupMultiline;
9   import gov.nist.secauto.metaschema.core.metapath.MetapathExpression;
10  import gov.nist.secauto.metaschema.core.model.constraint.impl.DefaultKeyField;
11  
12  import java.util.regex.Pattern;
13  
14  import edu.umd.cs.findbugs.annotations.NonNull;
15  import edu.umd.cs.findbugs.annotations.Nullable;
16  
17  /**
18   * Represents a component of a key used in a key-based index.
19   * <p>
20   * A key field is targeted at a Metaschema field or flag data object's value. An
21   * optional pattern can be used to extract a portion of the value for use in
22   * generating an index key.
23   */
24  public interface IKeyField {
25    /**
26     * Construct a new key field based on the provided target. An optional pattern
27     * can be used to extract a portion of the resulting key value.
28     *
29     * @param target
30     *          a Metapath expression identifying the target of the key field
31     * @param pattern
32     *          an optional used to extract a portion of the resulting key value
33     * @param remarks
34     *          optional remarks describing the intent of the constraint
35     * @param source
36     *          the descriptor for the resource containing the constraint
37     * @return the new key field
38     */
39    @SuppressWarnings("PMD.ShortMethodName")
40    @NonNull
41    static IKeyField of(
42        @NonNull String target,
43        @Nullable Pattern pattern,
44        @Nullable MarkupMultiline remarks,
45        @NonNull ISource source) {
46      return new DefaultKeyField(target, pattern, remarks, source);
47    }
48  
49    /**
50     * Get the Metapath expression that identifies the node item whose value will be
51     * used as the key value.
52     *
53     * @return the Metapath expression identifying the key value target
54     */
55    @NonNull
56    String getTarget();
57  
58    /**
59     * Get the compiled Metapath expression for the {@link #getTarget()}.
60     *
61     * @return the compiled Metapath expression
62     */
63    @NonNull
64    MetapathExpression getTargetMetapath();
65  
66    /**
67     * A pattern to use to retrieve the value. If non-{@code null}, the first
68     * capturing group is used to retrieve the value.
69     *
70     * @return a pattern to use to get the value or {@code null} if the full value
71     *         is to be used
72     */
73    @Nullable
74    Pattern getPattern();
75  
76    /**
77     * Any remarks about the key field as markup text.
78     *
79     * @return markup text or {@code null} if no text is provided
80     */
81    @Nullable
82    MarkupMultiline getRemarks();
83  }

6   package gov.nist.secauto.metaschema.core.model.constraint.impl;
7   
8   import gov.nist.secauto.metaschema.core.model.IAssemblyDefinition;
9   import gov.nist.secauto.metaschema.core.model.IDefinition;
10  import gov.nist.secauto.metaschema.core.model.IFieldDefinition;
11  import gov.nist.secauto.metaschema.core.model.IFlagDefinition;
12  import gov.nist.secauto.metaschema.core.model.ISource;
13  import gov.nist.secauto.metaschema.core.model.constraint.AbstractTargetedConstraints;
14  import gov.nist.secauto.metaschema.core.model.constraint.ConstraintInitializationException;
15  import gov.nist.secauto.metaschema.core.model.constraint.IValueConstrained;
16  
17  import java.util.Locale;
18  
19  import edu.umd.cs.findbugs.annotations.NonNull;
20  
21  /**
22   * Provides an base implementation for a set of constraints that target a given
23   * definition using a target Metapath expression.
24   *
25   * @param <T>
26   *          the Java type of the definition target
27   *
28   * @param <S>
29   *          the Java type of the constraint container
30   */
31  public abstract class AbstractDefinitionTargetedConstraints<
32      T extends IDefinition,
33      S extends IValueConstrained>
34      extends AbstractTargetedConstraints<S> {
35  
36    /**
37     * Construct a new set of targeted constraints.
38     *
39     * @param source
40     *          information about the resource the constraints were sources from
41     * @param target
42     *          the Metapath expression that can be used to find matching targets
43     * @param constraints
44     *          the constraints to apply to matching targets
45     */
46    protected AbstractDefinitionTargetedConstraints(
47        @NonNull ISource source,
48        @NonNull String target,
49        @NonNull S constraints) {
50      super(source, target, constraints);
51    }
52  
53    /**
54     * Apply the constraints to the provided {@code definition}.
55     * <p>
56     * This will be called when a definition is found that matches the target
57     * expression.
58     *
59     * @param definition
60     *          the definition to apply the constraints to.
61     */
62    protected void applyTo(@NonNull T definition) {
63      getAllowedValuesConstraints().forEach(definition::addConstraint);
64      getMatchesConstraints().forEach(definition::addConstraint);
65      getIndexHasKeyConstraints().forEach(definition::addConstraint);
66      getExpectConstraints().forEach(definition::addConstraint);
67    }
68  
69    @Override
70    public void target(@NonNull IFlagDefinition definition) {
71      wrongDefinitionTypeTargeted(definition);
72    }
73  
74    @Override
75    public void target(@NonNull IFieldDefinition definition) {
76      wrongDefinitionTypeTargeted(definition);
77    }
78  
79    @Override
80    public void target(@NonNull IAssemblyDefinition definition) {
81      wrongDefinitionTypeTargeted(definition);
82    }
83  
84    private void wrongDefinitionTypeTargeted(@NonNull IDefinition definition) {
85      throw new ConstraintInitializationException(
86          String.format(
87              "The %s named '%s' from metaschema '%s' is incorrectly targeted by a set of" +
88                  " constraints with the target '%s'. Ensure the target expression is correct in '%s'.",
89              definition.getModelType().name().toLowerCase(Locale.ROOT),
90              definition.getEffectiveName(),
91              definition.getContainingModule().getQName().toString(),
92              getTargetExpression(),
93              getSource().getLocationHint()));
94    }
95  }

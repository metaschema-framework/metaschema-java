8   import gov.nist.secauto.metaschema.core.model.impl.DefaultContainerModelAssemblySupport;
9   
10  import java.util.List;
11  import java.util.Map;
12  
13  import edu.umd.cs.findbugs.annotations.NonNull;
14  
15  /**
16   * Common interface for model container support classes.
17   *
18   * @param <MI>
19   *          the model instance Java type
20   * @param <NMI>
21   *          the named model instance Java type
22   * @param <FI>
23   *          the field instance Java type
24   * @param <AI>
25   *          the assembly instance Java type
26   * @param <CI>
27   *          the choice instance Java type
28   * @param <CGI>
29   *          the choice group instance Java type
30   */
31  public interface IContainerModelAssemblySupport<
32      MI extends IModelInstance,
33      NMI extends INamedModelInstance,
34      FI extends IFieldInstance,
35      AI extends IAssemblyInstance,
36      CI extends IChoiceInstance,
37      CGI extends IChoiceGroupInstance> extends IContainerModelSupport<MI, NMI, FI, AI> {
38  
39    /**
40     * Get an empty, immutable container.
41     *
42     * @param <MI>
43     *          the model instance Java type
44     * @param <NMI>
45     *          the named model instance Java type
46     * @param <FI>
47     *          the field instance Java type
48     * @param <AI>
49     *          the assembly instance Java type
50     * @param <CI>
51     *          the choice instance Java type
52     * @param <CGI>
53     *          the choice group instance Java type
54     * @return the empty container
55     */
56    @SuppressWarnings("unchecked")
57    @NonNull
58    static <
59        MI extends IModelInstance,
60        NMI extends INamedModelInstance,
61        FI extends IFieldInstance,
62        AI extends IAssemblyInstance,
63        CI extends IChoiceInstance,
64        CGI extends IChoiceGroupInstance> IContainerModelAssemblySupport<MI, NMI, FI, AI, CI, CGI> empty() {
65      return DefaultContainerModelAssemblySupport.EMPTY;
66    }
67  
68    /**
69     * Get a listing of all choice instances.
70     *
71     * @return the listing
72     */
73    @NonNull
74    List<CI> getChoiceInstances();
75  
76    /**
77     * Get a listing of all choice group instances.
78     *
79     * @return the listing
80     */
81    @NonNull
82    Map<String, CGI> getChoiceGroupInstanceMap();
83  }

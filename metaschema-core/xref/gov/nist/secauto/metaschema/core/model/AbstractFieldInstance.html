6   package gov.nist.secauto.metaschema.core.model;
7   
8   import gov.nist.secauto.metaschema.core.util.ObjectUtils;
9   
10  import javax.xml.namespace.QName;
11  
12  import edu.umd.cs.findbugs.annotations.NonNull;
13  
14  /**
15   * A base class for a field that is a member of a containing model.
16   *
17   * @param <PARENT>
18   *          the Java type of the parent model (i.e., assembly, choice,
19   *          choiceGroup).
20   * @param <DEFINITION>
21   *          the Java type of the definition for this member field
22   * @param <INSTANCE>
23   *          the Java type of the instance for this member field
24   * @param <PARENT_DEFINITION>
25   *          the Java type of the containing assembly definition
26   */
27  public abstract class AbstractFieldInstance<
28      PARENT extends IContainerModel,
29      DEFINITION extends IFieldDefinition,
30      INSTANCE extends IFieldInstance,
31      PARENT_DEFINITION extends IAssemblyDefinition>
32      extends AbstractNamedModelInstance<PARENT, PARENT_DEFINITION>
33      implements IFieldInstance, IFeatureDefinitionReferenceInstance<DEFINITION, INSTANCE> {
34  
35    /**
36     * Construct a new field instance.
37     *
38     * @param parent
39     *          the parent model containing this instance
40     */
41    protected AbstractFieldInstance(@NonNull PARENT parent) {
42      super(parent);
43    }
44  
45    @Override
46    public DEFINITION getDefinition() {
47      QName qname = getReferencedDefinitionQName();
48      // this should always be not null
49      IFieldDefinition definition = getContainingModule().getScopedFieldDefinitionByName(qname);
50      if (definition == null) {
51        throw new IllegalStateException(
52            String.format("Unable to resolve field reference '%s' in definition '%s' in module '%s'",
53                qname,
54                getParentContainer().getOwningDefinition().getName(),
55                getContainingModule().getShortName()));
56      }
57      return ObjectUtils.asType(definition);
58    }
59  
60    /**
61     * Generates a "coordinate" string for the assembly instance.
62     *
63     * @return the coordinate
64     */
65    @SuppressWarnings("null")
66    @Override
67    public String toCoordinates() {
68      IDefinition definition = getDefinition();
69      return String.format("field instance %s -> %s in module %s (@%d(%d)",
70          getXmlQName(),
71          definition.getDefinitionQName(),
72          getContainingDefinition().getContainingModule().getShortName(),
73          hashCode(),
74          definition.hashCode());
75    }
76  }

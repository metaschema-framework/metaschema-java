6   package gov.nist.secauto.metaschema.core.datatype.adapter;
7   
8   import com.fasterxml.jackson.databind.jsonFormatVisitors.JsonFormatTypes;
9   
10  import gov.nist.secauto.metaschema.core.datatype.AbstractDataTypeAdapter;
11  import gov.nist.secauto.metaschema.core.metapath.MetapathConstants;
12  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IYearMonthDurationItem;
13  import gov.nist.secauto.metaschema.core.qname.EQNameFactory;
14  import gov.nist.secauto.metaschema.core.qname.IEnhancedQName;
15  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
16  
17  import java.time.Period;
18  import java.time.format.DateTimeParseException;
19  import java.util.List;
20  
21  import edu.umd.cs.findbugs.annotations.NonNull;
22  
23  /**
24   * Support for the Metaschema <a href=
25   * "https://pages.nist.gov/metaschema/specification/datatypes/#year-month-duration">year-month-duration</a>
26   * data type.
27   */
28  public class YearMonthAdapter
29      extends AbstractDataTypeAdapter<Period, IYearMonthDurationItem> {
30    @NonNull
31    private static final List<IEnhancedQName> NAMES = ObjectUtils.notNull(
32        List.of(
33            EQNameFactory.instance().newQName(MetapathConstants.NS_METAPATH, "year-month-duration")));
34  
35    YearMonthAdapter() {
36      super(Period.class, IYearMonthDurationItem.class, IYearMonthDurationItem::cast);
37    }
38  
39    @Override
40    public List<IEnhancedQName> getNames() {
41      return NAMES;
42    }
43  
44    @Override
45    public JsonFormatTypes getJsonRawType() {
46      return JsonFormatTypes.STRING;
47    }
48  
49    @Override
50    public Period copy(Object obj) {
51      // value in immutable
52      return (Period) obj;
53    }
54  
55    @SuppressWarnings("null")
56    @Override
57    public Period parse(String value) {
58      try {
59        return Period.parse(value);
60      } catch (DateTimeParseException ex) {
61        throw new IllegalArgumentException(ex.getLocalizedMessage(), ex);
62      }
63    }
64  
65    @Override
66    public IYearMonthDurationItem newItem(Object value) {
67      Period item = toValue(value);
68      return IYearMonthDurationItem.valueOf(item);
69    }
70  
71  }

6   package gov.nist.secauto.metaschema.core.datatype.object;
7   
8   import gov.nist.secauto.metaschema.core.datatype.AbstractCustomJavaDataType;
9   
10  import java.time.temporal.Temporal;
11  
12  import edu.umd.cs.findbugs.annotations.NonNull;
13  
14  /**
15   * Implementations of this class represent a temporal value which may not have a
16   * timezone making it ambiguous as a point/window in time.
17   * <p>
18   * Metaschema has a need to represent dates and times that allow for an
19   * ambiguous time zone. This is due to some models not requiring a time zone as
20   * part of a date/time. An ambiguous dateTime allows a time zone to be inferred,
21   * without change information in the source content.
22   * <p>
23   * This class wraps a ZonedDateTime object and tracks if a time zone was found
24   * when parsing, which can be used to ensure that the assumed time zone is not
25   * written back out in such cases.
26   *
27   * @param <TYPE>
28   *          the bound object type that extends this class, used for proper type
29   *          inheritance in implementing classes like {@code AmbiguousDate} or
30   *          {@code AmbiguousDateTime}
31   * @param <U>
32   *          the Java type of the temporal value
33   */
34  public abstract class AbstractAmbiguousTemporal<TYPE extends AbstractAmbiguousTemporal<TYPE, U>, U extends Temporal>
35      extends AbstractCustomJavaDataType<TYPE, U> {
36    private final boolean timeZone;
37  
38    /**
39     * Construct a new object. This type supports ambiguous dates/times that were
40     * provided without a time zone.
41     *
42     * @param value
43     *          the date value
44     * @param hasTimeZone
45     *          {@code true} if the date is intended to have an associated time zone
46     *          or {@code false} otherwise
47     */
48    public AbstractAmbiguousTemporal(@NonNull U value, boolean hasTimeZone) {
49      super(value);
50      this.timeZone = hasTimeZone;
51    }
52  
53    /**
54     * Indicate if a time zone is configured.
55     *
56     * @return {@code true} if the date is intended to have an associated time zone
57     *         or {@code false} otherwise
58     */
59    public boolean hasTimeZone() {
60      return timeZone;
61    }
62  
63    @Override
64    public String toString() {
65      return getValue().toString() + (hasTimeZone() ? "" : "(abiguous)");
66    }
67  
68  }

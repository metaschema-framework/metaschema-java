8   import gov.nist.secauto.metaschema.core.datatype.AbstractDataTypeProvider;
9   
10  import edu.umd.cs.findbugs.annotations.NonNull;
11  
12  /**
13   * Provides for runtime discovery of built-in implementations of the markup
14   * Metaschema data types.
15   */
16  public final class MarkupDataTypeProvider
17      extends AbstractDataTypeProvider {
18    @NonNull
19    public static final MarkupLineAdapter MARKUP_LINE = new MarkupLineAdapter();
20    @NonNull
21    public static final MarkupMultilineAdapter MARKUP_MULTILINE = new MarkupMultilineAdapter();
22  
23    /**
24     * Create the data type provider.
25     */
26    public MarkupDataTypeProvider() {
27      registerDatatype(MARKUP_LINE);
28      registerDatatype(MARKUP_MULTILINE);
29    }
30  }

6   package gov.nist.secauto.metaschema.core.datatype.adapter;
7   
8   import gov.nist.secauto.metaschema.core.metapath.MetapathConstants;
9   import gov.nist.secauto.metaschema.core.metapath.item.atomic.INcNameItem;
10  import gov.nist.secauto.metaschema.core.qname.EQNameFactory;
11  import gov.nist.secauto.metaschema.core.qname.IEnhancedQName;
12  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
13  
14  import java.util.List;
15  import java.util.regex.Pattern;
16  
17  import edu.umd.cs.findbugs.annotations.NonNull;
18  
19  /**
20   * Support for the Metaschema <a href=
21   * "https://pages.nist.gov/metaschema/specification/datatypes/#ncname">ncname</a>
22   * data type.
23   */
24  @Deprecated(since = "0.7.0")
25  public class NcNameAdapter
26      extends AbstractStringAdapter<INcNameItem> {
27    @NonNull
28    private static final List<IEnhancedQName> NAMES = ObjectUtils.notNull(
29        List.of(
30            EQNameFactory.instance().newQName(MetapathConstants.NS_METAPATH, "ncname"),
31            // for backwards compatibility with original type name
32            EQNameFactory.instance().newQName(MetapathConstants.NS_METAPATH, "NCName")));
33    private static final Pattern NCNAME = Pattern.compile(String.format("^(\\p{L}|_)(\\p{L}|\\p{N}|[.\\-_])*$"));
34  
35    /**
36     * Determine if the name is a non-colonized name.
37     *
38     * @param name
39     *          the name to test
40     * @return {@code true} if the name is not colonized, or {@code false} otherwise
41     */
42    public static boolean isNcName(@NonNull String name) {
43      return NCNAME.matcher(name).matches();
44    }
45  
46    NcNameAdapter() {
47      super(INcNameItem.class, INcNameItem::cast);
48      // avoid general construction
49    }
50  
51    @Override
52    public List<IEnhancedQName> getNames() {
53      return NAMES;
54    }
55  
56    @Override
57    public String parse(String value) {
58      if (!isNcName(value)) {
59        throw new IllegalArgumentException(String.format("Value '%s' is not a valid ncname.", value));
60      }
61      return super.parse(value);
62    }
63  
64    @Override
65    public INcNameItem newItem(Object value) {
66      String item = asString(value);
67      return INcNameItem.valueOf(item);
68    }
69  }

6   package gov.nist.secauto.metaschema.core.datatype.markup.flexmark;
7   
8   import com.vladsch.flexmark.ast.Paragraph;
9   import com.vladsch.flexmark.html.HtmlRenderer;
10  import com.vladsch.flexmark.html.HtmlWriter;
11  import com.vladsch.flexmark.html.renderer.NodeRenderer;
12  import com.vladsch.flexmark.html.renderer.NodeRendererContext;
13  import com.vladsch.flexmark.html.renderer.NodeRendererFactory;
14  import com.vladsch.flexmark.html.renderer.NodeRenderingHandler;
15  import com.vladsch.flexmark.util.data.DataHolder;
16  import com.vladsch.flexmark.util.data.MutableDataHolder;
17  
18  import java.util.Collections;
19  import java.util.Set;
20  
21  import edu.umd.cs.findbugs.annotations.NonNull;
22  import edu.umd.cs.findbugs.annotations.Nullable;
23  
24  public class SuppressPTagExtension
25      implements HtmlRenderer.HtmlRendererExtension {
26  
27    /**
28     * Construct a new extension instance.
29     *
30     * @return the instance
31     */
32    public static SuppressPTagExtension newInstance() {
33      return new SuppressPTagExtension();
34    }
35  
36    @Override
37    public void rendererOptions(MutableDataHolder options) {
38      // do nothing
39    }
40  
41    @Override
42    public void extend(HtmlRenderer.Builder rendererBuilder, String rendererType) {
43      rendererBuilder.nodeRendererFactory(new PTagNodeRenderer.Factory());
44    }
45  
46    static class PTagNodeRenderer implements NodeRenderer {
47  
48      @Override
49      public @Nullable
50      Set<NodeRenderingHandler<?>> getNodeRenderingHandlers() {
51        return Collections.singleton(
52            new NodeRenderingHandler<>(Paragraph.class, this::render));
53      }
54  
55      protected void render(
56          @NonNull Paragraph node,
57          @NonNull NodeRendererContext context,
58          @SuppressWarnings("unused") @NonNull HtmlWriter html) {
59        context.renderChildren(node);
60      }
61  
62      public static class Factory implements NodeRendererFactory {
63  
64        @Override
65        public NodeRenderer apply(DataHolder options) {
66          return new PTagNodeRenderer();
67        }
68  
69      }
70    }
71  
72  }

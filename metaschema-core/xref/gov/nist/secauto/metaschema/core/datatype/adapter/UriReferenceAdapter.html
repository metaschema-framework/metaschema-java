6   package gov.nist.secauto.metaschema.core.datatype.adapter;
7   
8   import com.fasterxml.jackson.databind.jsonFormatVisitors.JsonFormatTypes;
9   
10  import gov.nist.secauto.metaschema.core.datatype.AbstractDataTypeAdapter;
11  import gov.nist.secauto.metaschema.core.metapath.MetapathConstants;
12  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IUriReferenceItem;
13  import gov.nist.secauto.metaschema.core.qname.EQNameFactory;
14  import gov.nist.secauto.metaschema.core.qname.IEnhancedQName;
15  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
16  
17  import java.net.URI;
18  import java.util.List;
19  
20  import edu.umd.cs.findbugs.annotations.NonNull;
21  
22  /**
23   * Support for the Metaschema <a href=
24   * "https://pages.nist.gov/metaschema/specification/datatypes/#uri-reference">uri-reference</a>
25   * data type.
26   */
27  public class UriReferenceAdapter
28      extends AbstractDataTypeAdapter<URI, IUriReferenceItem> {
29    @NonNull
30    private static final List<IEnhancedQName> NAMES = ObjectUtils.notNull(
31        List.of(
32            EQNameFactory.instance().newQName(MetapathConstants.NS_METAPATH, "uri-reference")));
33  
34    UriReferenceAdapter() {
35      super(URI.class, IUriReferenceItem.class, IUriReferenceItem::cast);
36    }
37  
38    @Override
39    public List<IEnhancedQName> getNames() {
40      return NAMES;
41    }
42  
43    @Override
44    public JsonFormatTypes getJsonRawType() {
45      return JsonFormatTypes.STRING;
46    }
47  
48    @SuppressWarnings("null")
49    @Override
50    public URI parse(String value) {
51      return URI.create(value);
52    }
53  
54    @Override
55    public URI copy(Object obj) {
56      // a URI is immutable
57      return (URI) obj;
58    }
59  
60    @Override
61    public IUriReferenceItem newItem(Object value) {
62      URI item = toValue(value);
63      return IUriReferenceItem.valueOf(item);
64    }
65  }

6   package gov.nist.secauto.metaschema.core.datatype.adapter;
7   
8   import com.fasterxml.jackson.databind.jsonFormatVisitors.JsonFormatTypes;
9   
10  import gov.nist.secauto.metaschema.core.datatype.AbstractCustomJavaDataTypeAdapter;
11  import gov.nist.secauto.metaschema.core.datatype.object.AmbiguousDateTime;
12  import gov.nist.secauto.metaschema.core.metapath.MetapathConstants;
13  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IDateTimeItem;
14  import gov.nist.secauto.metaschema.core.qname.EQNameFactory;
15  import gov.nist.secauto.metaschema.core.qname.IEnhancedQName;
16  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
17  
18  import java.time.LocalDateTime;
19  import java.time.ZoneOffset;
20  import java.time.ZonedDateTime;
21  import java.time.format.DateTimeParseException;
22  import java.util.List;
23  
24  import edu.umd.cs.findbugs.annotations.NonNull;
25  
26  /**
27   * Support for the Metaschema <a href=
28   * "https://pages.nist.gov/metaschema/specification/datatypes/#date-time">date-time</a>
29   * data type.
30   */
31  public class DateTimeAdapter
32      extends AbstractCustomJavaDataTypeAdapter<AmbiguousDateTime, IDateTimeItem> {
33    @NonNull
34    private static final List<IEnhancedQName> NAMES = ObjectUtils.notNull(
35        List.of(
36            EQNameFactory.instance().newQName(MetapathConstants.NS_METAPATH, "date-time"),
37            // for backwards compatibility with original type name
38            EQNameFactory.instance().newQName(MetapathConstants.NS_METAPATH, "dateTime")));
39  
40    DateTimeAdapter() {
41      super(AmbiguousDateTime.class, IDateTimeItem.class, IDateTimeItem::cast);
42    }
43  
44    @Override
45    public List<IEnhancedQName> getNames() {
46      return NAMES;
47    }
48  
49    @Override
50    public JsonFormatTypes getJsonRawType() {
51      return JsonFormatTypes.STRING;
52    }
53  
54    @Override
55    public AmbiguousDateTime parse(String value) {
56      AmbiguousDateTime retval;
57      try {
58        retval = parseWithTimeZone(value);
59      } catch (DateTimeParseException ex) {
60        try {
61          retval = parseWithoutTimeZone(value);
62        } catch (DateTimeParseException ex2) {
63          IllegalArgumentException newEx = new IllegalArgumentException(ex2.getLocalizedMessage(), ex2);
64          newEx.addSuppressed(ex);
65          throw newEx;
66        }
67      }
68      return retval;
69    }
70  
71    @NonNull
72    private static AmbiguousDateTime parseWithTimeZone(@NonNull String value) {
73      return new AmbiguousDateTime(
74          ObjectUtils.notNull(ZonedDateTime.from(DateFormats.DATE_TIME_WITH_TZ.parse(value))),
75          true);
76    }
77  
78    @NonNull
79    private static AmbiguousDateTime parseWithoutTimeZone(@NonNull String value) {
80      LocalDateTime dateTime = LocalDateTime.from(DateFormats.DATE_TIME_WITH_OPTIONAL_TZ.parse(value));
81      return new AmbiguousDateTime(
82          ObjectUtils.notNull(ZonedDateTime.of(dateTime, ZoneOffset.UTC)),
83          false);
84    }
85  
86    @Override
87    public String asString(Object obj) {
88      AmbiguousDateTime value = toValue(obj);
89      return ObjectUtils.notNull(value.hasTimeZone()
90          ? DateFormats.DATE_TIME_WITH_TZ.format(value.getValue())
91          : DateFormats.DATE_TIME_WITH_OPTIONAL_TZ.format(value.getValue()));
92    }
93  
94    @Override
95    public IDateTimeItem newItem(Object value) {
96      AmbiguousDateTime item = toValue(value);
97      return IDateTimeItem.valueOf(item);
98    }
99  }

6   package gov.nist.secauto.metaschema.core.datatype.adapter;
7   
8   import com.fasterxml.jackson.databind.jsonFormatVisitors.JsonFormatTypes;
9   
10  import gov.nist.secauto.metaschema.core.datatype.AbstractDataTypeAdapter;
11  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IStringItem;
12  
13  /**
14   * Provides a common base class for string-based data types.
15   *
16   * @param <ITEM_TYPE>
17   *          the Metapath item type supported by the adapter
18   */
19  public abstract class AbstractStringAdapter<ITEM_TYPE extends IStringItem>
20      extends AbstractDataTypeAdapter<String, ITEM_TYPE> {
21  
22    /**
23     * Construct a new string-based adapter.
24     */
25    protected AbstractStringAdapter() {
26      super(String.class);
27    }
28  
29    @Override
30    public JsonFormatTypes getJsonRawType() {
31      return JsonFormatTypes.STRING;
32    }
33  
34    @Override
35    public String parse(String value) {
36      return value;
37    }
38  
39    @Override
40    public String copy(Object obj) {
41      // a Java string is immutable
42      return (String) obj;
43    }
44  }

35  public class HtmlCodeRenderExtension
36      implements HtmlRenderer.HtmlRendererExtension {
37    private static final Pattern EOL_PATTERN = Pattern.compile("\r\n|\r|\n");
38  
39    /**
40     * Construct a new extension instance.
41     *
42     * @return the instance
43     */
44    public static HtmlCodeRenderExtension newInstance() {
45      return new HtmlCodeRenderExtension();
46    }
47  
48    @Override
49    public void rendererOptions(MutableDataHolder options) {
50      // do nothing
51    }
52  
53    @Override
54    public void extend(HtmlRenderer.Builder rendererBuilder, String rendererType) {
55      rendererBuilder.nodeRendererFactory(new CodeNodeHtmlRenderer.Factory());
56    }
57  
58    static final class CodeNodeHtmlRenderer implements NodeRenderer {
59      private final boolean codeSoftLineBreaks;
60  
61      private CodeNodeHtmlRenderer(DataHolder options) {
62        codeSoftLineBreaks = Parser.CODE_SOFT_LINE_BREAKS.get(options);
63      }
64  
65      @Override
66      @Nullable
67      public Set<NodeRenderingHandler<?>> getNodeRenderingHandlers() {
68        return Collections.singleton(
69            new NodeRenderingHandler<>(Code.class, this::render));
70      }
71  
72      @SuppressFBWarnings(value = "RV_RETURN_VALUE_IGNORED_NO_SIDE_EFFECT", justification = "false positive")
73      private void render( // NOPMD actually used in lambda
74          @NonNull Code node,
75          @NonNull NodeRendererContext context,
76          @NonNull HtmlWriter html) {
77        HtmlRendererOptions htmlOptions = context.getHtmlOptions();
78  
79        boolean customTag = htmlOptions.codeStyleHtmlOpen != null || htmlOptions.codeStyleHtmlClose != null;
80        if (customTag) {
81          html.raw(ObjectUtils.notNull(htmlOptions.codeStyleHtmlOpen));
82        } else {
83          if (context.getHtmlOptions().sourcePositionParagraphLines) {
84            html.withAttr().tag("code");
85          } else {
86            html.srcPos(node.getText()).withAttr().tag("code");
87          }
88        }
89  
90        if (codeSoftLineBreaks && !htmlOptions.isSoftBreakAllSpaces) {
91          for (Node child : node.getChildren()) {
92            if (child instanceof Text) {
93              html.text(Escaping.collapseWhitespace(child.getChars(), false));
94            } else {
95              context.render(child);
96            }
97          }
98        } else {
99          String text = EOL_PATTERN.matcher(node.getText()).replaceAll(" ");
100         if (!text.isBlank() && SequenceUtils.startsWithWhitespace(text) && SequenceUtils.endsWithWhitespace(text)) {
101           html.text(SequenceUtils.subSequence(text, Range.of(1, text.length() - 1)));
102         } else {
103           html.raw(Escaping.escapeHtml(text, false));
104         }
105         // html.text(Escaping.collapseWhitespace(node.getText(), true));
106       }
107 
108       if (customTag) {
109         html.raw(ObjectUtils.notNull(htmlOptions.codeStyleHtmlClose));
110       } else {
111         html.tag("/code");
112       }
113     }
114 
115     public static class Factory implements NodeRendererFactory {
116 
117       @Override
118       public NodeRenderer apply(DataHolder options) {
119         return new CodeNodeHtmlRenderer(options);
120       }
121 
122     }
123   }

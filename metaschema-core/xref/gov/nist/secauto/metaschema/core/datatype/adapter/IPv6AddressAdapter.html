6   package gov.nist.secauto.metaschema.core.datatype.adapter;
7   
8   import com.fasterxml.jackson.databind.jsonFormatVisitors.JsonFormatTypes;
9   
10  import gov.nist.secauto.metaschema.core.datatype.AbstractDataTypeAdapter;
11  import gov.nist.secauto.metaschema.core.metapath.MetapathConstants;
12  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IIPv6AddressItem;
13  import gov.nist.secauto.metaschema.core.qname.EQNameFactory;
14  import gov.nist.secauto.metaschema.core.qname.IEnhancedQName;
15  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
16  
17  import java.util.List;
18  
19  import edu.umd.cs.findbugs.annotations.NonNull;
20  import inet.ipaddr.AddressStringException;
21  import inet.ipaddr.IPAddressString;
22  import inet.ipaddr.IPAddressStringParameters;
23  import inet.ipaddr.IncompatibleAddressException;
24  import inet.ipaddr.ipv6.IPv6Address;
25  
26  /**
27   * Support for the Metaschema <a href=
28   * "https://pages.nist.gov/metaschema/specification/datatypes/#ip-v6-address">ip-v6-address</a>
29   * data type.
30   */
31  public class IPv6AddressAdapter
32      extends AbstractDataTypeAdapter<IPv6Address, IIPv6AddressItem> {
33    @NonNull
34    private static final List<IEnhancedQName> NAMES = ObjectUtils.notNull(
35        List.of(
36            EQNameFactory.instance().newQName(MetapathConstants.NS_METAPATH, "ip-v6-address")));
37    private static final IPAddressStringParameters IP_V_6;
38  
39    static {
40      IP_V_6 = new IPAddressStringParameters.Builder().allowIPv4(false).allowEmpty(false).allowSingleSegment(false)
41          .allowWildcardedSeparator(false).getIPv6AddressParametersBuilder().allowBinary(false)
42          .allowPrefixesBeyondAddressSize(false).getParentBuilder().toParams();
43    }
44  
45    IPv6AddressAdapter() {
46      super(IPv6Address.class, IIPv6AddressItem.class, IIPv6AddressItem::cast);
47    }
48  
49    @Override
50    public List<IEnhancedQName> getNames() {
51      return NAMES;
52    }
53  
54    @Override
55    public JsonFormatTypes getJsonRawType() {
56      return JsonFormatTypes.STRING;
57    }
58  
59    @SuppressWarnings("null")
60    @Override
61    public IPv6Address parse(String value) {
62      try {
63        return (IPv6Address) new IPAddressString(value, IP_V_6).toAddress();
64      } catch (AddressStringException | IncompatibleAddressException ex) {
65        throw new IllegalArgumentException(ex.getLocalizedMessage(), ex);
66      }
67    }
68  
69    @Override
70    public IPv6Address copy(Object obj) {
71      // value is immutable
72      return (IPv6Address) obj;
73    }
74  
75    @Override
76    public IIPv6AddressItem newItem(Object value) {
77      IPv6Address item = toValue(value);
78      return IIPv6AddressItem.valueOf(item);
79    }
80  }

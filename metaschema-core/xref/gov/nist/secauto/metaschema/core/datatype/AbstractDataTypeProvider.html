6   package gov.nist.secauto.metaschema.core.datatype;
7   
8   import gov.nist.secauto.metaschema.core.datatype.adapter.MetaschemaDataTypeProvider;
9   import gov.nist.secauto.metaschema.core.util.CollectionUtil;
10  
11  import java.util.ArrayList;
12  import java.util.LinkedList;
13  import java.util.List;
14  import java.util.concurrent.locks.Lock;
15  import java.util.concurrent.locks.ReadWriteLock;
16  import java.util.concurrent.locks.ReentrantReadWriteLock;
17  
18  import edu.umd.cs.findbugs.annotations.NonNull;
19  
20  /**
21   * A base implementation of an {@link IDataTypeProvider}, supporting dynamic
22   * loading of Metaschema data type extensions at runtime.
23   * <p>
24   * The {@link MetaschemaDataTypeProvider} class provides an example of how to
25   * use this class to provide new data types.
26   */
27  public abstract class AbstractDataTypeProvider implements IDataTypeProvider {
28    @NonNull
29    private final List<IDataTypeAdapter<?>> library = new LinkedList<>();
30    @NonNull
31    private final ReadWriteLock libraryLock = new ReentrantReadWriteLock();
32  
33    @Override
34    public List<? extends IDataTypeAdapter<?>> getJavaTypeAdapters() {
35      Lock readLock = libraryLock.readLock();
36      readLock.lock();
37      try {
38        // make a defensive copy to protect callers from potential modifications
39        return CollectionUtil.unmodifiableList(new ArrayList<>(library));
40      } finally {
41        readLock.unlock();
42      }
43    }
44  
45    /**
46     * Register the provided {@code adapter} with the type system.
47     *
48     * @param adapter
49     *          the adapter to register
50     * @throws IllegalArgumentException
51     *           if another type adapter has no name
52     */
53    protected void registerDatatype(@NonNull IDataTypeAdapter<?> adapter) {
54      if (adapter.getNames().isEmpty()) {
55        throw new IllegalArgumentException("The adapter has no name: " + adapter.getClass().getName());
56      }
57      Lock writeLock = libraryLock.writeLock();
58      writeLock.lock();
59      try {
60        library.add(adapter);
61      } finally {
62        writeLock.unlock();
63      }
64    }
65  }

6   package gov.nist.secauto.metaschema.core.datatype;
7   
8   import gov.nist.secauto.metaschema.core.datatype.adapter.MetaschemaDataTypeProvider;
9   import gov.nist.secauto.metaschema.core.metapath.type.IAtomicOrUnionType;
10  import gov.nist.secauto.metaschema.core.util.CollectionUtil;
11  
12  import java.util.ArrayList;
13  import java.util.LinkedList;
14  import java.util.List;
15  import java.util.concurrent.locks.Lock;
16  import java.util.concurrent.locks.ReadWriteLock;
17  import java.util.concurrent.locks.ReentrantReadWriteLock;
18  
19  import edu.umd.cs.findbugs.annotations.NonNull;
20  
21  /**
22   * A base implementation of an {@link IDataTypeProvider}, supporting dynamic
23   * loading of Metaschema data type extensions at runtime.
24   * <p>
25   * The {@link MetaschemaDataTypeProvider} class provides an example of how to
26   * use this class to provide new data types.
27   */
28  public abstract class AbstractDataTypeProvider implements IDataTypeProvider {
29    @NonNull
30    private final List<IAtomicOrUnionType<?>> abstractTypes = new LinkedList<>();
31    @NonNull
32    private final List<IDataTypeAdapter<?>> library = new LinkedList<>();
33    @NonNull
34    private final ReadWriteLock libraryLock = new ReentrantReadWriteLock();
35  
36    @Override
37    public List<? extends IAtomicOrUnionType<?>> getAbstractTypes() {
38      Lock readLock = libraryLock.readLock();
39      readLock.lock();
40      try {
41        return CollectionUtil.unmodifiableList(new ArrayList<>(abstractTypes));
42      } finally {
43        readLock.unlock();
44      }
45    }
46  
47    @Override
48    public List<? extends IDataTypeAdapter<?>> getJavaTypeAdapters() {
49      Lock readLock = libraryLock.readLock();
50      readLock.lock();
51      try {
52        // make a defensive copy to protect callers from potential modifications
53        return CollectionUtil.unmodifiableList(new ArrayList<>(library));
54      } finally {
55        readLock.unlock();
56      }
57    }
58  
59    /**
60     * Register the provided abstract {@link IAtomicOrUnionType} with the type
61     * system.
62     *
63     * @param type
64     *          the abstract type to register
65     * @throws IllegalArgumentException
66     *           if the type is not abstract
67     */
68    protected void register(@NonNull IAtomicOrUnionType<?> type) {
69      if (type.getAdapter() != null) {
70        throw new IllegalArgumentException(
71            String.format("The type '%s' has an adapter and must be registered using the adapter.",
72                type.toSignature()));
73      }
74      Lock writeLock = libraryLock.writeLock();
75      writeLock.lock();
76      try {
77        abstractTypes.add(type);
78      } finally {
79        writeLock.unlock();
80      }
81    }
82  
83    /**
84     * Register the provided {@link IDataTypeAdapter} with the type system.
85     *
86     * @param adapter
87     *          the adapter to register
88     * @throws IllegalArgumentException
89     *           if another type adapter has no name
90     */
91    protected void register(@NonNull IDataTypeAdapter<?> adapter) {
92      if (adapter.getNames().isEmpty()) {
93        throw new IllegalArgumentException("The adapter has no name: " + adapter.getClass().getName());
94      }
95      Lock writeLock = libraryLock.writeLock();
96      writeLock.lock();
97      try {
98        library.add(adapter);
99      } finally {
100       writeLock.unlock();
101     }
102   }

6   package gov.nist.secauto.metaschema.core.datatype.adapter;
7   
8   import com.fasterxml.jackson.databind.jsonFormatVisitors.JsonFormatTypes;
9   
10  import gov.nist.secauto.metaschema.core.datatype.AbstractDataTypeAdapter;
11  import gov.nist.secauto.metaschema.core.metapath.MetapathConstants;
12  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IIPv4AddressItem;
13  import gov.nist.secauto.metaschema.core.qname.EQNameFactory;
14  import gov.nist.secauto.metaschema.core.qname.IEnhancedQName;
15  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
16  
17  import java.util.List;
18  
19  import edu.umd.cs.findbugs.annotations.NonNull;
20  import inet.ipaddr.AddressStringException;
21  import inet.ipaddr.IPAddressString;
22  import inet.ipaddr.IPAddressStringParameters;
23  import inet.ipaddr.IncompatibleAddressException;
24  import inet.ipaddr.ipv4.IPv4Address;
25  
26  /**
27   * Support for the Metaschema <a href=
28   * "https://pages.nist.gov/metaschema/specification/datatypes/#ip-v4-address">ip-v4-address</a>
29   * data type.
30   */
31  public class IPv4AddressAdapter
32      extends AbstractDataTypeAdapter<IPv4Address, IIPv4AddressItem> {
33    @NonNull
34    private static final List<IEnhancedQName> NAMES = ObjectUtils.notNull(
35        List.of(
36            EQNameFactory.instance().newQName(MetapathConstants.NS_METAPATH, "ip-v4-address")));
37    private static final IPAddressStringParameters IP_V_4;
38  
39    static {
40      IP_V_4 = new IPAddressStringParameters.Builder()
41          .allowIPv6(false)
42          .allowEmpty(false)
43          .allowSingleSegment(false)
44          .allowWildcardedSeparator(false)
45          .getIPv4AddressParametersBuilder()
46          .allowBinary(false)
47          .allowLeadingZeros(false)
48          .allowPrefixesBeyondAddressSize(false)
49          .getParentBuilder()
50          .toParams();
51    }
52  
53    IPv4AddressAdapter() {
54      super(IPv4Address.class, IIPv4AddressItem.class, IIPv4AddressItem::cast);
55    }
56  
57    @Override
58    public List<IEnhancedQName> getNames() {
59      return NAMES;
60    }
61  
62    @Override
63    public JsonFormatTypes getJsonRawType() {
64      return JsonFormatTypes.STRING;
65    }
66  
67    @SuppressWarnings("null")
68    @Override
69    public IPv4Address parse(String value) {
70      try {
71        return (IPv4Address) new IPAddressString(value, IP_V_4).toAddress();
72      } catch (AddressStringException | IncompatibleAddressException ex) {
73        throw new IllegalArgumentException(ex.getLocalizedMessage(), ex);
74      }
75    }
76  
77    @Override
78    public IPv4Address copy(Object obj) {
79      // value is immutable
80      return (IPv4Address) obj;
81    }
82  
83    @Override
84    public IIPv4AddressItem newItem(Object value) {
85      IPv4Address item = toValue(value);
86      return IIPv4AddressItem.valueOf(item);
87    }
88  }

6   package gov.nist.secauto.metaschema.core.datatype.adapter;
7   
8   import com.fasterxml.jackson.databind.jsonFormatVisitors.JsonFormatTypes;
9   
10  import gov.nist.secauto.metaschema.core.datatype.AbstractDataTypeAdapter;
11  import gov.nist.secauto.metaschema.core.metapath.MetapathConstants;
12  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IUuidItem;
13  import gov.nist.secauto.metaschema.core.qname.EQNameFactory;
14  import gov.nist.secauto.metaschema.core.qname.IEnhancedQName;
15  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
16  
17  import java.util.List;
18  import java.util.UUID;
19  import java.util.regex.Pattern;
20  
21  import edu.umd.cs.findbugs.annotations.NonNull;
22  
23  /**
24   * Support for the Metaschema <a href=
25   * "https://pages.nist.gov/metaschema/specification/datatypes/#uuid">uuid</a>
26   * data type.
27   */
28  public class UuidAdapter
29      extends AbstractDataTypeAdapter<UUID, IUuidItem> {
30    @NonNull
31    private static final List<IEnhancedQName> NAMES = ObjectUtils.notNull(
32        List.of(
33            EQNameFactory.instance().newQName(MetapathConstants.NS_METAPATH, "uuid")));
34  
35    /**
36     * A regular expression that matches a valid UUID.
37     */
38    public static final Pattern UUID_PATTERN
39        = Pattern.compile("^[0-9A-Fa-f]{8}-[0-9A-Fa-f]{4}-[45][0-9A-Fa-f]{3}-[89ABab][0-9A-Fa-f]{3}-[0-9A-Fa-f]{12}$");
40  
41    UuidAdapter() {
42      super(UUID.class, IUuidItem.class, IUuidItem::cast);
43    }
44  
45    @Override
46    public List<IEnhancedQName> getNames() {
47      return NAMES;
48    }
49  
50    @Override
51    public JsonFormatTypes getJsonRawType() {
52      return JsonFormatTypes.STRING;
53    }
54  
55    @SuppressWarnings("null")
56    @Override
57    public UUID parse(String value) {
58      try {
59        return UUID.fromString(value);
60      } catch (IllegalArgumentException ex) {
61        throw new IllegalArgumentException(
62            String.format("Value '%s' is not a valid UUID.", value),
63            ex);
64      }
65    }
66  
67    @Override
68    public UUID copy(Object obj) {
69      // a UUID is immutable
70      return (UUID) obj;
71    }
72  
73    @Override
74    public IUuidItem newItem(Object value) {
75      UUID item = toValue(value);
76      return IUuidItem.valueOf(item);
77    }
78  }

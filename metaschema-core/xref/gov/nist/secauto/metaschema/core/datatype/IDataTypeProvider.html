6   package gov.nist.secauto.metaschema.core.datatype;
7   
8   import gov.nist.secauto.metaschema.core.metapath.type.IAtomicOrUnionType;
9   
10  import java.util.List;
11  
12  import edu.umd.cs.findbugs.annotations.NonNull;
13  
14  /**
15   * A service interface used to provide implementations of data types represented
16   * as {@link IDataTypeAdapter} instances.
17   * <p>
18   * Multiple providers can be used to support dynamic data type discovery in an
19   * Metaschema-based application, allowing data type extensions to be loaded at
20   * runtime using the {@link DataTypeService}.
21   */
22  public interface IDataTypeProvider {
23    /**
24     * Get the type information for abstract item types that do not have an
25     * associated data type adpater.
26     *
27     * @return the abstract item types provided
28     */
29    List<? extends IAtomicOrUnionType<?>> getAbstractTypes();
30  
31    /**
32     * Get the {@link IDataTypeAdapter} instances associated with this provider.
33     *
34     * @return the sequence of adapters in match priority order
35     */
36    @NonNull
37    List<? extends IDataTypeAdapter<?>> getJavaTypeAdapters();
38  }

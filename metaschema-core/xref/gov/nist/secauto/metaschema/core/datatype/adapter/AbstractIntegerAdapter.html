6   package gov.nist.secauto.metaschema.core.datatype.adapter;
7   
8   import com.fasterxml.jackson.core.JsonGenerator;
9   import com.fasterxml.jackson.databind.jsonFormatVisitors.JsonFormatTypes;
10  
11  import gov.nist.secauto.metaschema.core.datatype.AbstractDataTypeAdapter;
12  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IIntegerItem;
13  
14  import java.io.IOException;
15  import java.math.BigInteger;
16  
17  /**
18   * Provides a common base class for integer-based data types.
19   * <p>
20   * An underlying {@link BigInteger} is used to support arbitrary sized integers.
21   *
22   * @param <ITEM_TYPE>
23   *          the Metapath item type supported by the adapter
24   */
25  public abstract class AbstractIntegerAdapter<ITEM_TYPE extends IIntegerItem>
26      extends AbstractDataTypeAdapter<BigInteger, ITEM_TYPE> {
27  
28    /**
29     * Construct a new integer-based adapter.
30     */
31    protected AbstractIntegerAdapter() {
32      super(BigInteger.class);
33    }
34  
35    @Override
36    public JsonFormatTypes getJsonRawType() {
37      return JsonFormatTypes.INTEGER;
38    }
39  
40    @Override
41    public BigInteger parse(String value) {
42      return new BigInteger(value);
43    }
44  
45    @Override
46    public void writeJsonValue(Object value, JsonGenerator generator) throws IOException {
47      try {
48        generator.writeNumber((BigInteger) value);
49      } catch (ClassCastException ex) {
50        throw new IOException(ex);
51      }
52    }
53  
54    @Override
55    public BigInteger copy(Object obj) {
56      // a BigInteger is immutable
57      return (BigInteger) obj;
58    }
59  }

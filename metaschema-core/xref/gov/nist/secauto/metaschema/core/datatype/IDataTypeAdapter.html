7   
8   import com.fasterxml.jackson.core.JsonGenerator;
9   import com.fasterxml.jackson.core.JsonParser;
10  import com.fasterxml.jackson.databind.jsonFormatVisitors.JsonFormatTypes;
11  
12  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IAnyAtomicItem;
13  import gov.nist.secauto.metaschema.core.metapath.type.IAtomicOrUnionType;
14  import gov.nist.secauto.metaschema.core.qname.IEnhancedQName;
15  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
16  
17  import org.codehaus.stax2.XMLEventReader2;
18  import org.codehaus.stax2.XMLStreamWriter2;
19  import org.codehaus.stax2.evt.XMLEventFactory2;
20  
21  import java.io.IOException;
22  import java.lang.reflect.Field;
23  import java.net.URI;
24  import java.util.List;
25  
26  import javax.xml.namespace.QName;
27  import javax.xml.stream.XMLEventWriter;
28  import javax.xml.stream.events.StartElement;
29  import javax.xml.stream.events.XMLEvent;
30  
31  import edu.umd.cs.findbugs.annotations.NonNull;
32  
33  /**
34   * Represents a data type implementation.
35   *
36   * @param <TYPE>
37   *          the Java type of the underlying data value
38   */
39  public interface IDataTypeAdapter<TYPE> {
40  
41    /**
42     * Get the metaschema type names associated with this adapter. This name must be
43     * unique with respect to all other metaschema types.
44     * <p>
45     * At least one name must be provided, with the first name being the most
46     * preferred name.
47     *
48     * @return the name
49     */
50    @NonNull
51    List<IEnhancedQName> getNames();
52  
53    /**
54     * Get the most preferred name for this data type.
55     *
56     * @return the name
57     */
58    @NonNull
59    default IEnhancedQName getPreferredName() {
60      return ObjectUtils.notNull(getNames().iterator().next());
61    }
62  
63    /**
64     * The JSON primative type of the data type.
65     *
66     * @return the JSON data type
67     */
68    JsonFormatTypes getJsonRawType();
69  
70    /**
71     * Get the Java class supported by this adapter.
72     *
73     * @return the Java class
74     */
75    @NonNull
76    Class<TYPE> getJavaClass();
77  
78    /**
79     * Casts the provided value to the type associated with this adapter.
80     *
81     * @param value
82     *          a value of the provided type
83     * @return the typed value
84     */
85    @NonNull
86    TYPE toValue(@NonNull Object value);
87  
88    /**
89     * Gets the value as a string suitable for writing as text. This is intended for
90     * data types that have a simple string-based structure in XML and JSON, such as
91     * for XML attributes or JSON keys. An adapter for a complex data structures
92     * that consist of XML elements will throw an
93     * {@link UnsupportedOperationException} when this is called.
94     *
95     * @param value
96     *          the data to formatted as a string
97     * @return a string
98     * @throws IllegalArgumentException
99     *           if the data type cannot be represented as a string
100    */
101   @NonNull
102   String asString(@NonNull Object value);
103 
104   /**
105    * Create a copy of the provided value.
106    *
107    * @param obj
108    *          the value to copy
109    * @return the copy
110    */
111   @NonNull
112   TYPE copy(@NonNull Object obj);
113 
114   /**
115    * Determines if the data type is an atomic, scalar value. Complex structures
116    * such as Markup are not considered atomic.
117    *
118    * @return {@code true} if the data type is an atomic scalar value, or
119    *         {@code false} otherwise
120    */
121   default boolean isAtomic() {
122     return true;
123   }
124 
125   /**
126    * Get the item type information of the associated item.
127    *
128    * @return the item type information
129    */
130   @NonNull
131   IAtomicOrUnionType<?> getItemType();
132 
133   /**
134    * Construct a new item of this type using the provided value.
135    *
136    * @param value
137    *          the item's value
138    * @return a new item
139    */
140   @NonNull
141   IAnyAtomicItem newItem(@NonNull Object value);
142 
143   /**
144    * Determines if adapter can parse the next element. The next element's
145    * {@link QName} is provided for this purpose.
146    * <p>
147    * This will be called when the parser encounter's an element it does not
148    * recognize. This gives the adapter a chance to request parsing of the data.
149    *
150    * @param nextElementQName
151    *          the next element's namespace-qualified name
152    * @return {@code true} if the adapter will parse the element, or {@code false}
153    *         otherwise
154    */
155   boolean canHandleQName(@NonNull IEnhancedQName nextElementQName);
156 
157   /**
158    * Parses a provided string. Used to parse XML attributes, simple XML character
159    * data, and JSON/YAML property values.
160    *
161    * @param value
162    *          the string value to parse
163    * @return the parsed data as the adapter's type
164    * @throws IllegalArgumentException
165    *           if the data is not valid to the data type
166    */
167   @NonNull
168   TYPE parse(@NonNull String value);
169 
170   /**
171    * This method is expected to parse content starting at the next event. Parsing
172    * will continue until the next event represents content that is not handled by
173    * this adapter. This means the event stream should be positioned after any
174    * {@link XMLEvent#END_ELEMENT} that corresponds to an
175    * {@link XMLEvent#START_ELEMENT} parsed by this adapter.
176    * <p>
177    * If this method parses the {@link XMLEvent#START_ELEMENT} for the element that
178    * contains the value data, then this method must also parse the corresponding
179    * {@link XMLEvent#END_ELEMENT}. Otherwise, the first event to parse will be the
180    * value data.
181    * <p>
182    * The value data is expected to be parsed completely, leaving the event stream
183    * on a peeked event corresponding to content that is not handled by this
184    * method.
185    *
186    * @param eventReader
187    *          the XML parser used to read the parsed value
188    * @param resource
189    *          the resource being parsed
190    * @return the parsed value
191    * @throws IOException
192    *           if a parsing error occurs
193    */
194   // TODO: migrate code to XML parser implementation.
195   @NonNull
196   TYPE parse(@NonNull XMLEventReader2 eventReader, @NonNull URI resource) throws IOException;
197 
198   /**
199    * Parses a JSON property value.
200    *
201    * @param parser
202    *          the JSON parser used to read the parsed value
203    * @param resource
204    *          the resource being parsed
205    * @return the parsed value
206    * @throws IOException
207    *           if a parsing error occurs
208    */
209   // TODO: migrate code to JSON parser implementation.
210   @NonNull
211   TYPE parse(@NonNull JsonParser parser, @NonNull URI resource) throws IOException;
212 
213   /**
214    * Writes the provided Java class instance data as XML. The parent element
215    * information is provided as a {@link StartElement} event, which allows
216    * namespace information to be obtained from the parent element using the
217    * {@link StartElement#getName()} and {@link StartElement#getNamespaceContext()}
218    * methods, which can be used when writing the provided instance value.
219    *
220    * @param instance
221    *          the {@link Field} instance value to write
222    * @param parent
223    *          the {@link StartElement} XML event that is the parent of the data to
224    *          write
225    * @param eventFactory
226    *          the XML event factory used to generate XML writing events
227    * @param eventWriter
228    *          the XML writer used to output XML as events
229    * @throws IOException
230    *           if an unexpected error occurred while writing to the output stream
231    */
232   // TODO: migrate code to XML writer implementation.
233   void writeXmlValue(
234       @NonNull Object instance,
235       @NonNull StartElement parent,
236       @NonNull XMLEventFactory2 eventFactory,
237       @NonNull XMLEventWriter eventWriter) throws IOException;
238 
239   /**
240    * Writes the provided Java class instance data as XML. The parent element
241    * information is provided as an XML {@link QName}, which allows namespace
242    * information to be obtained from the parent element. Additional namespace
243    * information can be gathered using the
244    * {@link XMLStreamWriter2#getNamespaceContext()} method, which can be used when
245    * writing the provided instance value.
246    *
247    * @param instance
248    *          the {@link Field} instance value to write
249    * @param parentName
250    *          the qualified name of the XML data's parent element
251    * @param writer
252    *          the XML writer used to output the XML data
253    * @throws IOException
254    *           if an unexpected error occurred while processing the XML output
255    */
256   // TODO: migrate code to XML writer implementation.
257   void writeXmlValue(
258       @NonNull Object instance,
259       @NonNull IEnhancedQName parentName,
260       @NonNull XMLStreamWriter2 writer) throws IOException;
261 
262   /**
263    * Writes the provided Java class instance as a JSON/YAML field value.
264    *
265    * @param instance
266    *          the {@link Field} instance value to write
267    * @param writer
268    *          the JSON/YAML writer used to output the JSON/YAML data
269    * @throws IOException
270    *           if an unexpected error occurred while writing the JSON/YAML output
271    */
272   // TODO: migrate code to JSON writer implementation.
273   void writeJsonValue(@NonNull Object instance, @NonNull JsonGenerator writer) throws IOException;
274 
275   /**
276    * Gets the default value to use as the JSON/YAML field name for a Metaschema
277    * field value if no JSON value key flag or name is configured.
278    *
279    * @return the default field name to use
280    */
281   // TODO: migrate code to JSON implementations.
282   @NonNull
283   String getDefaultJsonValueKey();
284 
285   /**
286    * Determines if the data type's value is allowed to be unwrapped in XML when
287    * the value is a field value.
288    *
289    * @return {@code true} if allowed, or {@code false} otherwise.
290    */
291   // TODO: migrate code to XML implementations.
292   boolean isUnrappedValueAllowedInXml();
293 
294   /**
295    * Determines if the datatype uses mixed text and element content in XML.
296    *
297    * @return {@code true} if the datatype uses mixed text and element content in
298    *         XML, or {@code false} otherwise
299    */
300   // TODO: migrate code to XML implementations.
301   boolean isXmlMixed();
302 }

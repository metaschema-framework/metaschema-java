6   package gov.nist.secauto.metaschema.core.datatype.adapter;
7   
8   import gov.nist.secauto.metaschema.core.metapath.MetapathConstants;
9   import gov.nist.secauto.metaschema.core.metapath.item.atomic.INonNegativeIntegerItem;
10  import gov.nist.secauto.metaschema.core.qname.EQNameFactory;
11  import gov.nist.secauto.metaschema.core.qname.IEnhancedQName;
12  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
13  
14  import java.math.BigInteger;
15  import java.util.List;
16  
17  import edu.umd.cs.findbugs.annotations.NonNull;
18  
19  /**
20   * Support for the Metaschema <a href=
21   * "https://pages.nist.gov/metaschema/specification/datatypes/#non-negative-integer">non-negative-integer</a>
22   * data type.
23   */
24  public class NonNegativeIntegerAdapter
25      extends AbstractIntegerAdapter<INonNegativeIntegerItem> {
26    @NonNull
27    private static final List<IEnhancedQName> NAMES = ObjectUtils.notNull(
28        List.of(
29            EQNameFactory.instance().newQName(MetapathConstants.NS_METAPATH, "non-negative-integer"),
30            // for backwards compatibility with original type name
31            EQNameFactory.instance().newQName(MetapathConstants.NS_METAPATH, "nonNegativeInteger")));
32  
33    NonNegativeIntegerAdapter() {
34      super(INonNegativeIntegerItem.class, INonNegativeIntegerItem::cast);
35      // avoid general construction
36    }
37  
38    @Override
39    public List<IEnhancedQName> getNames() {
40      return NAMES;
41    }
42  
43    @Override
44    public INonNegativeIntegerItem newItem(Object value) {
45      BigInteger item = toValue(value);
46      return INonNegativeIntegerItem.valueOf(item);
47    }
48  }

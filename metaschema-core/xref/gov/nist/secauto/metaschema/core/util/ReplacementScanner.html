6   package gov.nist.secauto.metaschema.core.util;
7   
8   import java.util.function.Function;
9   import java.util.regex.Matcher;
10  import java.util.regex.Pattern;
11  
12  import edu.umd.cs.findbugs.annotations.NonNull;
13  
14  public final class ReplacementScanner {
15    private ReplacementScanner() {
16      // disable construction
17    }
18  
19    /**
20     * Search for instances of {@code pattern} in {@code text}. Replace each
21     * matching occurrence using the {@code replacementFunction}.
22     *
23     * @param text
24     *          the text to search
25     * @param pattern
26     *          the pattern to search for
27     * @param replacementFunction
28     *          a function that will provided the replacement text
29     * @return the resulting text after replacing matching occurrences in
30     *         {@code text}
31     */
32    public static CharSequence replaceTokens(@NonNull CharSequence text, @NonNull Pattern pattern,
33        Function<Matcher, CharSequence> replacementFunction) {
34      int lastIndex = 0;
35      StringBuilder retval = new StringBuilder();
36      Matcher matcher = pattern.matcher(text);
37      while (matcher.find()) {
38        retval.append(text, lastIndex, matcher.start())
39            .append(replacementFunction.apply(matcher));
40  
41        lastIndex = matcher.end();
42      }
43      if (lastIndex < text.length()) {
44        retval.append(text, lastIndex, text.length());
45      }
46      return retval;
47    }
48  }

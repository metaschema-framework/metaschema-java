17  import gov.nist.secauto.metaschema.core.model.IDefinition;
18  import gov.nist.secauto.metaschema.core.model.IFieldDefinition;
19  import gov.nist.secauto.metaschema.core.model.IFlagDefinition;

22  import edu.umd.cs.findbugs.annotations.NonNull;

24  /**
25   * Provides a variety of methods to check the dynamic type of a definition-based
26   * node item.
27   */
28  public final class DynamicTypeSupport {
29  
30    /**
31     * Checks that the provided actual item matches the provided expected type name.
32     *
33     * @param actual
34     *          the item to check
35     * @param expected
36     *          the expected definition or atomic type name
37     * @param staticContext
38     *          used to resolve definition names and lookup atomic type names
39     * @return {@code true} if the expected name matches either the node's
40     *         definition name or the node's atomic type, or {@code false} otherwise
41     */
42    public static boolean derivesFrom(
43        @NonNull IFlagNodeItem actual,
44        @NonNull String expected,
45        @NonNull StaticContext staticContext) {
46      IFlagDefinition definition = actual.getDefinition();
47      return compareDefinition(definition, expected, staticContext::parseFlagName)
48          || compareAtomicTypes(expected, definition.getJavaTypeAdapter().getItemType(), staticContext);
49    }
50  
51    /**
52     * Checks that the provided actual item matches the provided expected type name.
53     *
54     * @param actual
55     *          the item to check
56     * @param expected
57     *          the expected definition or atomic type name
58     * @param staticContext
59     *          used to resolve definition names and lookup atomic type names
60     * @return {@code true} if the expected name matches either the node's
61     *         definition name or the node's atomic type, or {@code false} otherwise
62     */
63    public static boolean derivesFrom(
64        @NonNull IFieldNodeItem actual,
65        @NonNull String expected,
66        @NonNull StaticContext staticContext) {
67      IFieldDefinition definition = actual.getDefinition();
68      return compareDefinition(definition, expected, staticContext::parseModelName)
69          || compareAtomicTypes(expected, definition.getJavaTypeAdapter().getItemType(), staticContext);
70    }
71  
72    /**
73     * Checks that the provided actual item matches the provided expected type name.
74     *
75     * @param actual
76     *          the item to check
77     * @param expected
78     *          the expected definition name
79     * @param staticContext
80     *          used to resolve definition names
81     * @return {@code true} if the expected name matches the node's definition name,
82     *         or {@code false} otherwise
83     */
84    public static boolean derivesFrom(
85        @NonNull IAssemblyNodeItem actual,
86        @NonNull String expected,
87        @NonNull StaticContext staticContext) {
88      try {
89        IEnhancedQName expectedName = staticContext.parseModelName(expected);
90        IAssemblyDefinition definition = actual.getDefinition();
91        return definition.getDefinitionQName().equals(expectedName); // AT is ET
92      } catch (StaticMetapathException ex) {
93        throw new InvalidTypeMetapathException(
94            null,
95            String.format("The expected type '%s' is not known to the type system.", expected),
96            ex);
97      }
98    }
99  
100   private static boolean compareDefinition(
101       @NonNull IDefinition definition,
102       @NonNull String expected,
103       @NonNull EQNameResolver nameResolver) {
104     boolean retval;
105     try {
106       IEnhancedQName expectedName = nameResolver.resolve(expected);
107       retval = definition.getDefinitionQName().equals(expectedName); // AT is ET
108     } catch (StaticMetapathException ex) {
109       // fail the definition name test
110       retval = false;
111     }
112     return retval;
113   }
114 
115   private static boolean compareAtomicTypes(
116       @NonNull String expected,
117       @NonNull IAtomicOrUnionType<?> actualType,
118       @NonNull StaticContext staticContext) {
119     // lookup the expected type
120     IAtomicOrUnionType<?> expectedType;
121     try {
122       expectedType = staticContext.lookupAtomicType(expected);
123     } catch (StaticMetapathException ex) {
124       throw new InvalidTypeMetapathException(
125           null,
126           String.format("The expected type '%s' is not known to the type system.", expected),
127           ex);
128     }
129 
130     return actualType.equals(expectedType) // ET is the base type of AT
131         || expectedType.isMemberType(actualType) // ET is a pure union type of which AT is a member
132         || expectedType.isSubType(actualType); // AT is derived from ET
133   }
134 
135   private DynamicTypeSupport() {
136     // disable construction
137   }
138 
139 }

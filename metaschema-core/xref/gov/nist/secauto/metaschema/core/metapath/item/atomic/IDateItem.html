6   package gov.nist.secauto.metaschema.core.metapath.item.atomic;
7   
8   import gov.nist.secauto.metaschema.core.datatype.adapter.MetaschemaDataTypeProvider;
9   import gov.nist.secauto.metaschema.core.datatype.object.AmbiguousDate;
10  import gov.nist.secauto.metaschema.core.datatype.object.AmbiguousDateTime;
11  import gov.nist.secauto.metaschema.core.metapath.function.InvalidValueForCastFunctionException;
12  import gov.nist.secauto.metaschema.core.metapath.item.atomic.impl.DateWithoutTimeZoneItemImpl;
13  import gov.nist.secauto.metaschema.core.metapath.type.IAtomicOrUnionType;
14  import gov.nist.secauto.metaschema.core.metapath.type.InvalidTypeMetapathException;
15  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
16  
17  import java.time.LocalDate;
18  import java.time.ZoneOffset;
19  import java.time.ZonedDateTime;
20  import java.time.temporal.ChronoUnit;
21  
22  import edu.umd.cs.findbugs.annotations.NonNull;
23  
24  /**
25   * An atomic Metapath item representing a date value in the Metapath system with
26   * or without an explicit time zone.
27   * <p>
28   * This interface provides functionality for handling date/time values with or
29   * without time zone information, supporting parsing, casting, and comparison
30   * operations. It works in conjunction with {@link ZonedDateTime} to handle time
31   * zone ambiguity.
32   */
33  public interface IDateItem extends ITemporalItem {
34    /**
35     * Get the type information for this item.
36     *
37     * @return the type information
38     */
39    @NonNull
40    static IAtomicOrUnionType<IDateItem> type() {
41      return MetaschemaDataTypeProvider.DATE.getItemType();
42    }
43  
44    @Override
45    default IAtomicOrUnionType<? extends IDateItem> getType() {
46      return type();
47    }
48  
49    /**
50     * Construct a new date item using the provided string {@code value}.
51     *
52     * @param value
53     *          a string representing a date
54     * @return the new item
55     */
56    @NonNull
57    static IDateItem valueOf(@NonNull String value) {
58      try {
59        return valueOf(MetaschemaDataTypeProvider.DATE.parse(value));
60      } catch (IllegalArgumentException ex) {
61        throw new InvalidTypeMetapathException(
62            null,
63            String.format("Invalid date value '%s'. %s",
64                value,
65                ex.getLocalizedMessage()),
66            ex);
67      }
68    }
69  
70    /**
71     * Construct a new date item using the provided {@code value}.
72     * <p>
73     * This method handles recording if an explicit timezone was provided using the
74     * {@code hasTimeZone} parameter. The {@link AmbiguousDate#hasTimeZone()} method
75     * can be called to determine if timezone information is present.
76     *
77     * @param value
78     *          a date, without time zone information
79     * @param hasTimeZone
80     *          {@code true} if the date/time is intended to have an associated time
81     *          zone or {@code false} otherwise
82     * @return the new item
83     * @see AmbiguousDateTime for more details on timezone handling
84     */
85    @NonNull
86    static IDateItem valueOf(@NonNull ZonedDateTime value, boolean hasTimeZone) {
87      return hasTimeZone
88          ? IDateWithTimeZoneItem.valueOf(value)
89          : valueOf(new AmbiguousDate(value, false));
90    }
91  
92    /**
93     * Construct a new date item using the provided {@code value}.
94     * <p>
95     * This method handles recording that the timezone is implicit.
96     *
97     * @param value
98     *          a date, without time zone information
99     * @return the new item
100    * @see AmbiguousDateTime for more details on timezone handling
101    */
102   @NonNull
103   static IDateItem valueOf(@NonNull LocalDate value) {
104     return valueOf(ObjectUtils.notNull(value.atStartOfDay(ZoneOffset.UTC)), false);
105   }
106 
107   /**
108    * Construct a new date item using the provided {@code value}.
109    *
110    * @param value
111    *          an ambiguous date with time zone information already identified
112    * @return the new item
113    */
114   @NonNull
115   static IDateItem valueOf(@NonNull AmbiguousDate value) {
116     return value.hasTimeZone()
117         ? IDateWithTimeZoneItem.valueOf(value.getValue())
118         : new DateWithoutTimeZoneItemImpl(value);
119   }
120 
121   /**
122    * Cast the provided type to this item type.
123    *
124    * @param item
125    *          the item to cast
126    * @return the original item if it is already this type, otherwise a new item
127    *         cast to this type
128    * @throws InvalidValueForCastFunctionException
129    *           if the provided {@code item} cannot be cast to this type
130    */
131   @NonNull
132   static IDateItem cast(@NonNull IAnyAtomicItem item) {
133     IDateItem retval;
134     if (item instanceof IDateItem) {
135       retval = (IDateItem) item;
136     } else if (item instanceof IDateTimeItem) {
137       IDateTimeItem dateTime = (IDateTimeItem) item;
138       // get the time at midnight
139       ZonedDateTime zdt = ObjectUtils.notNull(dateTime.asZonedDateTime().truncatedTo(ChronoUnit.DAYS));
140       // pass on the timezone ambiguity
141       retval = valueOf(zdt, dateTime.hasTimezone());
142     } else if (item instanceof IStringItem || item instanceof IUntypedAtomicItem) {
143       try {
144         retval = valueOf(item.asString());
145       } catch (IllegalStateException | InvalidTypeMetapathException ex) {
146         // asString can throw IllegalStateException exception
147         throw new InvalidValueForCastFunctionException(ex);
148       }
149     } else {
150       throw new InvalidValueForCastFunctionException(
151           String.format("unsupported item type '%s'", item.getClass().getName()));
152     }
153     return retval;
154   }
155 
156   @Override
157   default IDateItem castAsType(IAnyAtomicItem item) {
158     return cast(item);
159   }
160 
161   @Override
162   default int compareTo(IAnyAtomicItem item) {
163     return compareTo(cast(item));
164   }
165 }

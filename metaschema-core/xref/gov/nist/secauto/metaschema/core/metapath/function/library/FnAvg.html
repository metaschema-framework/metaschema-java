36  /**
37   * Implements the XPath 3.1
38   * <a href= "https://www.w3.org/TR/xpath-functions-31/#func-avg">fn:avg</a>
39   * function.
40   */
41  @SuppressWarnings("PMD.CouplingBetweenObjects")
42  public final class FnAvg {
43    private static final String NAME = "avg";
44  
45    @NonNull
46    static final IFunction SIGNATURE = IFunction.builder()
47        .name(NAME)
48        .namespace(MetapathConstants.NS_METAPATH_FUNCTIONS)
49        .deterministic()
50        .contextIndependent()
51        .focusIndependent()
52        .argument(IArgument.builder()
53            .name("arg")
54            .type(IAnyAtomicItem.type())
55            .zeroOrMore()
56            .build())
57        .returnType(IAnyAtomicItem.type())
58        .returnZeroOrOne()
59        .functionHandler(FnAvg::execute)
60        .build();
61  
62    private FnAvg() {
63      // disable construction
64    }
65  
66    @SuppressWarnings("unused")
67    @NonNull
68    private static ISequence<IAnyAtomicItem> execute(
69        @NonNull IFunction function,
70        @NonNull List<ISequence<?>> arguments,
71        @NonNull DynamicContext dynamicContext,
72        IItem focus) {
73      ISequence<? extends IAnyAtomicItem> sequence = FunctionUtils.asType(
74          ObjectUtils.requireNonNull(arguments.get(0)));
75  
76      List<? extends IAnyAtomicItem> items = sequence.getValue();
77  
78      return ISequence.of(average(items));
79    }
80  
81    /**
82     * An implementation of XPath 3.1
83     * <a href="https://www.w3.org/TR/xpath-functions-31/#func-avg">fn:avg</a>.
84     *
85     * @param items
86     *          the items to average
87     * @return the average
88     */
89    @SuppressWarnings("PMD.CyclomaticComplexity")
90    @Nullable
91    public static IAnyAtomicItem average(@NonNull Collection<? extends IAnyAtomicItem> items) {
92      if (items.isEmpty()) {
93        return null; // NOPMD - readability
94      }
95  
96      // tell cpd to start ignoring code - CPD-OFF
97  
98      Map<Class<? extends IAnyAtomicItem>, Integer> typeCounts = FunctionUtils.countTypes(
99          OperationFunctions.AGGREGATE_MATH_TYPES,
100         ObjectUtils.notNull(items));
101 
102     int count = items.size();
103     int dayTimeCount = typeCounts.getOrDefault(IDayTimeDurationItem.class, 0);
104     int yearMonthCount = typeCounts.getOrDefault(IYearMonthDurationItem.class, 0);
105     int numericCount = typeCounts.getOrDefault(INumericItem.class, 0);
106 
107     IAnyAtomicItem retval;
108     if (dayTimeCount > 0) {
109       if (dayTimeCount != count) {
110         throw new InvalidArgumentFunctionException(
111             InvalidArgumentFunctionException.INVALID_ARGUMENT_TYPE,
112             String.format("Values must all be of type '%s'.", IDayTimeDurationItem.class.getName()));
113       }
114 
115       List<IDayTimeDurationItem> values = items.stream()
116           .map(item -> (IDayTimeDurationItem) item)
117           .collect(Collectors.toList());
118       retval = averageDayTimeDurations(ObjectUtils.notNull(values));
119     } else if (yearMonthCount > 0) {
120       if (yearMonthCount != count) {
121         throw new InvalidArgumentFunctionException(
122             InvalidArgumentFunctionException.INVALID_ARGUMENT_TYPE,
123             String.format("Values must all be of type '%s'.", IYearMonthDurationItem.class.getName()));
124       }
125 
126       List<IYearMonthDurationItem> values = items.stream()
127           .map(item -> (IYearMonthDurationItem) item)
128           .collect(Collectors.toList());
129       retval = averageYearMonthDurations(ObjectUtils.notNull(values));
130     } else if (numericCount > 0) {
131       if (numericCount != count) {
132         throw new InvalidArgumentFunctionException(
133             InvalidArgumentFunctionException.INVALID_ARGUMENT_TYPE,
134             String.format("Values must all be of type '%s'.", INumericItem.class.getName()));
135       }
136 
137       List<INumericItem> values = items.stream()
138           .map(item -> IDecimalItem.cast(ObjectUtils.notNull(item)))
139           .collect(Collectors.toList());
140       retval = averageNumeric(ObjectUtils.notNull(values));
141     } else {
142       throw new InvalidArgumentFunctionException(
143           InvalidArgumentFunctionException.INVALID_ARGUMENT_TYPE,
144           String.format("Values must all be of type '%s'.",
145               OperationFunctions.AGGREGATE_MATH_TYPES.stream()
146                   .map(Class::getName)
147                   .collect(CustomCollectors.joiningWithOxfordComma(","))));
148     }
149 
150     // resume CPD analysis - CPD-ON
151 
152     return retval;
153   }
154 
155   @SuppressWarnings("PMD.UnnecessaryCast")
156   @NonNull
157   private static <T, R extends T> R average(
158       @NonNull Collection<? extends T> items,
159       @NonNull BinaryOperator<T> adder,
160       @NonNull BiFunction<T, IIntegerItem, R> divider) {
161     T sum = items.stream()
162         .map(item -> (T) item)
163         .reduce(adder)
164         .get();
165     return ObjectUtils.notNull(divider.apply(sum, IIntegerItem.valueOf(items.size())));
166   }

168   /**
169    * Get the average of a collection of day/time duration-based items.
170    *
171    * @param items
172    *          the Metapath items to average
173    * @return the average
174    */
175   @NonNull
176   public static IDayTimeDurationItem
177       averageDayTimeDurations(@NonNull Collection<? extends IDayTimeDurationItem> items) {
178     return average(
179         items,
180         (BinaryOperator<IDayTimeDurationItem>) OperationFunctions::opAddDayTimeDurations,
181         (BiFunction<IDayTimeDurationItem, IIntegerItem,
182             IDayTimeDurationItem>) OperationFunctions::opDivideDayTimeDuration);
183   }
184 
185   /**
186    * Get the average of a collection of year/month duration-based items.
187    *
188    * @param items
189    *          the Metapath items to average
190    * @return the average
191    */
192   @NonNull
193   public static IYearMonthDurationItem
194       averageYearMonthDurations(@NonNull Collection<? extends IYearMonthDurationItem> items) {
195     return average(
196         items,
197         (BinaryOperator<IYearMonthDurationItem>) OperationFunctions::opAddYearMonthDurations,
198         (BiFunction<IYearMonthDurationItem, IIntegerItem,
199             IYearMonthDurationItem>) OperationFunctions::opDivideYearMonthDuration);
200   }
201 
202   /**
203    * Get the average of a collection of numeric items.
204    *
205    * @param items
206    *          the Metapath items to average
207    * @return the average
208    */
209   @NonNull
210   public static IDecimalItem averageNumeric(@NonNull Collection<? extends INumericItem> items) {
211     return average(
212         items,
213         (BinaryOperator<INumericItem>) OperationFunctions::opNumericAdd,
214         (BiFunction<INumericItem, IIntegerItem, IDecimalItem>) OperationFunctions::opNumericDivide);
215   }

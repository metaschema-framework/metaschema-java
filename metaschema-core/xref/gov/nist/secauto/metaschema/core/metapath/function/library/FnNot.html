19  import edu.umd.cs.findbugs.annotations.NonNull;

21  /**
22   * Implements the XPath 3.1
23   * <a href= "https://www.w3.org/TR/xpath-functions-31/#func-not">fn:not</a>
24   * function.
25   */
26  public final class FnNot {
27    private static final String NAME = "not";
28    @NonNull
29    static final IFunction SIGNATURE = IFunction.builder()
30        .name(NAME)
31        .namespace(MetapathConstants.NS_METAPATH_FUNCTIONS)
32        .deterministic()
33        .contextIndependent()
34        .focusIndependent()
35        .argument(IArgument.builder()
36            .name("arg")
37            .type(IItem.class)
38            .zeroOrMore()
39            .build())
40        .returnType(IBooleanItem.class)
41        .returnOne()
42        .functionHandler(FnNot::execute)
43        .build();
44  
45    private FnNot() {
46      // disable construction
47    }
48  
49    @SuppressWarnings("unused")
50    @NonNull
51    private static ISequence<IBooleanItem> execute(@NonNull IFunction function,
52        @NonNull List<ISequence<?>> arguments,
53        @NonNull DynamicContext dynamicContext,
54        IItem focus) {
55  
56      ISequence<?> items = ObjectUtils.requireNonNull(arguments.iterator().next());
57  
58      IBooleanItem result = fnNot(items);
59      return ISequence.of(result);
60    }
61  
62    /**
63     * Get the negated, effective boolean value of the provided item.
64     * <p>
65     * Based on the XPath 3.1
66     * <a href="https://www.w3.org/TR/xpath-functions-31/#func-not">fn:not</a>
67     * function.
68     *
69     * @param item
70     *          the item to get the negated, effective boolean value for
71     * @return the negated boolean value
72     */
73    @NonNull
74    public static IBooleanItem fnNot(@NonNull IItem item) {
75      return IBooleanItem.valueOf(!FnBoolean.fnBooleanAsPrimitive(item));
76    }
77  
78    /**
79     * Get the negated, effective boolean value of the provided item.
80     * <p>
81     * Based on the XPath 3.1
82     * <a href="https://www.w3.org/TR/xpath-functions-31/#func-not">fn:not</a>
83     * function.
84     *
85     * @param sequence
86     *          the sequence to get the negated, effective boolean value for
87     * @return the negated boolean value
88     */
89    @NonNull
90    public static IBooleanItem fnNot(@NonNull ISequence<?> sequence) {
91      return FnBoolean.fnBoolean(sequence).negate();
92    }
93  }

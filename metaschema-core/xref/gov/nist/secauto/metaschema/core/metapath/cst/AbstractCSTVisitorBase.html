37    private static final Pattern QUALIFIED_NAME_PATTERN = Pattern.compile("^Q\\{([^}]*)\\}(.+)$");
38  
39    /**
40     * Get the QName for an
41     * <a href="https://www.w3.org/TR/xpath-31/#dt-expanded-qname">expanded
42     * QName</a>.
43     *
44     * @param eqname
45     *          the expanded QName
46     * @param context
47     *          the Metapath evaluation static context
48     * @param requireNamespace
49     *          if {@code true} require the resulting QName to have a namespace, or
50     *          {@code false} otherwise
51     * @return the QName
52     * @throws StaticMetapathException
53     *           if the expanded QName prefix is not bound or if the resulting
54     *           namespace is invalid
55     */
56    @SuppressWarnings({ "PMD.CyclomaticComplexity", "PMD.CognitiveComplexity" })
57    @NonNull
58    static QName toQName(@NonNull EqnameContext eqname, @NonNull StaticContext context, boolean requireNamespace) {
59      String namespaceUri;
60      String localName;
61      TerminalNode node;
62      if ((node = eqname.URIQualifiedName()) != null) {
63        // BracedURILiteral - Q{uri}name -
64        // https://www.w3.org/TR/xpath-31/#doc-xpath31-BracedURILiteral
65        Matcher matcher = QUALIFIED_NAME_PATTERN.matcher(node.getText());
66        if (matcher.matches()) {
67          namespaceUri = matcher.group(1);
68          localName = matcher.group(2);
69        } else {
70          // the syntax should always match above, since ANTLR is parsing it
71          throw new IllegalStateException();
72        }
73      } else {
74        String prefix;
75        String[] tokens = eqname.getText().split(":", 2);
76        if (tokens.length == 2) {
77          // lexical QName with prefix - prefix:name
78          // https://www.w3.org/TR/xpath-31/#dt-qname
79          prefix = ObjectUtils.notNull(tokens[0]);
80          localName = tokens[1];
81        } else {
82          // lexical QName without prefix - name
83          // https://www.w3.org/TR/xpath-31/#dt-qname
84          prefix = "";
85          localName = tokens[0];
86        }
87        namespaceUri = context.lookupNamespaceForPrefix(prefix);
88        if (namespaceUri == null && requireNamespace) {
89          throw new StaticMetapathException(
90              StaticMetapathException.PREFIX_NOT_EXPANDABLE,
91              String.format("The static context does not have a namespace URI configured for prefix '%s'.", prefix));
92        }
93      }
94  
95      QName retval;
96      if (namespaceUri == null) {
97        retval = new QName(localName);
98      } else {
99        if ("http://www.w3.org/2000/xmlns/".equals(namespaceUri)) {
100         throw new StaticMetapathException(StaticMetapathException.NAMESPACE_MISUSE,
101             "The namespace of an expanded QName cannot be: http://www.w3.org/2000/xmlns/");
102       }
103       retval = new QName(namespaceUri, localName);
104     }
105     return retval;
106   }

116   /**
117    * Parse the provided context as an n-ary phrase.
118    *
119    * @param <CONTEXT>
120    *          the Java type of the antlr context to parse
121    * @param <T>
122    *          the Java type of the child expressions produced by this parser
123    * @param <R>
124    *          the Java type of the outer expression produced by the parser
125    * @param context
126    *          the antlr context to parse
127    * @param startIndex
128    *          the child index to start parsing on
129    * @param step
130    *          the increment to advance while parsing child expressions
131    * @param parser
132    *          a binary function used to produce child expressions
133    * @param supplier
134    *          a function used to produce the other expression
135    * @return the outer expression or {@code null} if no children exist to parse
136    */
137   @Nullable
138   protected <CONTEXT extends ParserRuleContext, T extends IExpression, R extends IExpression>
139       R nairyToCollection(
140           @NonNull CONTEXT context,
141           int startIndex,
142           int step,
143           @NonNull BiFunction<CONTEXT, Integer, T> parser,
144           @NonNull Function<List<T>, R> supplier) {
145     int numChildren = context.getChildCount();
146 
147     R retval = null;
148     if (startIndex < numChildren) {
149       List<T> children = new ArrayList<>((numChildren - startIndex) / step);
150       for (int idx = startIndex; idx < numChildren; idx += step) {
151         T result = parser.apply(context, idx);
152         children.add(result);
153       }
154       retval = supplier.apply(children);
155     }
156     return retval;
157   }

159   /**
160    * Parse the provided context as an n-ary phrase, which will be one of the
161    * following.
162    * <ol>
163    * <li>A single <code>expr</code> for which that expr will be returned</li>
164    * <li><code>left (operator right)*</code> for which a collection of the left
165    * and right members will be returned based on what is provided by the
166    * supplier.</li>
167    * </ol>
168    *
169    * @param <CONTEXT>
170    *          the context type to parse
171    * @param <NODE>
172    *          the type of expression
173    * @param context
174    *          the context instance
175    * @param supplier
176    *          a supplier that will instantiate an expression based on the provided
177    *          parsed collection
178    * @return the left expression or the supplied expression for a collection
179    */
180   @NonNull
181   protected <CONTEXT extends ParserRuleContext, NODE extends IExpression> IExpression
182       handleNAiryCollection(
183           @NonNull CONTEXT context,
184           @NonNull Function<List<NODE>, IExpression> supplier) {
185     return handleNAiryCollection(context, 1, 2, (ctx, idx) -> {
186       // skip operator, since we know what it is
187       ParseTree tree = ctx.getChild(idx + 1);
188       @SuppressWarnings({ "unchecked", "null" })
189       @NonNull
190       NODE node = (NODE) tree.accept(this);
191       return node;
192     }, supplier);
193   }

195   /**
196    * Parse the provided context as an n-ary phrase, which will be one of the
197    * following.
198    * <ol>
199    * <li><code>expr</code> for which the expr will be returned.</li>
200    * <li><code>left</code> plus a number of additional recurring tokens as defined
201    * by the <em>step</em>.</li>
202    * </ol>
203    * <p>
204    * In the second case, the supplier will be used to generate an expression from
205    * the collection of tuples.
206    *
207    * @param <CONTEXT>
208    *          the context type to parse
209    * @param <EXPRESSION>
210    *          the child expression type
211    * @param context
212    *          the context instance
213    * @param startIndex
214    *          the starting context child position
215    * @param step
216    *          the amount to advance the loop over the context children
217    * @param parser
218    *          a binary function used to parse the context children
219    * @param supplier
220    *          a supplier that will instantiate an expression based on the provided
221    *          collection
222    * @return the left expression or the supplied expression for a collection
223    */
224   @NonNull
225   protected <CONTEXT extends ParserRuleContext, EXPRESSION extends IExpression> IExpression
226       handleNAiryCollection(
227           @NonNull CONTEXT context,
228           int startIndex,
229           int step,
230           @NonNull BiFunction<CONTEXT, Integer, EXPRESSION> parser,
231           @NonNull Function<List<EXPRESSION>, IExpression> supplier) {
232     int numChildren = context.getChildCount();
233 
234     if (numChildren == 0) {
235       throw new IllegalStateException("there should always be a child expression");
236     } else if (startIndex > numChildren) {
237       throw new IllegalStateException("Start index is out of bounds");
238     }
239 
240     ParseTree leftTree = context.getChild(0);
241     @SuppressWarnings({ "unchecked", "null" })
242     @NonNull
243     EXPRESSION leftResult = (EXPRESSION) leftTree.accept(this);
244 
245     IExpression retval;
246     if (numChildren == 1) {
247       retval = leftResult;
248     } else {
249       List<EXPRESSION> children = new ArrayList<>(numChildren - 1 / step);
250       children.add(leftResult);
251       for (int i = startIndex; i < numChildren; i = i + step) {
252         EXPRESSION result = parser.apply(context, i);
253         children.add(result);
254       }
255       IExpression result = ObjectUtils.notNull(supplier.apply(children));
256       retval = result;
257     }
258     return retval;
259   }

261   /**
262    * Parse the provided context as a simple n-ary phrase, which will be one of the
263    * following.
264    * <ol>
265    * <li><code>expr</code> for which the expr will be returned</li>
266    * <li><code>left (operator right)*</code> for which a collection of the left
267    * and right members will be returned based on what is provided by the supplier.
268    * </ol>
269    * <p>
270    * In the second case, the supplier will be used to generate an expression from
271    * the collection of tuples.
272    *
273    * @param <CONTEXT>
274    *          the context type to parse
275    * @param context
276    *          the context instance
277    * @param startingIndex
278    *          the index of the first child expression, which must be a
279    *          non-negative value that is less than the number of children
280    * @param step
281    *          the amount to advance the loop over the context children
282    * @param parser
283    *          a trinary function used to parse the context children and supply a
284    *          result
285    * @return the left expression or the supplied expression
286    */
287   protected <CONTEXT extends ParserRuleContext> IExpression handleGroupedNAiry(
288       @NonNull CONTEXT context,
289       int startingIndex,
290       int step,
291       @NonNull ITriFunction<CONTEXT, Integer, IExpression, IExpression> parser) {
292     int numChildren = context.getChildCount();
293     if (startingIndex >= numChildren) {
294       throw new IndexOutOfBoundsException(
295           String.format("The starting index '%d' exceeds the child count '%d'",
296               startingIndex,
297               numChildren));
298     }
299 
300     IExpression retval = null;
301     if (numChildren > 0) {
302       ParseTree leftTree = context.getChild(startingIndex);
303       retval = ObjectUtils.notNull(leftTree.accept(this));
304 
305       for (int i = startingIndex + 1; i < numChildren; i = i + step) {
306         retval = parser.apply(context, i, retval);
307       }
308     }
309     return retval;
310   }

312   @FunctionalInterface
313   interface ITriFunction<T, U, V, R> {
314 
315     R apply(T argT, U argU, V argV);
316 
317     default <W> ITriFunction<T, U, V, W> andThen(Function<? super R, ? extends W> after) {
318       Objects.requireNonNull(after);
319       return (T t, U u, V v) -> after.apply(apply(t, u, v));
320     }
321   }

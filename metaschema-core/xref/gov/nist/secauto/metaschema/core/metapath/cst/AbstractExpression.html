6   package gov.nist.secauto.metaschema.core.metapath.cst;
7   
8   import gov.nist.secauto.metaschema.core.metapath.ISequence;
9   import gov.nist.secauto.metaschema.core.metapath.item.atomic.IAnyAtomicItem;
10  import gov.nist.secauto.metaschema.core.metapath.type.TypeMetapathException;
11  
12  import edu.umd.cs.findbugs.annotations.NonNull;
13  import edu.umd.cs.findbugs.annotations.Nullable;
14  
15  public abstract class AbstractExpression implements IExpression {
16    /**
17     * Get the first data item of the provided {@code sequence} cast to an
18     * {@link IAnyAtomicItem}.
19     *
20     * @param sequence
21     *          the sequence to get the data item from
22     * @param requireSingleton
23     *          if {@code true} then a {@link TypeMetapathException} is thrown if
24     *          the sequence contains more than one item
25     * @return {@code null} if the sequence is empty, or the item otherwise
26     * @throws TypeMetapathException
27     *           if the sequence contains more than one item and requireSingleton is
28     *           {@code true}, or if the data item cannot be cast
29     */
30    @Nullable
31    public static IAnyAtomicItem getFirstDataItem(@NonNull ISequence<?> sequence,
32        boolean requireSingleton) {
33      return ISequence.of(sequence.atomize()).getFirstItem(requireSingleton);
34    }
35  
36    @Override
37    public String toString() {
38      return CSTPrinter.toString(this);
39    }
40  }

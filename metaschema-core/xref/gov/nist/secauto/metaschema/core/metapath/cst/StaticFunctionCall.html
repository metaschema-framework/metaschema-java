6   package gov.nist.secauto.metaschema.core.metapath.cst;
7   
8   import gov.nist.secauto.metaschema.core.metapath.DynamicContext;
9   import gov.nist.secauto.metaschema.core.metapath.ISequence;
10  import gov.nist.secauto.metaschema.core.metapath.StaticMetapathException;
11  import gov.nist.secauto.metaschema.core.metapath.function.FunctionService;
12  import gov.nist.secauto.metaschema.core.metapath.function.IFunction;
13  import gov.nist.secauto.metaschema.core.metapath.item.IItem;
14  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
15  
16  import java.util.List;
17  import java.util.Objects;
18  import java.util.stream.Collectors;
19  
20  import javax.xml.namespace.QName;
21  
22  import edu.umd.cs.findbugs.annotations.NonNull;
23  import nl.talsmasoftware.lazy4j.Lazy;
24  
25  public class StaticFunctionCall implements IExpression {
26    @NonNull
27    private final List<IExpression> arguments;
28    private final Lazy<IFunction> function;
29  
30    /**
31     * Construct a new function call expression.
32     *
33     * @param name
34     *          the function name
35     * @param arguments
36     *          the expressions used to provide arguments to the function call
37     */
38    public StaticFunctionCall(@NonNull QName name, @NonNull List<IExpression> arguments) {
39      this.arguments = Objects.requireNonNull(arguments, "arguments");
40      this.function = Lazy.lazy(() -> FunctionService.getInstance().getFunction(
41          Objects.requireNonNull(name, "name"),
42          arguments.size()));
43    }
44  
45    /**
46     * Retrieve the associated function.
47     *
48     * @return the function or {@code null} if no function matched the defined name
49     *         and arguments
50     * @throws StaticMetapathException
51     *           if the function was not found
52     */
53    public IFunction getFunction() {
54      return function.get();
55    }
56  
57    @Override
58    public List<IExpression> getChildren() {
59      return arguments;
60    }
61  
62    @Override
63    public Class<? extends IItem> getBaseResultType() {
64      Class<? extends IItem> retval = getFunction().getResult().getType();
65      if (retval == null) {
66        retval = IItem.class;
67      }
68      return retval;
69    }
70  
71    @SuppressWarnings("null")
72    @Override
73    public String toASTString() {
74      return String.format("%s[name=%s]", getClass().getName(), getFunction().getQName());
75    }
76  
77    @Override
78    public <RESULT, CONTEXT> RESULT accept(IExpressionVisitor<RESULT, CONTEXT> visitor, CONTEXT context) {
79      return visitor.visitFunctionCall(this, context);
80    }
81  
82    @Override
83    public ISequence<?> accept(DynamicContext dynamicContext, ISequence<?> focus) {
84      List<ISequence<?>> arguments = ObjectUtils.notNull(getChildren().stream().map(expression -> {
85        @NonNull ISequence<?> result = expression.accept(dynamicContext, focus);
86        return result;
87      }).collect(Collectors.toList()));
88  
89      IFunction function = getFunction();
90      return function.execute(arguments, dynamicContext, focus);
91    }
92  }

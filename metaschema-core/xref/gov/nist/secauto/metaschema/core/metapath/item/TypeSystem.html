94    private static Map.Entry<Class<? extends IItem>, QName> register(
95        @NonNull Class<? extends IItem> clazz,
96        @NonNull IDataTypeAdapter<?> adapter) {
97      return Map.entry(clazz, adapter.getPreferredName());
98    }
99  
100   private static Map.Entry<Class<? extends IItem>, QName> register(
101       @NonNull Class<? extends IItem> clazz,
102       @NonNull String typeName) {
103     return Map.entry(clazz, new QName(MetapathConstants.NS_METAPATH.toASCIIString(), typeName));
104   }
105 
106   @NonNull
107   private static Stream<Class<? extends IItem>> getItemInterfaces(@NonNull Class<?> clazz) {
108     @SuppressWarnings("unchecked") Stream<Class<? extends IItem>> retval = IItem.class.isAssignableFrom(clazz)
109         ? Stream.of((Class<? extends IItem>) clazz)
110         : Stream.empty();
111 
112     Class<?>[] interfaces = clazz.getInterfaces();
113     if (interfaces.length > 0) {
114       retval = Stream.concat(retval, Arrays.stream(interfaces).flatMap(TypeSystem::getItemInterfaces));
115     }
116 
117     return ObjectUtils.notNull(retval);
118   }
119 
120   /**
121    * Get the human-friendly data type name for the provided Metapath item class.
122    *
123    * @param clazz
124    *          the Metapath item class to get the name for
125    * @return the name or {@code null} if no name is registered for the item class
126    */
127   public static String getName(@NonNull Class<? extends IItem> clazz) {
128     Class<? extends IItem> itemClass = getItemInterfaces(clazz).findFirst().orElse(null);
129 
130     QName qname = ITEM_CLASS_TO_QNAME_MAP.get(itemClass);
131     return qname == null ? clazz.getName() : asPrefixedName(qname);
132   }
133 
134   private static String asPrefixedName(@NonNull QName qname) {
135     String namespace = qname.getNamespaceURI();
136     String prefix = namespace.isEmpty() ? null : StaticContext.getWellKnownPrefixForUri(namespace);
137     return prefix == null ? qname.toString() : prefix + ":" + qname.getLocalPart();
138   }

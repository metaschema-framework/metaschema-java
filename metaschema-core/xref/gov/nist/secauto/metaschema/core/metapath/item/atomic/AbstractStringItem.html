6   package gov.nist.secauto.metaschema.core.metapath.item.atomic;
7   
8   import gov.nist.secauto.metaschema.core.metapath.impl.AbstractStringMapKey;
9   import gov.nist.secauto.metaschema.core.metapath.item.function.IMapKey;
10  
11  import java.util.regex.Pattern;
12  
13  import edu.umd.cs.findbugs.annotations.NonNull;
14  
15  public abstract class AbstractStringItem
16      extends AbstractAnyAtomicItem<String>
17      implements IStringItem {
18    private static final String WHITESPACE_SEGMENT = "[ \t\r\n]";
19    private static final Pattern TRIM_END = Pattern.compile(WHITESPACE_SEGMENT + "++$");
20    private static final Pattern TRIM_START = Pattern.compile("^" + WHITESPACE_SEGMENT + "+");
21    private static final Pattern TRIM_MIDDLE = Pattern.compile(WHITESPACE_SEGMENT + "{2,}");
22  
23    /**
24     * Construct a new string item with the provided {@code value}.
25     *
26     * @param value
27     *          the value to wrap
28     */
29    protected AbstractStringItem(@NonNull String value) {
30      super(value);
31    }
32  
33    @Override
34    public String asString() {
35      return getValue();
36    }
37  
38    @Override
39    public IMapKey asMapKey() {
40      return new MapKey();
41    }
42  
43    @Override
44    public int hashCode() {
45      return asString().hashCode();
46    }
47  
48    @SuppressWarnings("PMD.OnlyOneReturn")
49    @Override
50    public boolean equals(Object obj) {
51      return this == obj
52          || (obj instanceof IStringItem && compareTo((IStringItem) obj) == 0);
53    }
54  
55    private final class MapKey
56        extends AbstractStringMapKey {
57  
58      @Override
59      public IStringItem getKey() {
60        return AbstractStringItem.this;
61      }
62    }
63  
64    @Override
65    public IStringItem normalizeSpace() {
66      String value = asString();
67      value = TRIM_START.matcher(value).replaceFirst("");
68      value = TRIM_MIDDLE.matcher(value).replaceAll(" ");
69      value = TRIM_END.matcher(value).replaceFirst("");
70      assert value != null;
71  
72      return IStringItem.valueOf(value);
73    }
74  
75  }

6   package gov.nist.secauto.metaschema.core.metapath.type.impl;
7   
8   import gov.nist.secauto.metaschema.core.metapath.item.function.IMapItem;
9   import gov.nist.secauto.metaschema.core.metapath.type.IAtomicOrUnionType;
10  import gov.nist.secauto.metaschema.core.metapath.type.IMapTest;
11  import gov.nist.secauto.metaschema.core.metapath.type.ISequenceType;
12  
13  import edu.umd.cs.findbugs.annotations.NonNull;
14  
15  /**
16   * An item type that applies to all {@link IMapItem} with a specific key and
17   * value type.
18   */
19  public class MapTestImpl implements IMapTest {
20    @NonNull
21    private final IAtomicOrUnionType<?> keyType;
22    @NonNull
23    private final ISequenceType valueType;
24  
25    /**
26     * Construct a new item type.
27     *
28     * @param keyType
29     *          the atomic type to match map keys against
30     * @param valueType
31     *          the sequence type to match map values against
32     */
33    public MapTestImpl(
34        @NonNull IAtomicOrUnionType<?> keyType,
35        @NonNull ISequenceType valueType) {
36      this.keyType = keyType;
37      this.valueType = valueType;
38    }
39  
40    @Override
41    public IAtomicOrUnionType<?> getKeyType() {
42      return keyType;
43    }
44  
45    @Override
46    public ISequenceType getValueType() {
47      return valueType;
48    }
49  
50    @Override
51    public String toString() {
52      return toSignature();
53    }
54  }

2   package gov.nist.secauto.metaschema.core.metapath.item.node;
3   
4   import gov.nist.secauto.metaschema.core.util.ObjectUtils;
5   
6   import edu.umd.cs.findbugs.annotations.Nullable;
7   
8   public abstract class AbstractNodeItem implements INodeItem {
9   
10    @Override
11    public final String toSignature() {
12      StringBuilder builder = new StringBuilder()
13          .append(getType().toSignature())
14          .append("⪻")
15          .append(getMetapath())
16          .append("⪼");
17      String value = getValueSignature();
18      if (value != null) {
19        builder.append("(");
20        builder.append(value);
21        builder.append(")");
22      }
23      return ObjectUtils.notNull(builder.toString());
24    }
25  
26    @Nullable
27    protected abstract String getValueSignature();
28  
29    @Override
30    public final String toString() {
31      return toSignature();
32    }
33  }

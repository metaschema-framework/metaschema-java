6   package gov.nist.secauto.metaschema.core.metapath.type;
7   
8   import gov.nist.secauto.metaschema.core.metapath.item.atomic.IAnyAtomicItem;
9   import gov.nist.secauto.metaschema.core.metapath.type.impl.AbstractItemType;
10  
11  import edu.umd.cs.findbugs.annotations.NonNull;
12  
13  /**
14   * An abstract implementation of an atomic type.
15   *
16   * @param <T>
17   *          the Java type of the item supported by the implementation
18   */
19  public abstract class AbstractAtomicOrUnionType<T extends IAnyAtomicItem>
20      extends AbstractItemType<T>
21      implements IAtomicOrUnionType<T> {
22    @NonNull
23    private final ICastExecutor<T> castExecutor;
24  
25    /**
26     * Construct a new atomic type.
27     *
28     * @param itemClass
29     *          the item class this atomic type supports
30     * @param castExecutor
31     *          the executor used to cast an item to an item of this type
32     */
33    public AbstractAtomicOrUnionType(
34        @NonNull Class<T> itemClass,
35        @NonNull ICastExecutor<T> castExecutor) {
36      super(itemClass);
37      this.castExecutor = castExecutor;
38    }
39  
40    @Override
41    public T cast(IAnyAtomicItem item) {
42      return castExecutor.cast(item);
43    }
44  }

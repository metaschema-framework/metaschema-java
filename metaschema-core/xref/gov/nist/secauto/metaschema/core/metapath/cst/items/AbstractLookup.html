6   package gov.nist.secauto.metaschema.core.metapath.cst.items;
7   
8   import gov.nist.secauto.metaschema.core.metapath.cst.IExpression;
9   import gov.nist.secauto.metaschema.core.metapath.item.function.IArrayItem;
10  import gov.nist.secauto.metaschema.core.metapath.item.function.IKeySpecifier;
11  import gov.nist.secauto.metaschema.core.metapath.item.function.IMapItem;
12  
13  import edu.umd.cs.findbugs.annotations.NonNull;
14  
15  /**
16   * An implementation of
17   * <a href="https://www.w3.org/TR/xpath-31/#id-lookup">Lookup Operators</a>
18   * supporting access to items in Metapath maps and arrays.
19   * <p>
20   * Provides support for various types of key- and index-based lookups related to
21   * {@link IMapItem} and {@link IArrayItem} objects.
22   */
23  public abstract class AbstractLookup implements IExpression {
24    @NonNull
25    private final IKeySpecifier keySpecifier;
26  
27    /**
28     * Construct a new lookup expression that uses the provided key specifier.
29     *
30     * @param keySpecifier
31     *          the key specifier that identifies how to lookup entries
32     */
33    protected AbstractLookup(@NonNull IKeySpecifier keySpecifier) {
34      this.keySpecifier = keySpecifier;
35    }
36  
37    /**
38     * Get the key specifier implementation.
39     *
40     * @return the key specifier
41     */
42    @NonNull
43    public IKeySpecifier getKeySpecifier() {
44      return keySpecifier;
45    }
46  }

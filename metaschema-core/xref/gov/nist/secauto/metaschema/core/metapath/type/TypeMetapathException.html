6   package gov.nist.secauto.metaschema.core.metapath.type;
7   
8   import gov.nist.secauto.metaschema.core.metapath.AbstractCodedMetapathException;
9   
10  /**
11   * MPTY: Exceptions related to Metapath type errors.
12   */
13  public class TypeMetapathException
14      extends AbstractCodedMetapathException {
15    /**
16     * <a href= "https://www.w3.org/TR/xpath-31/#ERRXPTY0004">err:MPTY0004</a>: It
17     * is a <a href="https://www.w3.org/TR/xpath-31/#dt-type-error">type error</a>
18     * if during the
19     * <a href="https://www.w3.org/TR/xpath-31/#dt-static-analysis">static analysis
20     * phase</a>, an expression is found to have a
21     * <a href="https://www.w3.org/TR/xpath-31/#dt-static-type">static type</a> that
22     * is not appropriate for the context in which the expression occurs, or during
23     * the <a href="https://www.w3.org/TR/xpath-31/#dt-dynamic-evaluation">dynamic
24     * evaluation phase</a>, the
25     * <a href="https://www.w3.org/TR/xpath-31/#dt-dynamic-type">dynamic type</a> of
26     * a value does not match a required type as specified by the matching rules in
27     * <a href="https://www.w3.org/TR/xpath-31/#id-sequencetype-matching">2.5.5
28     * SequenceType Matching</a>.
29     */
30    public static final int INVALID_TYPE_ERROR = 4;
31    /**
32     * <a href= "https://www.w3.org/TR/xpath-31/#ERRXPTY0019">err:MPTY0019</a>: It
33     * is a <a href="https://www.w3.org/TR/xpath-31/#dt-type-error">type error</a>
34     * if {@code E1} in a path expression {@code E1/E2} does not evaluate to a
35     * sequence of nodes.
36     */
37    public static final int BASE_PATH_NOT_A_SEQUENCE = 19;
38    /**
39     * The context item is not a node when evaluating an axis step.
40     */
41    public static final int NOT_A_NODE_ITEM_FOR_STEP = 20;
42  
43    /**
44     * the serial version UID.
45     */
46    private static final long serialVersionUID = 2L;
47  
48    /**
49     * Constructs a new exception with the provided {@code code}, {@code message},
50     * and {@code cause}.
51     *
52     * @param code
53     *          the error code value
54     * @param message
55     *          the exception message
56     * @param cause
57     *          the original exception cause
58     */
59    public TypeMetapathException(int code, String message, Throwable cause) {
60      super(code, message, cause);
61    }
62  
63    /**
64     * Constructs a new exception with the provided {@code code}, {@code message},
65     * and no cause.
66     *
67     * @param code
68     *          the error code value
69     * @param message
70     *          the exception message
71     */
72    public TypeMetapathException(int code, String message) {
73      super(code, message);
74    }
75  
76    /**
77     * Constructs a new exception with the provided {@code code}, no message, and
78     * the {@code cause}.
79     *
80     * @param code
81     *          the error code value
82     * @param cause
83     *          the original exception cause
84     */
85    public TypeMetapathException(int code, Throwable cause) {
86      super(code, cause);
87    }
88  
89    @Override
90    public String getCodePrefix() {
91      return "MPTY";
92    }
93  }

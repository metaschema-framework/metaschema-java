6   package gov.nist.secauto.metaschema.core.metapath.item.atomic;
7   
8   import gov.nist.secauto.metaschema.core.datatype.IDataTypeAdapter;
9   import gov.nist.secauto.metaschema.core.util.ObjectUtils;
10  
11  import edu.umd.cs.findbugs.annotations.NonNull;
12  
13  /**
14   * Provides a common implementation for all atomic types.
15   *
16   * @param <TYPE>
17   *          the Java type associated with the atomic type.
18   */
19  public abstract class AbstractAnyAtomicItem<TYPE> implements IAnyAtomicItem {
20    @NonNull
21    private final TYPE value;
22  
23    /**
24     * Construct a new atomic item using the provided {@code value}.
25     *
26     * @param value
27     *          the value to assign to this atomic item
28     */
29    protected AbstractAnyAtomicItem(@NonNull TYPE value) {
30      this.value = ObjectUtils.requireNonNull(value, "value");
31    }
32  
33    @Override
34    @NonNull
35    public TYPE getValue() {
36      return value;
37    }
38  
39    @Override
40    @NonNull
41    public abstract IDataTypeAdapter<TYPE> getJavaTypeAdapter();
42  
43    @Override
44    public String asString() {
45      return getJavaTypeAdapter().asString(getValue());
46    }
47  
48    @Override
49    public String toString() {
50      return asString();
51    }
52  }

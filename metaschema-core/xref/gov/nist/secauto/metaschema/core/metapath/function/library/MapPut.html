27  /**
28   * Implements the XPath 3.1
29   * <a href= "https://www.w3.org/TR/xpath-functions-31/#func-map-put">map:put</a>
30   * function.
31   */
32  public final class MapPut {
33    private static final String NAME = "put";
34    @NonNull
35    static final IFunction SIGNATURE = IFunction.builder()
36        .name(NAME)
37        .namespace(MetapathConstants.NS_METAPATH_FUNCTIONS_MAP)
38        .deterministic()
39        .contextIndependent()
40        .focusIndependent()
41        .argument(IArgument.builder()
42            .name("map")
43            .type(IMapItem.class)
44            .one()
45            .build())
46        .argument(IArgument.builder()
47            .name("key")
48            .type(IAnyAtomicItem.class)
49            .one()
50            .build())
51        .argument(IArgument.builder()
52            .name("value")
53            .type(IItem.class)
54            .zeroOrMore()
55            .build())
56        .returnType(IMapItem.class)
57        .returnOne()
58        .functionHandler(MapPut::execute)
59        .build();
60  
61    private MapPut() {
62      // disable construction
63    }
64  
65    @SuppressWarnings("unused")
66    @NonNull
67    private static <V extends ICollectionValue> ISequence<?> execute(@NonNull IFunction function,
68        @NonNull List<ISequence<?>> arguments,
69        @NonNull DynamicContext dynamicContext,
70        IItem focus) {
71      IMapItem<V> map = FunctionUtils.asType(ObjectUtils.requireNonNull(arguments.get(0).getFirstItem(true)));
72      IAnyAtomicItem key = FunctionUtils.asType(ObjectUtils.requireNonNull(arguments.get(1).getFirstItem(true)));
73      @SuppressWarnings("unchecked")
74      V value = (V) ObjectUtils.requireNonNull(arguments.get(2)).toCollectionValue();
75  
76      return put(map, key, value).asSequence();
77    }
78  
79    /**
80     * An implementation of XPath 3.1 <a href=
81     * "https://www.w3.org/TR/xpath-functions-31/#func-map-put">map:put</a>.
82     *
83     * @param <V>
84     *          the type of items in the given Metapath map
85     * @param map
86     *          the map of Metapath items that is to be modified
87     * @param key
88     *          the key for the value to add to the map
89     * @param value
90     *          the value to add to the map
91     * @return the modified map
92     */
93    @NonNull
94    public static <V extends ICollectionValue> IMapItem<V> put(
95        @NonNull IMapItem<V> map,
96        @NonNull IAnyAtomicItem key,
97        @NonNull V value) {
98      @SuppressWarnings("PMD.UseConcurrentHashMap")
99      Map<IMapKey, V> copy = new HashMap<>(map);
100     copy.put(key.asMapKey(), value);
101 
102     return IMapItem.ofCollection(copy);
103   }
104 }

6   package gov.nist.secauto.metaschema.core.metapath.item.atomic;
7   
8   import gov.nist.secauto.metaschema.core.datatype.adapter.MetaschemaDataTypeProvider;
9   import gov.nist.secauto.metaschema.core.metapath.function.InvalidValueForCastFunctionException;
10  
11  import java.time.Duration;
12  
13  import edu.umd.cs.findbugs.annotations.NonNull;
14  
15  public interface IDayTimeDurationItem extends IDurationItem {
16    /**
17     * Construct a new day time duration item using the provided string
18     * {@code value}.
19     *
20     * @param value
21     *          a string representing a day time duration
22     * @return the new item
23     */
24    @NonNull
25    static IDayTimeDurationItem valueOf(@NonNull String value) {
26      try {
27        return valueOf(MetaschemaDataTypeProvider.DAY_TIME_DURATION.parse(value));
28      } catch (IllegalArgumentException ex) {
29        throw new InvalidValueForCastFunctionException(String.format("Unable to parse string value '%s'", value),
30            ex);
31      }
32    }
33  
34    /**
35     * Construct a new day time duration item using the provided {@code value}.
36     *
37     * @param value
38     *          a duration
39     * @return the new item
40     */
41    @NonNull
42    static IDayTimeDurationItem valueOf(@NonNull Duration value) {
43      return new DayTimeDurationItemImpl(value);
44    }
45  
46    /**
47     * Cast the provided type to this item type.
48     *
49     * @param item
50     *          the item to cast
51     * @return the original item if it is already this type, otherwise a new item
52     *         cast to this type
53     * @throws InvalidValueForCastFunctionException
54     *           if the provided {@code item} cannot be cast to this type
55     */
56    @NonNull
57    static IDayTimeDurationItem cast(@NonNull IAnyAtomicItem item) {
58      return MetaschemaDataTypeProvider.DAY_TIME_DURATION.cast(item);
59    }
60  
61    /**
62     * Get the items wrapped value as a duration.
63     *
64     * @return the wrapped value as a duration
65     */
66    @NonNull
67    Duration asDuration();
68  
69    /**
70     * Get the "wrapped" duration value in seconds.
71     *
72     * @return the underlying duration in seconds
73     */
74    default long asSeconds() {
75      return asDuration().toSeconds();
76    }
77  
78    @Override
79    default IDayTimeDurationItem castAsType(IAnyAtomicItem item) {
80      return cast(item);
81    }
82  
83    /**
84     * Compares this value with the argument.
85     *
86     * @param item
87     *          the item to compare with this value
88     * @return a negative integer, zero, or a positive integer if this value is less
89     *         than, equal to, or greater than the {@code item}.
90     */
91    default int compareTo(@NonNull IDayTimeDurationItem item) {
92      return asDuration().compareTo(item.asDuration());
93  
94    }
95  
96    @Override
97    default int compareTo(IAnyAtomicItem item) {
98      return compareTo(cast(item));
99    }
100 }

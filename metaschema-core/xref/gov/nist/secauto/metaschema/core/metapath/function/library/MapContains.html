21  import java.util.List;
22  
23  import edu.umd.cs.findbugs.annotations.NonNull;
24  
25  public final class MapContains {
26    private static final String NAME = "contains";
27    @NonNull
28    static final IFunction SIGNATURE = IFunction.builder()
29        .name(NAME)
30        .namespace(MetapathConstants.NS_METAPATH_FUNCTIONS_MAP)
31        .deterministic()
32        .contextIndependent()
33        .focusIndependent()
34        .argument(IArgument.builder()
35            .name("map")
36            .type(IMapItem.class)
37            .one()
38            .build())
39        .argument(IArgument.builder()
40            .name("key")
41            .type(IAnyAtomicItem.class)
42            .one()
43            .build())
44        .returnType(IBooleanItem.class)
45        .returnOne()
46        .functionHandler(MapContains::execute)
47        .build();
48  
49    private MapContains() {
50      // disable construction
51    }
52  
53    @SuppressWarnings("unused")
54    @NonNull
55    private static ISequence<?> execute(@NonNull IFunction function,
56        @NonNull List<ISequence<?>> arguments,
57        @NonNull DynamicContext dynamicContext,
58        IItem focus) {
59      IMapItem<?> map = FunctionUtils.asType(ObjectUtils.requireNonNull(arguments.get(0).getFirstItem(true)));
60      IAnyAtomicItem key = FunctionUtils.asType(ObjectUtils.requireNonNull(arguments.get(1).getFirstItem(true)));
61  
62      return IBooleanItem.valueOf(contains(map, key)).asSequence();
63    }
64  
65    /**
66     * An implementation of XPath 3.1 <a href=
67     * "https://www.w3.org/TR/xpath-functions-31/#func-map-contains">map:contains</a>.
68     *
69     * @param <V>
70     *          the type of items in the given Metapath map
71     * @param map
72     *          the map of Metapath items that is the target of retrieval
73     * @param key
74     *          the key for the item to retrieve
75     * @return {@code true} if the key exists in the map, or {@code false} otherwise
76     */
77    public static <V extends ICollectionValue> boolean contains(
78        @NonNull IMapItem<V> map,
79        @NonNull IAnyAtomicItem key) {
80      return map.get(key.asMapKey()) != null;
81    }
82  }

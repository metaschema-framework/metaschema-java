6   package gov.nist.secauto.metaschema.core.metapath.item.atomic;
7   
8   import gov.nist.secauto.metaschema.core.datatype.IDataTypeAdapter;
9   import gov.nist.secauto.metaschema.core.metapath.IPrintable;
10  import gov.nist.secauto.metaschema.core.metapath.item.IItemVisitor;
11  import gov.nist.secauto.metaschema.core.metapath.item.function.IMapItem;
12  import gov.nist.secauto.metaschema.core.metapath.item.function.IMapKey;
13  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
14  
15  import java.util.Set;
16  
17  import edu.umd.cs.findbugs.annotations.NonNull;
18  
19  public interface IAnyAtomicItem extends IAtomicValuedItem, IPrintable {
20    @NonNull
21    Set<Class<? extends IAnyAtomicItem>> PRIMITIVE_ITEM_TYPES = ObjectUtils.notNull(Set.of(
22        IStringItem.class,
23        IBooleanItem.class,
24        IDecimalItem.class,
25        IDurationItem.class,
26        IDateTimeItem.class,
27        IDateItem.class,
28        IBase64BinaryItem.class,
29        IAnyUriItem.class));
30  
31    @Override
32    @NonNull
33    default IAnyAtomicItem toAtomicItem() {
34      return this;
35    }
36  
37    /**
38     * Get the "wrapped" value represented by this item.
39     *
40     * @return the value
41     */
42    @Override
43    @NonNull
44    Object getValue();
45  
46    @Override
47    @NonNull
48    String toString();
49  
50    /**
51     * Get the item's string value.
52     *
53     * @return the string value value of the item
54     */
55    @Override
56    @NonNull
57    String asString();
58  
59    /**
60     * Get the atomic item value as a map key for use with an {@link IMapItem}.
61     *
62     * @return the map key
63     */
64    @NonNull
65    IMapKey asMapKey();
66  
67    /**
68     * Get a new {@link IStringItem} based on the the textual value of the item's
69     * "wrapped" value.
70     *
71     * @return a new string item
72     */
73    @NonNull
74    default IStringItem asStringItem() {
75      return IStringItem.valueOf(asString());
76    }
77  
78    /**
79     * Get the item's type adapter.
80     *
81     * @return the type adapter for the item
82     */
83    @NonNull
84    IDataTypeAdapter<?> getJavaTypeAdapter();
85    //
86    // <T extends IValuedItem> T cast(IValuedItem item);
87  
88    /**
89     * Cast the provided {@code item} to be the same type as this item.
90     *
91     * @param item
92     *          the item to cast
93     * @return the result from casting
94     */
95    @NonNull
96    IAnyAtomicItem castAsType(@NonNull IAnyAtomicItem item);
97  
98    /**
99     * Compares this value with the argument. Ordering is item type dependent.
100    *
101    * @param other
102    *          the item to compare with this value
103    * @return a negative integer, zero, or a positive integer if this value is less
104    *         than, equal to, or greater than the {@code item}.
105    */
106   int compareTo(@NonNull IAnyAtomicItem other);
107 
108   @Override
109   default void accept(IItemVisitor visitor) {
110     visitor.visit(this);
111   }
112 }

6   package gov.nist.secauto.metaschema.core.metapath.item.atomic;
7   
8   import gov.nist.secauto.metaschema.core.datatype.IDataTypeAdapter;
9   import gov.nist.secauto.metaschema.core.metapath.function.InvalidValueForCastFunctionException;
10  import gov.nist.secauto.metaschema.core.metapath.item.IItemVisitor;
11  import gov.nist.secauto.metaschema.core.metapath.item.function.IMapItem;
12  import gov.nist.secauto.metaschema.core.metapath.item.function.IMapKey;
13  import gov.nist.secauto.metaschema.core.metapath.type.IAtomicOrUnionType;
14  import gov.nist.secauto.metaschema.core.metapath.type.IItemType;
15  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
16  
17  import java.util.stream.Stream;
18  
19  import edu.umd.cs.findbugs.annotations.NonNull;
20  
21  /**
22   * The interface shared by all atomic items, representing indivisible data
23   * values that serve as the fundamental building blocks for complex data
24   * structures in the Metaschema framework.
25   */
26  public interface IAnyAtomicItem extends IAtomicValuedItem {
27    /**
28     * Get the type information for this item.
29     *
30     * @return the type information
31     */
32    @NonNull
33    static IAtomicOrUnionType<?> type() {
34      return IItemType.anyAtomic();
35    }
36  
37    @Override
38    @NonNull
39    default IAnyAtomicItem toAtomicItem() {
40      return this;
41    }
42  
43    /**
44     * Get the "wrapped" value represented by this item.
45     *
46     * @return the value
47     */
48    @Override
49    @NonNull
50    Object getValue();
51  
52    /**
53     * Get a new {@link IStringItem} based on the the textual value of the item's
54     * "wrapped" value.
55     *
56     * @return a new string item
57     */
58    default IStringItem asStringItem() {
59      return IStringItem.valueOf(asString());
60    }
61  
62    /**
63     * Get the item's string value.
64     *
65     * @return the string value value of the item
66     */
67    @NonNull
68    String asString();
69  
70    @Override
71    default Stream<IAnyAtomicItem> atomize() {
72      return ObjectUtils.notNull(Stream.of(this));
73    }
74  
75    /**
76     * Get the atomic item value as a map key for use with an {@link IMapItem}.
77     *
78     * @return the map key
79     */
80    @NonNull
81    IMapKey asMapKey();
82  
83    /**
84     * Get the item's type adapter.
85     *
86     * @return the type adapter for the item
87     */
88    @NonNull
89    IDataTypeAdapter<?> getJavaTypeAdapter();
90  
91    /**
92     * Cast the provided type to this item type.
93     * <p>
94     * This method simply returns the provided item, since it is already the same
95     * type.
96     *
97     * @param item
98     *          the item to cast
99     * @return the provided item
100    */
101   static IAnyAtomicItem cast(@NonNull IAnyAtomicItem item) {
102     return item;
103   }
104 
105   /**
106    * Cast the provided {@code item} to be the same type as this item.
107    *
108    * @param item
109    *          the item to cast
110    * @return an atomic item of this type
111    * @throws InvalidValueForCastFunctionException
112    *           if the provided item type cannot be cast to this item type
113    */
114   @NonNull
115   IAnyAtomicItem castAsType(@NonNull IAnyAtomicItem item);
116 
117   /**
118    * Compares this value with the argument. Ordering is item type dependent.
119    *
120    * @param other
121    *          the item to compare with this value
122    * @return a negative integer, zero, or a positive integer if this value is less
123    *         than, equal to, or greater than the {@code item}.
124    */
125   int compareTo(@NonNull IAnyAtomicItem other);
126 
127   @Override
128   default void accept(IItemVisitor visitor) {
129     visitor.visit(this);
130   }
131 }

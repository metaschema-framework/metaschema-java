22  /**
23   * Implements the XPath 3.1 <a href=
24   * "https://www.w3.org/TR/xpath-functions-31/#func-ends-with">fn:ends-with</a>
25   * function.
26   */
27  public final class FnEndsWith {
28    // CPD-OFF
29    @NonNull
30    private static final String NAME = "ends-with";
31    @NonNull
32    static final IFunction SIGNATURE = IFunction.builder()
33        .name(NAME)
34        .namespace(MetapathConstants.NS_METAPATH_FUNCTIONS)
35        .deterministic()
36        .contextDependent()
37        .focusIndependent()
38        .argument(IArgument.builder()
39            .name("arg1")
40            .type(IStringItem.type())
41            .zeroOrOne()
42            .build())
43        .argument(IArgument.builder()
44            .name("arg2")
45            .type(IStringItem.type())
46            .zeroOrOne()
47            .build())
48        .returnType(IBooleanItem.type())
49        .returnOne()
50        .functionHandler(FnEndsWith::execute)
51        .build();
52    // CPD-ON
53  
54    @SuppressWarnings("unused")
55    @NonNull
56    private static ISequence<IBooleanItem> execute(@NonNull IFunction function,
57        @NonNull List<ISequence<?>> arguments,
58        @NonNull DynamicContext dynamicContext,
59        IItem focus) {
60      IStringItem arg1 = FunctionUtils.asTypeOrNull(arguments.get(0).getFirstItem(true));
61  
62      IStringItem arg2 = FunctionUtils.asTypeOrNull(arguments.get(1).getFirstItem(true));
63  
64      return ISequence.of(IBooleanItem.valueOf(
65          fnEndsWith(
66              arg1 == null ? "" : arg1.asString(),
67              arg2 == null ? "" : arg2.asString())));
68    }
69  
70    /**
71     * Determine if the string provided in the first argument ends with the string
72     * in the second argument.
73     * <p>
74     * Based on the XPath 3.1 <a href=
75     * "https://www.w3.org/TR/xpath-functions-31/#func-ends-with">fn:ends-with</a>
76     * function.
77     *
78     * @param arg1
79     *          the string to examine
80     * @param arg2
81     *          the string to check as the leading substring
82     * @return {@code true} if {@code arg1} ends with {@code arg2}, or {@code false}
83     *         otherwise
84     */
85    public static boolean fnEndsWith(@NonNull String arg1, @NonNull String arg2) {
86      boolean retval;
87      if (arg2.isEmpty()) {
88        retval = true;
89      } else if (arg1.isEmpty()) {
90        retval = false;
91      } else {
92        retval = arg1.endsWith(arg2);
93      }
94      return retval;
95    }
96  
97    private FnEndsWith() {
98      // disable construction
99    }
100 }

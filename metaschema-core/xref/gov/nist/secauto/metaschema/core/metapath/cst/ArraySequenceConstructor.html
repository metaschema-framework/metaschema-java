6   package gov.nist.secauto.metaschema.core.metapath.cst;
7   
8   import gov.nist.secauto.metaschema.core.metapath.DynamicContext;
9   import gov.nist.secauto.metaschema.core.metapath.ISequence;
10  import gov.nist.secauto.metaschema.core.metapath.item.function.IArrayItem;
11  
12  import java.util.List;
13  
14  import edu.umd.cs.findbugs.annotations.Nullable;
15  
16  /**
17   * An implementation of the
18   * <a href="https://www.w3.org/TR/xpath-31/#id-array-constructors">Array Curly
19   * Constructor</a> supporting the creation of a Metapath {@link IArrayItem}.
20   */
21  public class ArraySequenceConstructor implements IExpression {
22    @Nullable
23    private final IExpression expr;
24  
25    /**
26     * Construct a new array constructor expression that uses the provided
27     * expression to initialize the array.
28     *
29     * @param expr
30     *          the expression used to produce the array members
31     */
32    public ArraySequenceConstructor(@Nullable IExpression expr) {
33      this.expr = expr;
34    }
35  
36    @SuppressWarnings("rawtypes")
37    @Override
38    public Class<IArrayItem> getBaseResultType() {
39      return IArrayItem.class;
40    }
41  
42    @SuppressWarnings("rawtypes")
43    @Override
44    public Class<IArrayItem> getStaticResultType() {
45      return IArrayItem.class;
46    }
47  
48    @SuppressWarnings("null")
49    @Override
50    public List<? extends IExpression> getChildren() {
51      return List.of(expr);
52    }
53  
54    @Override
55    public ISequence<IArrayItem<?>> accept(DynamicContext dynamicContext, ISequence<?> focus) {
56      ISequence<IArrayItem<?>> retval;
57      if (expr != null) {
58        IArrayItem<?> array = IArrayItem.ofCollection(expr.accept(dynamicContext, focus));
59        retval = ISequence.of(array);
60      } else {
61        retval = ISequence.of();
62      }
63      return retval;
64    }
65  
66    @Override
67    public <RESULT, CONTEXT> RESULT accept(IExpressionVisitor<RESULT, CONTEXT> visitor, CONTEXT context) {
68      return visitor.visitArray(this, context);
69    }
70  }

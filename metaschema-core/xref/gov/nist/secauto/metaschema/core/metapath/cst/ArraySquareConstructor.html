6   package gov.nist.secauto.metaschema.core.metapath.cst;
7   
8   import gov.nist.secauto.metaschema.core.metapath.DynamicContext;
9   import gov.nist.secauto.metaschema.core.metapath.ISequence;
10  import gov.nist.secauto.metaschema.core.metapath.item.IItem;
11  import gov.nist.secauto.metaschema.core.metapath.item.function.IArrayItem;
12  
13  import java.util.List;
14  
15  import edu.umd.cs.findbugs.annotations.NonNull;
16  
17  /**
18   * An implementation of the
19   * <a href="https://www.w3.org/TR/xpath-31/#id-array-constructors">Array Square
20   * Constructor</a> supporting the creation of a Metapath {@link IArrayItem}.
21   */
22  public class ArraySquareConstructor implements IExpression {
23    @NonNull
24    private final List<IExpression> children;
25  
26    /**
27     * Construct a new array constructor expression that uses the provided
28     * expression to initialize the array.
29     * <p>
30     * Each resulting array member contains the value of the corresponding argument
31     * expression.
32     *
33     * @param children
34     *          the expressions used to produce the array members
35     */
36    public ArraySquareConstructor(@NonNull List<IExpression> children) {
37      this.children = children;
38    }
39  
40    @Override
41    public List<? extends IExpression> getChildren() {
42      return children;
43    }
44  
45    @Override
46    public ISequence<? extends IItem> accept(DynamicContext dynamicContext, ISequence<?> focus) {
47      return ISequence.of(getChildren().stream()
48          .map(expr -> expr.accept(dynamicContext, focus))
49          .map(ISequence::toCollectionValue)
50          .collect(IArrayItem.toArrayItem()));
51    }
52  
53    @Override
54    public <RESULT, CONTEXT> RESULT accept(IExpressionVisitor<RESULT, CONTEXT> visitor, CONTEXT context) {
55      return visitor.visitArray(this, context);
56    }
57  }

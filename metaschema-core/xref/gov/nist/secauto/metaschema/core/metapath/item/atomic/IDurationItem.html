6   package gov.nist.secauto.metaschema.core.metapath.item.atomic;
7   
8   import gov.nist.secauto.metaschema.core.metapath.function.InvalidValueForCastFunctionException;
9   
10  import java.time.temporal.TemporalAmount;
11  
12  import edu.umd.cs.findbugs.annotations.NonNull;
13  
14  public interface IDurationItem extends IAnyAtomicItem {
15    /**
16     * Cast the provided type to this item type.
17     *
18     * @param item
19     *          the item to cast
20     * @return the original item if it is already this type, otherwise a new item
21     *         cast to this type
22     * @throws InvalidValueForCastFunctionException
23     *           if the provided {@code item} cannot be cast to this type
24     */
25    @NonNull
26    static IDurationItem cast(@NonNull IAnyAtomicItem item) {
27      IDurationItem retval;
28      if (item instanceof IDurationItem) {
29        retval = (IDurationItem) item;
30      } else {
31        try {
32          retval = IDayTimeDurationItem.valueOf(item.asString());
33        } catch (IllegalArgumentException ex) {
34          try {
35            retval = IYearMonthDurationItem.valueOf(item.asString());
36          } catch (IllegalArgumentException ex2) {
37            InvalidValueForCastFunctionException newEx = new InvalidValueForCastFunctionException(ex2);
38            newEx.addSuppressed(ex);
39            throw newEx; // NOPMD context as suppressed
40          }
41        }
42      }
43      return retval;
44    }
45  
46    @Override
47    TemporalAmount getValue();
48  
49    @Override
50    default IDurationItem castAsType(IAnyAtomicItem item) {
51      return cast(item);
52    }
53  }

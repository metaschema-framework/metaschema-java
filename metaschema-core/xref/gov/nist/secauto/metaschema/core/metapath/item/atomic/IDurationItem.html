6   package gov.nist.secauto.metaschema.core.metapath.item.atomic;
7   
8   import gov.nist.secauto.metaschema.core.metapath.function.InvalidValueForCastFunctionException;
9   import gov.nist.secauto.metaschema.core.metapath.type.IAtomicOrUnionType;
10  import gov.nist.secauto.metaschema.core.metapath.type.impl.TypeConstants;
11  
12  import java.time.temporal.TemporalAmount;
13  
14  import edu.umd.cs.findbugs.annotations.NonNull;
15  
16  /**
17   * An atomic Metapath item representing a duration data value.
18   * <p>
19   * This interface supports both day-time and year-month duration formats
20   * following the ISO 8601 standard. Examples of valid durations include:
21   * <ul>
22   * <li>P1Y2M (1 year, 2 months)</li>
23   * <li>P3DT4H5M (3 days, 4 hours, 5 minutes)</li>
24   * </ul>
25   *
26   * @see IDayTimeDurationItem
27   * @see IYearMonthDurationItem
28   */
29  public interface IDurationItem extends IAnyAtomicItem {
30    /**
31     * Get the type information for this item.
32     *
33     * @return the type information
34     */
35    @NonNull
36    static IAtomicOrUnionType<IDurationItem> type() {
37      return TypeConstants.DURATION_TYPE;
38    }
39  
40    /**
41     * Cast the provided type to this item type.
42     *
43     * @param item
44     *          the item to cast
45     * @return the original item if it is already this type, otherwise a new item
46     *         cast to this type
47     * @throws InvalidValueForCastFunctionException
48     *           if the provided {@code item} cannot be cast to this type
49     */
50    @NonNull
51    static IDurationItem cast(@NonNull IAnyAtomicItem item) {
52      IDurationItem retval;
53      if (item instanceof IDurationItem) {
54        retval = (IDurationItem) item;
55      } else {
56        String value;
57        try {
58          value = item.asString();
59        } catch (IllegalStateException ex) {
60          // asString can throw IllegalStateException exception
61          throw new InvalidValueForCastFunctionException(ex);
62        }
63  
64        try {
65          retval = IDayTimeDurationItem.valueOf(value);
66        } catch (IllegalStateException ex) {
67          try {
68            retval = IYearMonthDurationItem.valueOf(value);
69          } catch (IllegalStateException ex2) {
70            InvalidValueForCastFunctionException newEx = new InvalidValueForCastFunctionException(
71                String.format("Value '%s' cannot be parsed as either a day-time or year-month duration", value),
72                ex2);
73            newEx.addSuppressed(ex);
74            throw newEx; // NOPMD context as suppressed
75          }
76        }
77      }
78      return retval;
79    }
80  
81    @Override
82    TemporalAmount getValue();
83  
84    @Override
85    default IDurationItem castAsType(IAnyAtomicItem item) {
86      return cast(item);
87    }
88  }

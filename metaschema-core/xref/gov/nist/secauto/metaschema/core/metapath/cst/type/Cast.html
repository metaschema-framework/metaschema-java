6   package gov.nist.secauto.metaschema.core.metapath.cst.type;
7   
8   import gov.nist.secauto.metaschema.core.metapath.DynamicContext;
9   import gov.nist.secauto.metaschema.core.metapath.ISequence;
10  import gov.nist.secauto.metaschema.core.metapath.cst.IExpression;
11  import gov.nist.secauto.metaschema.core.metapath.cst.IExpressionVisitor;
12  import gov.nist.secauto.metaschema.core.metapath.item.IItem;
13  import gov.nist.secauto.metaschema.core.metapath.type.IAtomicOrUnionType;
14  
15  import edu.umd.cs.findbugs.annotations.NonNull;
16  
17  /**
18   * A compact syntax tree node that supports the Metapath
19   * <a href="https://www.w3.org/TR/xpath-31/#id-cast">"cast as" operator</a>.
20   */
21  @SuppressWarnings("PMD.ShortClassName")
22  public class Cast
23      extends AbstractCastingExpression {
24  
25    /**
26     * Construct a new cast expression.
27     *
28     * @param value
29     *          the expression that will produce the item to cast
30     * @param type
31     *          the atomic type to cast to
32     * @param allowEmptySequence
33     *          {@code true} if the value expression is allowed to produce an empty
34     *          sequence, or {@code false} otherwise
35     */
36    public Cast(
37        @NonNull IExpression value,
38        @NonNull IAtomicOrUnionType<?> type,
39        boolean allowEmptySequence) {
40      super(value, type, allowEmptySequence);
41    }
42  
43    @Override
44    public Class<? extends IItem> getBaseResultType() {
45      return getType().getItemClass();
46    }
47  
48    @Override
49    public ISequence<? extends IItem> accept(DynamicContext dynamicContext, ISequence<?> focus) {
50      return cast(focus);
51    }
52  
53    @Override
54    public <RESULT, CONTEXT> RESULT accept(IExpressionVisitor<RESULT, CONTEXT> visitor, CONTEXT context) {
55      return visitor.visitCast(this, context);
56    }
57  
58  }

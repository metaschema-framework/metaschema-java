2   package gov.nist.secauto.metaschema.core.metapath.item.node;
3   
4   import gov.nist.secauto.metaschema.core.util.CollectionUtil;
5   
6   import java.util.Collection;
7   import java.util.List;
8   import java.util.Map;
9   
10  import javax.xml.namespace.QName;
11  
12  import edu.umd.cs.findbugs.annotations.NonNull;
13  
14  /**
15   * This mixin interface indicates that the implementation is a {@link INodeItem}
16   * that may have both flag and model children.
17   */
18  public interface IFeatureModelContainerItem extends IFeatureFlagContainerItem {
19  
20    @Override
21    ModelContainer getModel();
22  
23    @Override
24    default Collection<? extends List<? extends IModelNodeItem<?, ?>>> getModelItems() {
25      return getModel().getModelItems();
26    }
27  
28    @Override
29    default List<? extends IModelNodeItem<?, ?>> getModelItemsByName(QName name) {
30      return getModel().getModelItemsByName(name);
31    }
32  
33    /**
34     * Provides an abstract implementation of a lazy loaded model.
35     */
36    class ModelContainer
37        extends FlagContainer {
38      private final Map<QName, List<? extends IModelNodeItem<?, ?>>> modelItems;
39  
40      /**
41       * Creates a new collection of flags and model items.
42       *
43       * @param flags
44       *          a mapping of flag name to a flag item
45       * @param modelItems
46       *          a mapping of model item name to a list of model items
47       */
48      protected ModelContainer(
49          @NonNull Map<QName, IFlagNodeItem> flags,
50          @NonNull Map<QName, List<? extends IModelNodeItem<?, ?>>> modelItems) {
51        super(flags);
52        this.modelItems = modelItems;
53      }
54  
55      /**
56       * Get the matching list of model items having the provided name.
57       *
58       * @param name
59       *          the name of the model items to retrieve
60       * @return a lisy of matching model items or {@code null} if no match was found
61       */
62      @NonNull
63      public List<? extends IModelNodeItem<?, ?>> getModelItemsByName(@NonNull QName name) {
64        List<? extends IModelNodeItem<?, ?>> result = modelItems.get(name);
65        return result == null ? CollectionUtil.emptyList() : result;
66      }
67  
68      /**
69       * Get all model items grouped by model item name.
70       *
71       * @return a collection of lists containg model items grouped by names
72       */
73      @SuppressWarnings("null")
74      @NonNull
75      public Collection<List<? extends IModelNodeItem<?, ?>>> getModelItems() {
76        return modelItems.values();
77      }
78    }
79  }

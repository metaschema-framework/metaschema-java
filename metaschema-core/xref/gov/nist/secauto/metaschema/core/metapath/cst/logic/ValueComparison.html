6   package gov.nist.secauto.metaschema.core.metapath.cst.logic;
7   
8   import gov.nist.secauto.metaschema.core.metapath.DynamicContext;
9   import gov.nist.secauto.metaschema.core.metapath.ISequence;
10  import gov.nist.secauto.metaschema.core.metapath.cst.IExpression;
11  import gov.nist.secauto.metaschema.core.metapath.cst.IExpressionVisitor;
12  import gov.nist.secauto.metaschema.core.metapath.function.ComparisonFunctions;
13  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IAnyAtomicItem;
14  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IBooleanItem;
15  
16  import edu.umd.cs.findbugs.annotations.NonNull;
17  import edu.umd.cs.findbugs.annotations.Nullable;
18  
19  /**
20   * Supports Metapath
21   * <a href="https://www.w3.org/TR/xpath-31/#id-value-comparisons">value
22   * comparisons</a>.
23   */
24  public class ValueComparison
25      extends AbstractComparison {
26  
27    /**
28     * Create a new value comparison expression.
29     *
30     * @param left
31     *          the expression to compare against
32     * @param operator
33     *          the comparison operator
34     * @param right
35     *          the expression to compare with
36     */
37    public ValueComparison(
38        @NonNull IExpression left,
39        @NonNull ComparisonFunctions.Operator operator,
40        @NonNull IExpression right) {
41      super(left, operator, right);
42    }
43  
44    @Override
45    public <RESULT, CONTEXT> RESULT accept(IExpressionVisitor<RESULT, CONTEXT> visitor, CONTEXT context) {
46      return visitor.visitValueComparison(this, context);
47    }
48  
49    @Override
50    public ISequence<? extends IBooleanItem> accept(DynamicContext dynamicContext, ISequence<?> focus) {
51      IAnyAtomicItem left = getFirstDataItem(getLeft().accept(dynamicContext, focus), false);
52      IAnyAtomicItem right = getFirstDataItem(getRight().accept(dynamicContext, focus), false);
53  
54      return resultOrEmpty(left, right);
55    }
56  
57    /**
58     * Compare the two atomic items.
59     *
60     * @param leftItem
61     *          the first item to compare
62     * @param rightItem
63     *          the second item to compare
64     * @return a or an empty {@link ISequence} if either item is {@code null}
65     */
66    @NonNull
67    protected ISequence<? extends IBooleanItem> resultOrEmpty(@Nullable IAnyAtomicItem leftItem,
68        @Nullable IAnyAtomicItem rightItem) {
69      ISequence<? extends IBooleanItem> retval;
70      if (leftItem == null || rightItem == null) {
71        retval = ISequence.empty();
72      } else {
73        IBooleanItem result = ComparisonFunctions.valueCompairison(leftItem, getOperator(), rightItem);
74        retval = ISequence.of(result);
75      }
76      return retval;
77    }
78  
79  }

6   package gov.nist.secauto.metaschema.core.metapath.item.atomic;
7   
8   import gov.nist.secauto.metaschema.core.datatype.adapter.MetaschemaDataTypeProvider;
9   import gov.nist.secauto.metaschema.core.metapath.function.InvalidValueForCastFunctionException;
10  import gov.nist.secauto.metaschema.core.metapath.item.atomic.impl.TimeWithTimeZoneItemImpl;
11  import gov.nist.secauto.metaschema.core.metapath.type.IAtomicOrUnionType;
12  import gov.nist.secauto.metaschema.core.metapath.type.InvalidTypeMetapathException;
13  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
14  
15  import java.time.OffsetTime;
16  import java.time.ZonedDateTime;
17  
18  import edu.umd.cs.findbugs.annotations.NonNull;
19  
20  /**
21   * An atomic Metapath item representing a time value in the Metapath system with
22   * an explicit time zone.
23   * <p>
24   * This interface provides functionality for handling time values with time zone
25   * information, supporting parsing, casting, and comparison operations. It works
26   * in conjunction with {@link ZonedDateTime} to eliminate time zone ambiguity.
27   */
28  public interface ITimeWithTimeZoneItem extends ITimeItem {
29    /**
30     * Get the type information for this item.
31     *
32     * @return the type information
33     */
34    @NonNull
35    static IAtomicOrUnionType<ITimeWithTimeZoneItem> type() {
36      return MetaschemaDataTypeProvider.TIME_WITH_TZ.getItemType();
37    }
38  
39    @Override
40    default IAtomicOrUnionType<ITimeWithTimeZoneItem> getType() {
41      return type();
42    }
43  
44    /**
45     * Construct a new time item using the provided string {@code value}.
46     *
47     * @param value
48     *          a string representing a time
49     * @return the new item
50     */
51    @NonNull
52    static ITimeWithTimeZoneItem valueOf(@NonNull String value) {
53      try {
54        return valueOf(MetaschemaDataTypeProvider.TIME_WITH_TZ.parse(value));
55      } catch (IllegalArgumentException ex) {
56        throw new InvalidTypeMetapathException(
57            null,
58            String.format("Invalid time value '%s'. %s",
59                value,
60                ex.getLocalizedMessage()),
61            ex);
62      }
63    }
64  
65    /**
66     * Construct a new time item using the provided {@code value}.
67     * <p>
68     * This method handles dates with explicit timezone information using
69     * ZonedDateTime. The timezone is preserved as specified in the input and is
70     * significant for time operations and comparisons.
71     *
72     * @param value
73     *          a time, with time zone information
74     * @return the new item
75     */
76    @NonNull
77    static ITimeWithTimeZoneItem valueOf(@NonNull ZonedDateTime value) {
78      return new TimeWithTimeZoneItemImpl(ObjectUtils.notNull(value.toOffsetDateTime().toOffsetTime()));
79    }
80  
81    /**
82     * Construct a new time item using the provided {@code value}.
83     * <p>
84     * This method handles dates with explicit timezone information using
85     * ZonedDateTime. The timezone is preserved as specified in the input and is
86     * significant for time operations and comparisons.
87     *
88     * @param value
89     *          a time, with time zone information
90     * @return the new item
91     */
92    @NonNull
93    static ITimeWithTimeZoneItem valueOf(@NonNull OffsetTime value) {
94      return new TimeWithTimeZoneItemImpl(value);
95    }
96  
97    /**
98     * Cast the provided type to this item type.
99     *
100    * @param item
101    *          the item to cast
102    * @return the original item if it is already this type, otherwise a new item
103    *         cast to this type
104    * @throws InvalidValueForCastFunctionException
105    *           if the provided {@code item} cannot be cast to this type
106    */
107   @NonNull
108   static ITimeWithTimeZoneItem cast(@NonNull IAnyAtomicItem item) {
109     ITimeWithTimeZoneItem retval;
110     if (item instanceof ITimeWithTimeZoneItem) {
111       retval = (ITimeWithTimeZoneItem) item;
112     } else if (item instanceof ITemporalItem) {
113       retval = fromTemporal((ITemporalItem) item);
114     } else if (item instanceof IStringItem || item instanceof IUntypedAtomicItem) {
115       try {
116         retval = valueOf(item.asString());
117       } catch (IllegalStateException | InvalidTypeMetapathException ex) {
118         // asString can throw IllegalStateException exception
119         throw new InvalidValueForCastFunctionException(ex);
120       }
121     } else {
122       throw new InvalidValueForCastFunctionException(
123           String.format("unsupported item type '%s'", item.getClass().getName()));
124     }
125     return retval;
126   }
127 
128   @NonNull
129   private static ITimeWithTimeZoneItem fromTemporal(@NonNull ITemporalItem temporal) {
130     if (!temporal.hasTimezone()) {
131       // asString can throw IllegalStateException exception
132       throw new InvalidValueForCastFunctionException(
133           String.format("Unable to cast the temporal value '%s', since it lacks timezone information.",
134               temporal.asString()));
135     }
136     // get the time
137     return valueOf(ObjectUtils.notNull(temporal.asZonedDateTime().toOffsetDateTime().toOffsetTime()));
138   }
139 
140   @Override
141   default ITimeWithTimeZoneItem castAsType(IAnyAtomicItem item) {
142     return cast(item);
143   }
144 }

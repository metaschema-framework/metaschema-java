6   package gov.nist.secauto.metaschema.core.metapath.cst;
7   
8   import java.util.List;
9   import java.util.Objects;
10  
11  import edu.umd.cs.findbugs.annotations.NonNull;
12  
13  /**
14   * An immutable expression with a single sub-expression.
15   */
16  public abstract class AbstractUnaryExpression
17      extends AbstractExpression {
18    @NonNull
19    private final IExpression expr;
20  
21    /**
22     * Construct a new unary expression.
23     *
24     * @param expr
25     *          the single sub-expression
26     */
27    public AbstractUnaryExpression(@NonNull IExpression expr) {
28      this.expr = Objects.requireNonNull(expr, "expr");
29    }
30  
31    /**
32     * Retrieve the single child sub-expression.
33     *
34     * @return the sub-expression
35     */
36    @NonNull
37    public IExpression getChild() {
38      return expr;
39    }
40  
41    @SuppressWarnings("null")
42    @Override
43    public List<? extends IExpression> getChildren() {
44      return List.of(expr);
45    }
46  }

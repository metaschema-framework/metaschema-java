17  /**
18   * A Metapath node valued item representing a Metaschema module field.
19   */
20  public interface IFieldNodeItem
21      extends IModelNodeItem<IFieldDefinition, IFieldInstance>,
22      IAtomicValuedItem {
23    @Override
24    default NodeItemKind getNodeItemKind() {
25      return NodeItemKind.FIELD;
26    }
27  
28    @Override
29    default IFieldNodeItem getNodeItem() {
30      return this;
31    }
32  
33    @Override
34    default IKindTest<IFieldNodeItem> getType() {
35      StaticContext staticContext = getStaticContext();
36      return IItemType.field(
37          getQName(),
38          getDefinition().getDefinitionQName().toEQName(staticContext),
39          staticContext);
40    }
41  
42    @Override
43    @Nullable
44    default URI getBaseUri() {
45      INodeItem parent = getParentNodeItem();
46      return parent == null ? null : parent.getBaseUri();
47    }
48  
49    @Override
50    default @NonNull
51    String format(@NonNull IPathFormatter formatter) {
52      return formatter.formatField(this);
53    }
54  
55    @Override
56    default <CONTEXT, RESULT> RESULT accept(@NonNull INodeItemVisitor<CONTEXT, RESULT> visitor, CONTEXT context) {
57      return visitor.visitField(this, context);
58    }
59  }

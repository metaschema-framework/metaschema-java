6   package gov.nist.secauto.metaschema.core.metapath.item.atomic.impl;
7   
8   import gov.nist.secauto.metaschema.core.datatype.adapter.IPv4AddressAdapter;
9   import gov.nist.secauto.metaschema.core.datatype.adapter.MetaschemaDataTypeProvider;
10  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IIPv4AddressItem;
11  
12  import edu.umd.cs.findbugs.annotations.NonNull;
13  import inet.ipaddr.ipv4.IPv4Address;
14  
15  /**
16   * An implementation of a Metapath atomic item containing an IPv4 address data
17   * value.
18   */
19  public class IPv4AddressItemImpl
20      extends AbstractIPAddressItem<IPv4Address>
21      implements IIPv4AddressItem {
22    /**
23     * Construct a new item with the provided {@code value}.
24     *
25     * @param value
26     *          the value to wrap
27     */
28    public IPv4AddressItemImpl(@NonNull IPv4Address value) {
29      super(value);
30    }
31  
32    @Override
33    public IPv4AddressAdapter getJavaTypeAdapter() {
34      return MetaschemaDataTypeProvider.IP_V4_ADDRESS;
35    }
36  
37    @Override
38    public IPv4Address asIpAddress() {
39      return getValue();
40    }
41  }

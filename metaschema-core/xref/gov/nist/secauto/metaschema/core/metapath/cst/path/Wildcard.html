6   package gov.nist.secauto.metaschema.core.metapath.cst.path;
7   
8   import gov.nist.secauto.metaschema.core.metapath.DynamicContext;
9   import gov.nist.secauto.metaschema.core.metapath.ISequence;
10  import gov.nist.secauto.metaschema.core.metapath.cst.IExpressionVisitor;
11  import gov.nist.secauto.metaschema.core.metapath.item.ItemUtils;
12  import gov.nist.secauto.metaschema.core.metapath.item.node.IDefinitionNodeItem;
13  import gov.nist.secauto.metaschema.core.metapath.item.node.INodeItem;
14  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
15  
16  import java.util.function.Predicate;
17  import java.util.stream.Stream;
18  
19  import edu.umd.cs.findbugs.annotations.NonNull;
20  import edu.umd.cs.findbugs.annotations.Nullable;
21  
22  /**
23   * The CST node for a Metapath
24   * <a href="https://www.w3.org/TR/xpath-31/#doc-xpath31-Wildcard">wildcard name
25   * test</a>.
26   */
27  public class Wildcard implements INameTestExpression {
28    @Nullable
29    private final Predicate<IDefinitionNodeItem<?, ?>> matcher;
30  
31    /**
32     * Construct a new wildcard name test expression using the provided matcher.
33     *
34     * @param matcher
35     *          the matcher used to determine matching nodes
36     */
37    public Wildcard(@Nullable IWildcardMatcher matcher) {
38      this.matcher = matcher;
39    }
40  
41    @Override
42    public <RESULT, CONTEXT> RESULT accept(IExpressionVisitor<RESULT, CONTEXT> visitor, CONTEXT context) {
43      return visitor.visitWildcard(this, context);
44    }
45  
46    @Override
47    public ISequence<? extends INodeItem> accept(
48        DynamicContext dynamicContext, ISequence<?> focus) {
49      Stream<? extends INodeItem> nodes = ObjectUtils.notNull(focus.stream().map(ItemUtils::checkItemIsNodeItemForStep));
50      return ISequence.of(match(nodes));
51    }
52  
53    /**
54     * Check the provided items to determine if each item matches the wildcard. All
55     * items that match are returned.
56     * <p>
57     * This is an intermediate stream operation.
58     *
59     * @param <T>
60     *          the item Java type
61     * @param items
62     *          the items to check if they match
63     * @return the matching items
64     */
65    @NonNull
66    public <T extends INodeItem> Stream<T> match(@SuppressWarnings("resource") @NonNull Stream<T> items) {
67      Stream<T> nodes = items;
68      if (matcher != null) {
69        Predicate<IDefinitionNodeItem<?, ?>> test = matcher;
70        nodes = ObjectUtils.notNull(nodes.filter(item -> {
71          assert matcher != null;
72          return !(item instanceof IDefinitionNodeItem) ||
73              test.test((IDefinitionNodeItem<?, ?>) item);
74        }));
75      }
76      return nodes;
77    }
78  
79    @SuppressWarnings("null")
80    @Override
81    public String toASTString() {
82      return String.format("%s[%s]", getClass().getName(), matcher == null ? "*:*" : matcher.toString());
83    }
84  
85  }

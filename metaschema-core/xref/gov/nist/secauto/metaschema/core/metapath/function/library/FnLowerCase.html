6   package gov.nist.secauto.metaschema.core.metapath.function.library;

8   import gov.nist.secauto.metaschema.core.metapath.DynamicContext;
9   import gov.nist.secauto.metaschema.core.metapath.ISequence;
10  import gov.nist.secauto.metaschema.core.metapath.MetapathConstants;
11  import gov.nist.secauto.metaschema.core.metapath.function.FunctionUtils;
12  import gov.nist.secauto.metaschema.core.metapath.function.IArgument;
13  import gov.nist.secauto.metaschema.core.metapath.function.IFunction;
14  import gov.nist.secauto.metaschema.core.metapath.item.IItem;
15  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IStringItem;
16  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
17  
18  import java.util.List;
19  import java.util.Locale;
20  
21  import edu.umd.cs.findbugs.annotations.NonNull;
22  
23  /**
24   * Based on the XPath 3.1 <a href=
25   * "https://www.w3.org/TR/xpath-functions-31/#func-lower-case">fn:lower-case</a>
26   * functions.
27   */
28  public final class FnLowerCase {
29    private static final String NAME = "lower-case";
30  
31    @NonNull
32    static final IFunction SIGNATURE_ONE_ARG = IFunction.builder()
33        .name(NAME)
34        .namespace(MetapathConstants.NS_METAPATH_FUNCTIONS)
35        .deterministic()
36        .contextIndependent()
37        .focusIndependent()
38        .argument(IArgument.builder()
39            .name("arg")
40            .type(IStringItem.class)
41            .zeroOrOne()
42            .build())
43        .returnType(IStringItem.class)
44        .returnOne()
45        .functionHandler(FnLowerCase::executeOneArg)
46        .build();
47  
48    private FnLowerCase() {
49      // disable construction
50    }
51  
52    @SuppressWarnings("unused")
53    @NonNull
54    private static ISequence<IStringItem> executeOneArg(@NonNull IFunction function,
55        @NonNull List<ISequence<?>> arguments, @NonNull DynamicContext dynamicContext, IItem focus) {
56  
57      // From the XPath 3.1 specification:
58      // If the value of $arg is the empty sequence, the zero-length string is
59      // returned.
60      return ISequence.of(arguments.get(0).isEmpty()
61          ? IStringItem.valueOf("")
62          : fnLowerCase(FunctionUtils.asType(ObjectUtils.notNull(arguments.get(0).getFirstItem(true)))));
63    }
64  
65    /**
66     * An implementation of <a href=
67     * "https://www.w3.org/TR/xpath-functions-31/#func-lower-case">fn:lower-case</a>.
68     *
69     * @param arg
70     *          the string to be converted to lower case
71     * @return the resulting string in lower case
72     */
73    @NonNull
74    public static IStringItem fnLowerCase(@NonNull IStringItem arg) {
75      return IStringItem.valueOf(ObjectUtils.notNull(arg.toString().toLowerCase(Locale.ROOT)));
76    }
77  }

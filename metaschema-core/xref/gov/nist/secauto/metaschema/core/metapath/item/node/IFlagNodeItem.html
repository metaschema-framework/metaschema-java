22  /**
23   * A Metapath node valued item representing a Metaschema module flag.
24   */
25  
26  public interface IFlagNodeItem
27      extends IDefinitionNodeItem<IFlagDefinition, IFlagInstance>, IAtomicValuedItem {
28    @Override
29    default NodeItemKind getNodeItemKind() {
30      return NodeItemKind.FLAG;
31    }
32  
33    @Override
34    default IFlagNodeItem getNodeItem() {
35      return this;
36    }
37  
38    @Override
39    IFlagDefinition getDefinition();
40  
41    @Override
42    IFlagInstance getInstance();
43  
44    @Override
45    default IKindTest<IFlagNodeItem> getType() {
46      StaticContext staticContext = getStaticContext();
47      return IItemType.flag(
48          getQName(),
49          getDefinition().getDefinitionQName().toEQName(staticContext),
50          staticContext);
51    }
52  
53    @Override
54    @Nullable
55    default URI getBaseUri() {
56      INodeItem parent = getParentNodeItem();
57      return parent == null ? null : parent.getBaseUri();
58    }
59  
60    /**
61     * FlagContainer do not have flag items. This call should return an empty
62     * collection.
63     */
64    @SuppressWarnings("null")
65    @Override
66    default Collection<? extends IFlagNodeItem> getFlags() {
67      // a flag does not have flags
68      return Collections.emptyList();
69    }
70  
71    /**
72     * FlagContainer do not have flag items. This call should return {@code null}.
73     */
74    @Override
75    default IFlagNodeItem getFlagByName(@NonNull IEnhancedQName name) {
76      // a flag does not have flags
77      return null;
78    }
79  
80    /**
81     * FlagContainer do not have flag items. This call should return an empty
82     * stream.
83     */
84    @SuppressWarnings("null")
85    @Override
86    default @NonNull
87    Stream<? extends IFlagNodeItem> flags() {
88      // a flag does not have flags
89      return Stream.empty();
90    }
91  
92    /**
93     * FlagContainer do not have model items. This call should return an empty
94     * collection.
95     */
96    @SuppressWarnings("null")
97    @Override
98    default @NonNull
99    Collection<? extends List<? extends IModelNodeItem<?, ?>>> getModelItems() {
100     // a flag does not have model items
101     return Collections.emptyList();
102   }
103 
104   /**
105    * FlagContainer do not have model items. This call should return an empty list.
106    */
107   @SuppressWarnings("null")
108   @Override
109   default List<? extends IModelNodeItem<?, ?>> getModelItemsByName(IEnhancedQName name) {
110     // a flag does not have model items
111     return Collections.emptyList();
112   }
113 
114   /**
115    * FlagContainer do not have model items. This call should return an empty
116    * stream.
117    */
118   @SuppressWarnings("null")
119   @NonNull
120   @Override
121   default Stream<? extends IModelNodeItem<?, ?>> modelItems() {
122     // a flag does not have model items
123     return Stream.empty();
124   }
125 
126   @Override
127   default @NonNull
128   String format(@NonNull IPathFormatter formatter) {
129     return formatter.formatFlag(this);
130   }
131 
132   @Override
133   default <CONTEXT, RESULT> RESULT accept(@NonNull INodeItemVisitor<CONTEXT, RESULT> visitor, CONTEXT context) {
134     return visitor.visitFlag(this, context);
135   }
136 }

9   import java.net.URI;
10  import java.util.List;
11  import java.util.stream.Collectors;
12  import java.util.stream.Stream;
13  
14  import javax.xml.namespace.QName;
15  
16  import edu.umd.cs.findbugs.annotations.NonNull;
17  
18  /**
19   * Supports querying of global definitions and associated instances in a
20   * Metaschema module by effective name.
21   * <p>
22   * All definitions in the {@link ModuleScopeEnum#INHERITED} scope. This allows
23   * the exported structure of the Metaschema module to be queried.
24   */
25  public interface IModuleNodeItem extends IDocumentBasedNodeItem, IFeatureNoDataValuedItem {
26  
27    /**
28     * The Metaschema module this item is based on.
29     *
30     * @return the Metaschema module
31     */
32    @NonNull
33    IModule getModule();
34  
35    @Override
36    default URI getDocumentUri() {
37      return getModule().getLocation();
38    }
39  
40    @Override
41    default NodeItemType getNodeItemType() {
42      return NodeItemType.METASCHEMA;
43    }
44  
45    @Override
46    default IModuleNodeItem getNodeItem() {
47      return this;
48    }
49  
50    /**
51     * Get the root items having the provided {@code name}.
52     *
53     * @param name
54     *          the root item's name to retrieve
55     * @return a list of matching root items
56     */
57    // TODO: delete
58    default List<? extends IRootAssemblyNodeItem> getRootNodeItemByName(@NonNull QName name) {
59      List<? extends IModelNodeItem<?, ?>> result = getModelItemsByName(name);
60      return result.stream().flatMap(item -> {
61        IRootAssemblyNodeItem retval = null;
62        if (item instanceof IRootAssemblyNodeItem) {
63          retval = (IRootAssemblyNodeItem) item;
64        }
65  
66        return retval == null ? null : Stream.of(retval);
67      }).collect(Collectors.toUnmodifiableList());
68    }
69  
70    @Override
71    default String format(@NonNull IPathFormatter formatter) {
72      return formatter.formatMetaschema(this);
73    }
74  
75    @Override
76    default <CONTEXT, RESULT> RESULT accept(@NonNull INodeItemVisitor<CONTEXT, RESULT> visitor, CONTEXT context) {
77      return visitor.visitMetaschema(this, context);
78    }
79  
80    @Override
81    default StaticContext getStaticContext() {
82      return getModule().getModuleStaticContext();
83    }
84  }

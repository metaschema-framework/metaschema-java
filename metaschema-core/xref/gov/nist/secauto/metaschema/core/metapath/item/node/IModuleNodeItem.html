2   package gov.nist.secauto.metaschema.core.metapath.item.node;
3   
4   import gov.nist.secauto.metaschema.core.metapath.StaticContext;
5   import gov.nist.secauto.metaschema.core.metapath.format.IPathFormatter;
6   import gov.nist.secauto.metaschema.core.model.IModule;
7   
8   import java.net.URI;
9   
10  import edu.umd.cs.findbugs.annotations.NonNull;
11  
12  /**
13   * Supports querying of global definitions and associated instances in a
14   * Metaschema module by effective name.
15   * <p>
16   * All definitions in the
17   * {@link gov.nist.secauto.metaschema.core.model.IDefinition.ModuleScope#PUBLIC}
18   * are visible. This allows the exported structure of the Metaschema module to
19   * be queried.
20   */
21  public interface IModuleNodeItem extends IDocumentBasedNodeItem, IFeatureNoDataValuedItem {
22  
23    /**
24     * The Metaschema module this item is based on.
25     *
26     * @return the Metaschema module
27     */
28    @NonNull
29    IModule getModule();
30  
31    @Override
32    default URI getDocumentUri() {
33      return getModule().getLocation();
34    }
35  
36    @Override
37    default NodeItemType getNodeItemType() {
38      return NodeItemType.METASCHEMA;
39    }
40  
41    @Override
42    default IModuleNodeItem getNodeItem() {
43      return this;
44    }
45  
46    @Override
47    default String format(@NonNull IPathFormatter formatter) {
48      return formatter.formatMetaschema(this);
49    }
50  
51    @Override
52    default <CONTEXT, RESULT> RESULT accept(@NonNull INodeItemVisitor<CONTEXT, RESULT> visitor, CONTEXT context) {
53      return visitor.visitMetaschema(this, context);
54    }
55  
56    @Override
57    default StaticContext getStaticContext() {
58      return getModule().getModuleStaticContext();
59    }
60  }

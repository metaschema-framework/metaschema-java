6   package gov.nist.secauto.metaschema.core.metapath.item.atomic;
7   
8   import gov.nist.secauto.metaschema.core.datatype.adapter.MetaschemaDataTypeProvider;
9   import gov.nist.secauto.metaschema.core.metapath.function.InvalidValueForCastFunctionException;
10  import gov.nist.secauto.metaschema.core.metapath.item.atomic.impl.YearMonthDurationItemImpl;
11  import gov.nist.secauto.metaschema.core.metapath.type.IAtomicOrUnionType;
12  import gov.nist.secauto.metaschema.core.metapath.type.InvalidTypeMetapathException;
13  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
14  
15  import java.time.Period;
16  
17  import edu.umd.cs.findbugs.annotations.NonNull;
18  
19  /**
20   * An atomic Metapath item containing a duration data value in years, months,
21   * and days.
22   */
23  public interface IYearMonthDurationItem extends IDurationItem {
24    /**
25     * Get the type information for this item.
26     *
27     * @return the type information
28     */
29    @NonNull
30    static IAtomicOrUnionType<IYearMonthDurationItem> type() {
31      return MetaschemaDataTypeProvider.YEAR_MONTH_DURATION.getItemType();
32    }
33  
34    @Override
35    default IAtomicOrUnionType<IYearMonthDurationItem> getType() {
36      return type();
37    }
38  
39    /**
40     * Construct a new year month day duration item using the provided string
41     * {@code value}.
42     *
43     * @param value
44     *          a string representing a year month day duration
45     * @return the new item
46     * @throws InvalidTypeMetapathException
47     *           if the provided string value is not a day/time duration value
48     *           according to ISO 8601
49     */
50    @NonNull
51    static IYearMonthDurationItem valueOf(@NonNull String value) {
52      try {
53        Period period = ObjectUtils.notNull(MetaschemaDataTypeProvider.YEAR_MONTH_DURATION.parse(value).withDays(0));
54        return valueOf(period);
55      } catch (IllegalArgumentException ex) {
56        throw new InvalidTypeMetapathException(
57            null,
58            String.format("Invalid year/month duration value '%s'. %s",
59                value,
60                ex.getLocalizedMessage()),
61            ex);
62      }
63    }
64  
65    /**
66     * Construct a new year month day duration item using the provided
67     * {@code value}.
68     *
69     * @param value
70     *          a duration
71     * @return the new item
72     */
73    @NonNull
74    static IYearMonthDurationItem valueOf(@NonNull Period value) {
75      return new YearMonthDurationItemImpl(ObjectUtils.notNull(value.withDays(0)));
76    }
77  
78    /**
79     * Construct a new year month day duration item using the provided values.
80     *
81     * @param years
82     *          the number of years in the period
83     * @param months
84     *          the number of months in the period
85     * @return the new item
86     */
87    @NonNull
88    static IYearMonthDurationItem valueOf(int years, int months) {
89      return valueOf(ObjectUtils.notNull(Period.of(years, months, 0)));
90    }
91  
92    /**
93     * Cast the provided type to this item type.
94     *
95     * @param item
96     *          the item to cast
97     * @return the original item if it is already this type, otherwise a new item
98     *         cast to this type
99     * @throws InvalidValueForCastFunctionException
100    *           if the provided {@code item} cannot be cast to this type
101    */
102   @NonNull
103   static IYearMonthDurationItem cast(@NonNull IAnyAtomicItem item) {
104     try {
105       return item instanceof IYearMonthDurationItem
106           ? (IYearMonthDurationItem) item
107           : valueOf(item.asString());
108     } catch (IllegalStateException | InvalidTypeMetapathException ex) {
109       // asString can throw IllegalStateException exception
110       throw new InvalidValueForCastFunctionException(ex);
111     }
112   }
113 
114   /**
115    * Get the "wrapped" duration value.
116    *
117    * @return the underlying duration value
118    */
119   @NonNull
120   Period asPeriod();
121 
122   @Override
123   default IYearMonthDurationItem castAsType(IAnyAtomicItem item) {
124     return cast(item);
125   }
126 
127   @Override
128   default int compareTo(IAnyAtomicItem item) {
129     return compareTo(cast(item));
130   }
131 
132   /**
133    * Compares this value with the argument.
134    *
135    * @param item
136    *          the item to compare with this value
137    * @return a negative integer, zero, or a positive integer if this value is less
138    *         than, equal to, or greater than the {@code item}.
139    */
140   default int compareTo(IYearMonthDurationItem item) {
141     Period thisPeriod = asPeriod().normalized();
142     Period thatPeriod = item.asPeriod().normalized();
143 
144     int result = Integer.compare(thisPeriod.getYears(), thatPeriod.getYears());
145     return result == 0 ? Integer.compare(thisPeriod.getMonths(), thatPeriod.getMonths()) : result;
146   }
147 }

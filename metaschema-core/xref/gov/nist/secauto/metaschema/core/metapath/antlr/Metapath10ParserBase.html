6   package gov.nist.secauto.metaschema.core.metapath.antlr;
7   
8   import org.antlr.v4.runtime.Parser;
9   import org.antlr.v4.runtime.TokenStream;
10  
11  import java.util.Set;
12  
13  /**
14   * Base class for Metapath ANTLR-based parsing operations. This class extends
15   * the ANTLR Parser to provide specialized processing for Metapath expressions.
16   * <p>
17   * ANTLR (ANother Tool for Language Recognition) is used here to generate the
18   * parser for Metapath expressions. This base class provides common
19   * functionality that can be used by the generated parser implementation.
20   * <p>
21   * Implementations should extend this class to define specific parsing rules and
22   * behaviors for Metapath processing.
23   */
24  public abstract class Metapath10ParserBase
25      extends Parser {
26  
27    private static final Set<Integer> KEYWORD_TOKENS = Set.of(
28        Metapath10.KW_ARRAY,
29        Metapath10.KW_FLAG,
30        Metapath10.KW_COMMENT,
31        Metapath10.KW_DOCUMENT_NODE,
32        Metapath10.KW_FIELD,
33        Metapath10.KW_ASSEMBLY,
34        Metapath10.KW_EMPTY_SEQUENCE,
35        Metapath10.KW_FUNCTION,
36        Metapath10.KW_IF,
37        Metapath10.KW_ITEM,
38        Metapath10.KW_MAP,
39        Metapath10.KW_NAMESPACE_NODE,
40        Metapath10.KW_NODE,
41        Metapath10.KW_PROCESSING_INSTRUCTION,
42        Metapath10.KW_SCHEMA_ATTRIBUTE,
43        Metapath10.KW_SCHEMA_ELEMENT,
44        Metapath10.KW_TEXT);
45  
46    /**
47     * Construct a new parser base.
48     *
49     * @param input
50     *          the input token stream
51     */
52    protected Metapath10ParserBase(TokenStream input) {
53      super(input);
54    }
55  
56    /**
57     * Check if functional call name does not include a keyword.
58     *
59     * @return {@code true} if the function call name is free of keywords, or
60     *         {@code false} otherwise
61     */
62    protected boolean isFuncCall() {
63      int nextToken = getInputStream().LA(1);
64      return !KEYWORD_TOKENS.contains(nextToken);
65    }
66  }

6   package gov.nist.secauto.metaschema.core.metapath.function;
7   
8   import gov.nist.secauto.metaschema.core.metapath.DynamicContext;
9   import gov.nist.secauto.metaschema.core.metapath.ISequence;
10  import gov.nist.secauto.metaschema.core.metapath.MetapathException;
11  import gov.nist.secauto.metaschema.core.metapath.item.IItem;
12  
13  import java.util.List;
14  
15  import edu.umd.cs.findbugs.annotations.NonNull;
16  import edu.umd.cs.findbugs.annotations.Nullable;
17  
18  /**
19   * This functional interface provides a dispatch method for executing a function
20   * call.
21   */
22  @FunctionalInterface
23  public interface IFunctionExecutor {
24    /**
25     * Execute the provided function using the provided arguments, dynamic context,
26     * and focus.
27     *
28     * @param function
29     *          the signature of the function
30     * @param arguments
31     *          the function arguments
32     * @param dynamicContext
33     *          the dynamic evaluation context
34     * @param focus
35     *          the current focus
36     * @return a sequence containing the result of the execution
37     * @throws MetapathException
38     *           if an error occurred while executing the function
39     */
40    @NonNull
41    ISequence<?> execute(
42        @NonNull IFunction function,
43        @NonNull List<ISequence<?>> arguments,
44        @NonNull DynamicContext dynamicContext,
45        @Nullable IItem focus);
46  }

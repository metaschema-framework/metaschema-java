8   /**
9    * A mixin interface used to identify that the implementation is a
10   * {@link IModelNodeItem} that is based on a {@link IDefinition} that is an
11   * orphan in it's hierarchy. As a result, this item has no other siblings, since
12   * definitions cannot be instantiated.
13   * <p>
14   * This interface inherits the traits of the
15   * {@link IFeatureOrhpanedDefinitionNodeItem} interface.
16   *
17   * @param <D>
18   *          the definition's type
19   * @param <I>
20   *          the type of the instance that could be created from the definition
21   */
22  interface IFeatureOrhpanedDefinitionModelNodeItem<D extends IModelDefinition, I extends INamedModelInstance>
23      extends IModelNodeItem<D, I>, IFeatureOrhpanedDefinitionNodeItem<D, I> {
24  
25    @Override
26    default int getPosition() {
27      // always a singleton as a global definition
28      return 1;
29    }
30  
31    @Override
32    default IAssemblyNodeItem getParentContentNodeItem() {
33      // never has parent content as definition only
34      return null;
35    }
36  }

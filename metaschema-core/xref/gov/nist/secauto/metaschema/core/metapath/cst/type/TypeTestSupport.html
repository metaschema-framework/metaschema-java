58    /**
59     * Parse the provided sequence type.
60     *
61     * @param ctx
62     *          the sequence type parse tree
63     * @param staticContext
64     *          the static context used to resolve types and other information
65     * @return the parse sequence type
66     */
67    @NonNull
68    public static ISequenceType parseSequenceType(
69        @NonNull Metapath10.SequencetypeContext ctx,
70        @NonNull StaticContext staticContext) {
71      return ObjectUtils.notNull(ctx.KW_EMPTY_SEQUENCE() == null
72          ? parseSequenceType(
73              ObjectUtils.requireNonNull(ctx.itemtype()),
74              ctx.occurrenceindicator(),
75              staticContext)
76          : ISequenceType.empty());
77    }
78  
79    private static ISequenceType parseSequenceType(
80        @NonNull Metapath10.ItemtypeContext itemTypeCtx,
81        @Nullable Metapath10.OccurrenceindicatorContext occurrenceIndicatorCtx,
82        @NonNull StaticContext staticContext) {
83      IItemType itemType = parseItemType(itemTypeCtx, staticContext);
84      Occurrence occurrence = parseOccurrence(occurrenceIndicatorCtx);
85      return ISequenceType.of(itemType, occurrence);
86    }

88    @NonNull
89    private static IItemType parseItemType(
90        @NonNull Metapath10.ItemtypeContext ctx,
91        @NonNull StaticContext staticContext) {
92      IItemType retval;
93      if (ctx.KW_ITEM() == null) {
94        ParseTree tree = ctx.getChild(0);
95        retval = ITEM_TYPE_HANDLER_MAP.get(tree.getClass()).parse(tree, staticContext);
96      } else {
97        retval = IItemType.item();
98      }
99      return retval;
100   }

122   private static IKindTest<IAssemblyNodeItem> parseKindAssemblyTest(
123       @NonNull ParseTree tree,
124       @NonNull StaticContext staticContext) {
125 
126     Metapath10.AssemblytestContext ctx = (Metapath10.AssemblytestContext) tree;
127     IEnhancedQName instanceName
128         = ctx.elementnameorwildcard() == null || ctx.elementnameorwildcard().elementname() == null
129             ? null
130             : staticContext.parseModelName(ObjectUtils.notNull(
131                 ctx.elementnameorwildcard().elementname().eqname().getText()));
132     String typeName = ctx.typename_() == null ? null : ctx.typename_().eqname().getText();
133     return instanceName == null
134         ? typeName == null
135             ? IItemType.assembly()
136             : IItemType.assembly(typeName, staticContext)
137         : IItemType.assembly(instanceName, typeName, staticContext);
138   }

195   @NonNull
196   private static IItemType parseMapTest(
197       @NonNull ParseTree tree,
198       @NonNull StaticContext staticContext) {
199     Metapath10.MaptestContext ctx = (Metapath10.MaptestContext) tree;
200 
201     ctx.anymaptest();
202     ctx.typedmaptest();
203 
204     IItemType retval;
205     if (ctx.anymaptest() != null) {
206       retval = IItemType.map();
207     } else {
208       Metapath10.TypedmaptestContext typedMapCtx = ctx.typedmaptest();
209 
210       String dataTypeName = ObjectUtils.notNull(typedMapCtx.atomicoruniontype().getText());
211       IAtomicOrUnionType<?> dataType = staticContext.lookupAtomicType(dataTypeName);
212       retval = IItemType.map(
213           dataType,
214           parseSequenceType(ObjectUtils.requireNonNull(typedMapCtx.sequencetype()), staticContext));
215     }
216     return retval;
217   }

254   /**
255    * Parse the occurrence indicator.
256    *
257    * @param ctx
258    *          the occurrence parse tree
259    * @return the parsed occurrence value
260    */
261   @NonNull
262   public static Occurrence parseOccurrence(@Nullable Metapath10.OccurrenceindicatorContext ctx) {
263     Occurrence retval;
264     if (ctx == null) {
265       retval = Occurrence.ONE;
266     } else {
267       int type = ((TerminalNode) ctx.getChild(0)).getSymbol().getType();
268       switch (type) {
269       case Metapath10Lexer.QM:
270         retval = Occurrence.ZERO_OR_ONE;
271         break;
272       case Metapath10Lexer.STAR:
273         retval = Occurrence.ZERO_OR_MORE;
274         break;
275       case Metapath10Lexer.PLUS:
276         retval = Occurrence.ONE_OR_MORE;
277         break;
278       default:
279         throw new UnsupportedOperationException(((TerminalNode) ctx.getChild(0)).getSymbol().getText());
280       }
281     }
282     return retval;
283   }

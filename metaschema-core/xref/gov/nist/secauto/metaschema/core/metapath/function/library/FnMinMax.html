37  /**
38   * Implements the XPath 3.1
39   * <a href="https://www.w3.org/TR/xpath-functions-31/#func-min">fn:min</a> and
40   * <a href="https://www.w3.org/TR/xpath-functions-31/#func-max">fn:max</a>
41   * functions.
42   */
43  public final class FnMinMax {
44    private static final String NAME_MIN = "min";
45    private static final String NAME_MAX = "max";
46    /**
47     * Defines the set of primitive atomic item types supported by the min/max
48     * functions. This set is used for type validation and normalization during
49     * comparison operations.
50     */
51    @NonNull
52    private static final Set<Class<? extends IAnyAtomicItem>> PRIMITIVE_ITEM_TYPES = ObjectUtils.notNull(Set.of(
53        IStringItem.class,
54        IBooleanItem.class,
55        IDecimalItem.class,
56        IDurationItem.class,
57        IDateTimeItem.class,
58        IDateItem.class,
59        IBase64BinaryItem.class,
60        IAnyUriItem.class));
61    @NonNull
62    static final IFunction SIGNATURE_MIN = IFunction.builder()
63        .name(NAME_MIN)
64        .namespace(MetapathConstants.NS_METAPATH_FUNCTIONS)
65        .deterministic()
66        .contextIndependent()
67        .focusIndependent()
68        .argument(IArgument.builder()
69            .name("arg")
70            .type(IAnyAtomicItem.type())
71            .zeroOrMore()
72            .build())
73        .returnType(IAnyAtomicItem.type())
74        .returnZeroOrOne()
75        .functionHandler(FnMinMax::executeMin)
76        .build();
77  
78    @NonNull
79    static final IFunction SIGNATURE_MAX = IFunction.builder()
80        .name(NAME_MAX)
81        .namespace(MetapathConstants.NS_METAPATH_FUNCTIONS)
82        .deterministic()
83        .contextIndependent()
84        .focusIndependent()
85        .argument(IArgument.builder()
86            .name("arg")
87            .type(IAnyAtomicItem.type())
88            .zeroOrMore()
89            .build())
90        .returnType(IAnyAtomicItem.type())
91        .returnZeroOrOne()
92        .functionHandler(FnMinMax::executeMax)
93        .build();
94  
95    private FnMinMax() {
96      // disable construction
97    }
98  
99    @SuppressWarnings("unused")
100   @NonNull
101   private static ISequence<IAnyAtomicItem> executeMin(
102       @NonNull IFunction function,
103       @NonNull List<ISequence<?>> arguments,
104       @NonNull DynamicContext dynamicContext,
105       IItem focus) {
106     ISequence<? extends IAnyAtomicItem> sequence = FunctionUtils.asType(
107         ObjectUtils.requireNonNull(arguments.get(0)));
108 
109     return ISequence.of(min(sequence.getValue()));
110   }
111 
112   @SuppressWarnings("unused")
113   @NonNull
114   private static ISequence<IAnyAtomicItem> executeMax(
115       @NonNull IFunction function,
116       @NonNull List<ISequence<?>> arguments,
117       @NonNull DynamicContext dynamicContext,
118       IItem focus) {
119     ISequence<? extends IAnyAtomicItem> sequence = FunctionUtils.asType(
120         ObjectUtils.requireNonNull(arguments.get(0)));
121 
122     return ISequence.of(max(sequence.getValue()));
123   }
124 
125   /**
126    * An implementation of XPath 3.1
127    * <a href="https://www.w3.org/TR/xpath-functions-31/#func-min">fn:min</a>.
128    *
129    * @param items
130    *          the items to find the minimum value for
131    * @return the average
132    */
133   @Nullable
134   public static IAnyAtomicItem min(@NonNull List<? extends IAnyAtomicItem> items) {
135     return normalize(items)
136         .min((item1, item2) -> {
137           assert item2 != null;
138           return item1.compareTo(item2);
139         })
140         .orElse(null);
141   }
142 
143   /**
144    * An implementation of XPath 3.1
145    * <a href="https://www.w3.org/TR/xpath-functions-31/#func-max">fn:max</a>.
146    *
147    * @param items
148    *          the items to find the maximum value for
149    * @return the average
150    */
151   @Nullable
152   public static IAnyAtomicItem max(@NonNull List<? extends IAnyAtomicItem> items) {
153     return normalize(items)
154         .max((item1, item2) -> {
155           assert item2 != null;
156           return item1.compareTo(item2);
157         })
158         .orElse(null);
159   }
160 
161   @SuppressWarnings("PMD.OnlyOneReturn") // readability
162   private static Stream<? extends IAnyAtomicItem> normalize(
163       @NonNull List<? extends IAnyAtomicItem> items) {
164     if (items.isEmpty()) {
165       return Stream.empty();
166     }
167 
168     if (items.size() == 1) {
169       return Stream.of(items.get(0));
170     }
171 
172     List<? extends IAnyAtomicItem> resultingItems = convertUntypedItems(items);
173     Map<Class<? extends IAnyAtomicItem>, Integer> counts = countItemTypes(resultingItems);
174     return createNormalizedStream(resultingItems, counts);
175   }
176 
177   @NonNull
178   private static List<? extends IAnyAtomicItem> convertUntypedItems(
179       @NonNull List<? extends IAnyAtomicItem> items) {
180     return ObjectUtils.notNull(items.stream()
181         .map(item -> item instanceof IUntypedAtomicItem ? IDecimalItem.cast(item) : item)
182         .collect(Collectors.toList()));
183   }
184 
185   @NonNull
186   private static Map<Class<? extends IAnyAtomicItem>, Integer> countItemTypes(
187       @NonNull List<? extends IAnyAtomicItem> items) {
188     return FunctionUtils.countTypes(PRIMITIVE_ITEM_TYPES, items);
189   }
190 
191   @SuppressWarnings("PMD.OnlyOneReturn")
192   @NonNull
193   private static Stream<? extends IAnyAtomicItem> createNormalizedStream(
194       @NonNull List<? extends IAnyAtomicItem> items,
195       @NonNull Map<Class<? extends IAnyAtomicItem>, Integer> counts) {
196 
197     // Single type - no conversion needed
198     if (counts.size() == 1) {
199       return ObjectUtils.notNull(items.stream());
200     }
201 
202     // Multiple types - attempt conversion
203     int size = items.size();
204     if (counts.size() > 1) {
205       // Check if all items are either String or AnyUri
206       if (counts.getOrDefault(IStringItem.class, 0) + counts.getOrDefault(IAnyUriItem.class, 0) == size) {
207         return ObjectUtils.notNull(items.stream().map(IAnyAtomicItem::asStringItem));
208       }
209 
210       // Check if all items are Decimal
211       if (counts.getOrDefault(IDecimalItem.class, 0) == size) {
212         return ObjectUtils.notNull(items.stream().map(item -> (IDecimalItem) item));
213       }
214     }
215 
216     // No valid conversion possible
217     throw new InvalidArgumentFunctionException(
218         InvalidArgumentFunctionException.INVALID_ARGUMENT_TYPE,
219         String.format(
220             "Values must all be of a single atomic type. Found multiple types: [%s]",
221             FunctionUtils.getTypes(items).stream()
222                 .map(Class::getSimpleName)
223                 .collect(Collectors.joining(", "))));
224   }

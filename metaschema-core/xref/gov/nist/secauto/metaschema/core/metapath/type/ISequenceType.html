6   package gov.nist.secauto.metaschema.core.metapath.type;
7   
8   import gov.nist.secauto.metaschema.core.metapath.ICollectionValue;
9   import gov.nist.secauto.metaschema.core.metapath.type.impl.SequenceTypeImpl;
10  
11  import edu.umd.cs.findbugs.annotations.NonNull;
12  
13  /**
14   * Supports
15   * <a href="https://www.w3.org/TR/xpath-31/#id-sequencetype-syntax">sequence
16   * type</a> testing.
17   */
18  public interface ISequenceType {
19    /**
20     * A sequence type representing an empty sequence.
21     *
22     * @return the sequence type
23     */
24    @NonNull
25    static ISequenceType empty() {
26      return SequenceTypeImpl.empty();
27    }
28  
29    /**
30     * Create new sequence type using the provide type and occurrence.
31     *
32     * @param type
33     *          the required sequence item type
34     * @param occurrence
35     *          the expected occurrence of the sequence
36     * @return the new sequence type
37     */
38    @SuppressWarnings("PMD.ShortMethodName")
39    @NonNull
40    static ISequenceType of(
41        @NonNull IItemType type,
42        @NonNull Occurrence occurrence) {
43      return Occurrence.ZERO.equals(occurrence)
44          ? empty()
45          : new SequenceTypeImpl(type, occurrence);
46    }
47  
48    /**
49     * Determine if the sequence is empty (if it holds any data) or not.
50     *
51     * @return {@code true} if the sequence is empty or {@code false} otherwise
52     */
53    boolean isEmpty();
54  
55    /**
56     * Get the type of the sequence.
57     *
58     * @return the type of the sequence or {@code null} if the sequence is empty
59     */
60    @NonNull
61    IItemType getType();
62  
63    /**
64     * Get the occurrence of the sequence.
65     *
66     * @return the occurrence of the sequence or {@code Occurrence#ZERO} if the
67     *         sequence is empty
68     */
69    @NonNull
70    Occurrence getOccurrence();
71  
72    /**
73     * Get the signature of the function as a string.
74     *
75     * @return the signature
76     */
77    @NonNull
78    String toSignature();
79  
80    /**
81     * Tests that a given collection value matches this sequence type.
82     *
83     * @param value
84     *          the collection value to test
85     * @return {@code true} if the value matches the expectations of this sequence
86     *         type or {@code false} otherwise
87     */
88    boolean matches(@NonNull ICollectionValue value);
89  }

6   package gov.nist.secauto.metaschema.core.metapath.item.atomic;
7   
8   import gov.nist.secauto.metaschema.core.datatype.adapter.MetaschemaDataTypeProvider;
9   import gov.nist.secauto.metaschema.core.metapath.function.InvalidValueForCastFunctionException;
10  
11  import edu.umd.cs.findbugs.annotations.NonNull;
12  
13  @Deprecated(forRemoval = true, since = "0.7.0")
14  public interface INcNameItem extends IStringItem {
15    /**
16     * Construct a new item using the provided string {@code value}.
17     *
18     * @param value
19     *          a string representing a NCName value
20     * @return the new item
21     */
22  
23    @NonNull
24    static INcNameItem valueOf(@NonNull String value) {
25      try {
26        return new NcNameItemImpl(MetaschemaDataTypeProvider.NCNAME.parse(value));
27      } catch (IllegalArgumentException ex) {
28        throw new InvalidValueForCastFunctionException(String.format("Unable to parse string value '%s'", value),
29            ex);
30      }
31    }
32  
33    /**
34     * Cast the provided type to this item type.
35     *
36     * @param item
37     *          the item to cast
38     * @return the original item if it is already this type, otherwise a new item
39     *         cast to this type
40     * @throws InvalidValueForCastFunctionException
41     *           if the provided {@code item} cannot be cast to this type
42     */
43    @NonNull
44    static INcNameItem cast(@NonNull IAnyAtomicItem item) {
45      return MetaschemaDataTypeProvider.NCNAME.cast(item);
46    }
47  
48    @Override
49    default INcNameItem castAsType(IAnyAtomicItem item) {
50      return cast(item);
51    }
52  }

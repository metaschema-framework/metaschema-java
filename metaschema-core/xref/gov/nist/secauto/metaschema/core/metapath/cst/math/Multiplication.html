6   package gov.nist.secauto.metaschema.core.metapath.cst.math;
7   
8   import gov.nist.secauto.metaschema.core.metapath.ISequence;
9   import gov.nist.secauto.metaschema.core.metapath.cst.IExpression;
10  import gov.nist.secauto.metaschema.core.metapath.cst.IExpressionVisitor;
11  import gov.nist.secauto.metaschema.core.metapath.function.FunctionUtils;
12  import gov.nist.secauto.metaschema.core.metapath.function.OperationFunctions;
13  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IAnyAtomicItem;
14  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IDayTimeDurationItem;
15  import gov.nist.secauto.metaschema.core.metapath.item.atomic.INumericItem;
16  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IYearMonthDurationItem;
17  import gov.nist.secauto.metaschema.core.metapath.type.InvalidTypeMetapathException;
18  
19  import edu.umd.cs.findbugs.annotations.NonNull;
20  
21  public class Multiplication
22      extends AbstractBasicArithmeticExpression {
23  
24    /**
25     * An expression that gets the product result by multiplying two values.
26     *
27     * @param left
28     *          the item to be divided
29     * @param right
30     *          the item to divide by
31     */
32    public Multiplication(@NonNull IExpression left, @NonNull IExpression right) {
33      super(left, right);
34    }
35  
36    @Override
37    public <RESULT, CONTEXT> RESULT accept(IExpressionVisitor<RESULT, CONTEXT> visitor, CONTEXT context) {
38      return visitor.visitMultiplication(this, context);
39    }
40  
41    /**
42     * Get the sum of two atomic items.
43     *
44     * @param left
45     *          the first item to multiply
46     * @param right
47     *          the second item to multiply
48     * @return the product of both items or an empty {@link ISequence} if either
49     *         item is {@code null}
50     */
51    @Override
52    @NonNull
53    protected IAnyAtomicItem operation(@NonNull IAnyAtomicItem left, @NonNull IAnyAtomicItem right) {
54      return multiply(left, right);
55    }
56  
57    /**
58     * Get the sum of two atomic items.
59     *
60     * @param leftItem
61     *          the first item to multiply
62     * @param rightItem
63     *          the second item to multiply
64     * @return the product of both items
65     */
66    @NonNull
67    public static IAnyAtomicItem multiply(@NonNull IAnyAtomicItem leftItem, // NOPMD - intentional
68        @NonNull IAnyAtomicItem rightItem) {
69      IAnyAtomicItem retval = null;
70      if (leftItem instanceof IYearMonthDurationItem) {
71        IYearMonthDurationItem left = (IYearMonthDurationItem) leftItem;
72        if (rightItem instanceof INumericItem) {
73          retval = OperationFunctions.opMultiplyYearMonthDuration(left, (INumericItem) rightItem);
74        }
75      } else if (leftItem instanceof IDayTimeDurationItem) {
76        IDayTimeDurationItem left = (IDayTimeDurationItem) leftItem;
77        if (rightItem instanceof INumericItem) {
78          retval = OperationFunctions.opMultiplyDayTimeDuration(left, (INumericItem) rightItem);
79        }
80      } else {
81        // handle as numeric
82        INumericItem left = FunctionUtils.toNumeric(leftItem);
83        if (rightItem instanceof INumericItem) {
84          INumericItem right = FunctionUtils.toNumeric(rightItem);
85          retval = OperationFunctions.opNumericMultiply(left, right);
86        } else if (rightItem instanceof IYearMonthDurationItem) {
87          retval = OperationFunctions.opMultiplyYearMonthDuration((IYearMonthDurationItem) rightItem, left);
88        } else if (rightItem instanceof IDayTimeDurationItem) {
89          retval = OperationFunctions.opMultiplyDayTimeDuration((IDayTimeDurationItem) rightItem, left);
90        }
91      }
92      if (retval == null) {
93        throw new InvalidTypeMetapathException(
94            null,
95            String.format("The expression '%s * %s' is not supported", leftItem.getClass().getName(),
96                rightItem.getClass().getName()));
97      }
98      return retval;
99    }
100 }

6   package gov.nist.secauto.metaschema.core.metapath.cst.comparison;
7   
8   import gov.nist.secauto.metaschema.core.metapath.cst.AbstractBinaryExpression;
9   import gov.nist.secauto.metaschema.core.metapath.cst.IBooleanLogicExpression;
10  import gov.nist.secauto.metaschema.core.metapath.cst.IExpression;
11  import gov.nist.secauto.metaschema.core.metapath.function.ComparisonFunctions;
12  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
13  
14  import edu.umd.cs.findbugs.annotations.NonNull;
15  
16  /**
17   * A common base class for all comparison nodes, which consist of two
18   * expressions representing the left and right sides of the comparison, and a
19   * comparison operator.
20   */
21  public abstract class AbstractComparison // NOPMD - unavoidable
22      extends AbstractBinaryExpression<IExpression, IExpression>
23      implements IBooleanLogicExpression {
24  
25    @NonNull
26    private final ComparisonFunctions.Operator operator;
27  
28    /**
29     * Construct an expression that compares the result of the {@code right}
30     * expression with the result of the {@code left} expression using the specified
31     * {@code operator}.
32     *
33     * @param left
34     *          the expression to compare against
35     * @param operator
36     *          the comparison operator
37     * @param right
38     *          the expression to compare with
39     */
40    public AbstractComparison(@NonNull IExpression left, @NonNull ComparisonFunctions.Operator operator,
41        @NonNull IExpression right) {
42      super(left, right);
43      this.operator = ObjectUtils.requireNonNull(operator, "operator");
44    }
45  
46    /**
47     * Get the comparison operator.
48     *
49     * @return the operator
50     */
51    @NonNull
52    public ComparisonFunctions.Operator getOperator() {
53      return operator;
54    }
55  
56    @SuppressWarnings("null")
57    @Override
58    public String toASTString() {
59      return String.format("%s[operator=%s]", getClass().getName(), operator);
60    }
61  
62  }

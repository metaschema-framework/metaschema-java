6   package gov.nist.secauto.metaschema.core.metapath.item.atomic;
7   
8   import gov.nist.secauto.metaschema.core.datatype.adapter.MetaschemaDataTypeProvider;
9   import gov.nist.secauto.metaschema.core.metapath.function.InvalidValueForCastFunctionException;
10  import gov.nist.secauto.metaschema.core.metapath.item.atomic.impl.TokenItemImpl;
11  import gov.nist.secauto.metaschema.core.metapath.type.IAtomicOrUnionType;
12  import gov.nist.secauto.metaschema.core.metapath.type.InvalidTypeMetapathException;
13  
14  import edu.umd.cs.findbugs.annotations.NonNull;
15  
16  /**
17   * An atomic Metapath item containing a text token data value.
18   */
19  public interface ITokenItem extends IStringItem {
20    /**
21     * Get the type information for this item.
22     *
23     * @return the type information
24     */
25    @NonNull
26    static IAtomicOrUnionType<ITokenItem> type() {
27      return MetaschemaDataTypeProvider.TOKEN.getItemType();
28    }
29  
30    @Override
31    default IAtomicOrUnionType<ITokenItem> getType() {
32      return type();
33    }
34  
35    /**
36     * Construct a new item using the provided string {@code value}.
37     *
38     * @param value
39     *          a string representing a token value
40     * @return the new item
41     * @throws InvalidTypeMetapathException
42     *           if the provided value is not a token
43     */
44    @NonNull
45    static ITokenItem valueOf(@NonNull String value) {
46      try {
47        return new TokenItemImpl(MetaschemaDataTypeProvider.TOKEN.parse(value));
48      } catch (IllegalArgumentException ex) {
49        throw new InvalidTypeMetapathException(
50            null,
51            String.format("Invalid token value '%s'. %s",
52                value,
53                ex.getLocalizedMessage()),
54            ex);
55      }
56    }
57  
58    /**
59     * Cast the provided type to this item type.
60     *
61     * @param item
62     *          the item to cast
63     * @return the original item if it is already this type, otherwise a new item
64     *         cast to this type
65     * @throws InvalidValueForCastFunctionException
66     *           if the provided {@code item} cannot be cast to this type
67     */
68    @NonNull
69    static ITokenItem cast(@NonNull IAnyAtomicItem item) {
70      try {
71        return item instanceof ITokenItem
72            ? (ITokenItem) item
73            : valueOf(item.asString());
74      } catch (InvalidTypeMetapathException ex) {
75        throw new InvalidValueForCastFunctionException(ex);
76      }
77    }
78  
79    @Override
80    default ITokenItem castAsType(IAnyAtomicItem item) {
81      return cast(item);
82    }
83  }

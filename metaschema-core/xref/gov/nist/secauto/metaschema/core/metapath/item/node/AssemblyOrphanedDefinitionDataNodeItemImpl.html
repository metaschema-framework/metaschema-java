15  class AssemblyOrphanedDefinitionDataNodeItemImpl
16      extends AbstractOrphanedDefinitionNodeItem<IAssemblyDefinition, IAssemblyInstance>
17      implements IAssemblyNodeItem,
18      IFeatureModelContainerItem,
19      IFeatureRequiredDataItem,
20      IFeatureOrhpanedDefinitionModelNodeItem<IAssemblyDefinition, IAssemblyInstance> {
21    @NonNull
22    private final Lazy<ModelContainer> model;
23    @NonNull
24    private final Object value;
25  
26    public AssemblyOrphanedDefinitionDataNodeItemImpl(
27        @NonNull IAssemblyDefinition definition,
28        @Nullable URI baseUri,
29        @NonNull Object value,
30        @NonNull INodeItemGenerator generator) {
31      super(definition, baseUri);
32      this.value = value;
33      this.model = ObjectUtils.notNull(Lazy.lazy(generator.newDataModelSupplier(this)));
34    }
35  
36    @SuppressWarnings("null")
37    @Override
38    public ModelContainer getModel() {
39      return model.get();
40    }
41  
42    @Override
43    public Object getValue() {
44      return value;
45    }
46  
47    @Override
48    public String stringValue() {
49      return ObjectUtils.notNull(modelItems()
50          .map(INodeItem::stringValue)
51          .collect(Collectors.joining()));
52    }

6   package gov.nist.secauto.metaschema.core.metapath.cst.path;
7   
8   import gov.nist.secauto.metaschema.core.metapath.DynamicContext;
9   import gov.nist.secauto.metaschema.core.metapath.ISequence;
10  import gov.nist.secauto.metaschema.core.metapath.cst.IExpression;
11  import gov.nist.secauto.metaschema.core.metapath.cst.IExpressionVisitor;
12  
13  import edu.umd.cs.findbugs.annotations.NonNull;
14  
15  public class RelativeSlashPath
16      extends AbstractRelativePathExpression {
17  
18    /**
19     * Construct a new expression that finds a child of the {@code left} expression
20     * using the {@code right} expression.
21     *
22     * @param left
23     *          the context path
24     * @param right
25     *          the path to evaluate in the context of the left
26     */
27    public RelativeSlashPath(@NonNull IExpression left, @NonNull IExpression right) {
28      super(left, right);
29    }
30  
31    @Override
32    public <RESULT, CONTEXT> RESULT accept(IExpressionVisitor<RESULT, CONTEXT> visitor, CONTEXT context) {
33      return visitor.visitRelativeSlashPath(this, context);
34    }
35  
36    @Override
37    public ISequence<?> accept(
38        DynamicContext dynamicContext,
39        ISequence<?> focus) {
40      ISequence<?> leftResult = getLeft().accept(dynamicContext, focus);
41  
42      // ensures the left sequence is list backed
43      return leftResult.isEmpty()
44          ? ISequence.empty()
45          : getRight().accept(dynamicContext, leftResult);
46    }
47  }

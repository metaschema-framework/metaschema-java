51    @SuppressWarnings("null")
52    @Test
53    void testAllowedValuesAllowOther() {
54      MockNodeItemFactory itemFactory = new MockNodeItemFactory();
55  
56      IFlagNodeItem flag = itemFactory.flag(qname("value"), IStringItem.valueOf("value"));
57  
58      IFlagDefinition flagDefinition = mock(IFlagDefinition.class);
59  
60      ISource source = mock(ISource.class);
61  
62      IAllowedValuesConstraint allowedValues = IAllowedValuesConstraint.builder()
63          .source(source)
64          .allowedValue(IAllowedValue.of(
65              "other",
66              MarkupLine.fromMarkdown("some documentation"),
67              "1.0.0"))
68          .allowsOther(true)
69          .build();
70  
71      doReturn(flagDefinition).when(flag).getDefinition();
72      doReturn("flag/path").when(flag).toPath(any(IPathFormatter.class));
73  
74      doReturn(CollectionUtil.emptyMap()).when(flagDefinition).getLetExpressions();
75      doReturn(CollectionUtil.singletonList(allowedValues)).when(flagDefinition).getAllowedValuesConstraints();
76      doReturn(CollectionUtil.emptyList()).when(flagDefinition).getExpectConstraints();
77      doReturn(CollectionUtil.emptyList()).when(flagDefinition).getMatchesConstraints();
78      doReturn(CollectionUtil.emptyList()).when(flagDefinition).getIndexHasKeyConstraints();
79  
80      doReturn(StaticContext.instance()).when(source).getStaticContext();
81  
82      FindingCollectingConstraintValidationHandler handler = new FindingCollectingConstraintValidationHandler();
83      DefaultConstraintValidator validator = new DefaultConstraintValidator(handler);
84      DynamicContext dynamicContext = new DynamicContext();
85      validator.validate(flag, dynamicContext);
86      validator.finalizeValidation(dynamicContext);
87  
88      assertTrue(handler.isPassing(), "doesn't pass");
89    }

141   @SuppressWarnings("null")
142   @Test
143   void testMultipleAllowedValuesConflictingAllowOther() {
144     MockNodeItemFactory itemFactory = new MockNodeItemFactory();
145 
146     IFlagNodeItem flag1 = itemFactory.flag(qname("value"), IStringItem.valueOf("value"));
147     IFlagNodeItem flag2 = itemFactory.flag(qname("other2"), IStringItem.valueOf("other2"));
148 
149     IFlagDefinition flagDefinition = mock(IFlagDefinition.class);
150 
151     ISource source = mock(ISource.class);
152 
153     IAllowedValuesConstraint allowedValues1 = IAllowedValuesConstraint.builder()
154         .source(source)
155         .allowedValue(IAllowedValue.of(
156             "other",
157             MarkupLine.fromMarkdown("some documentation"),
158             null))
159         .allowsOther(true)
160         .build();
161     IAllowedValuesConstraint allowedValues2 = IAllowedValuesConstraint.builder()
162         .source(source)
163         .allowedValue(IAllowedValue.of(
164             "other2",
165             MarkupLine.fromMarkdown("some documentation"),
166             "1.0.0"))
167         .allowsOther(false)
168         .build();
169 
170     List<? extends IAllowedValuesConstraint> allowedValuesConstraints
171         = List.of(allowedValues1, allowedValues2);
172 
173     DynamicContext dynamicContext = new DynamicContext();
174 
175     doReturn(flagDefinition).when(flag1).getDefinition();
176     doAnswer(invocation -> invocation.getArgument(0, DefaultConstraintValidator.Visitor.class)
177         .visitFlag((IFlagNodeItem) invocation.getMock(), dynamicContext))
178             .when(flag1).accept(any(IItemVisitor.class));
179     doReturn("flag1/path").when(flag1).toPath(any(IPathFormatter.class));
180 
181     doReturn(flagDefinition).when(flag2).getDefinition();
182     doAnswer(invocation -> invocation.getArgument(0, DefaultConstraintValidator.Visitor.class)
183         .visitFlag((IFlagNodeItem) invocation.getMock(), dynamicContext))
184             .when(flag2).accept(any(IItemVisitor.class));
185     doReturn("flag1/path").when(flag2).toPath(any(IPathFormatter.class));
186 
187     doReturn(CollectionUtil.emptyMap()).when(flagDefinition).getLetExpressions();
188     doReturn(allowedValuesConstraints).when(flagDefinition).getAllowedValuesConstraints();
189     doReturn(CollectionUtil.emptyList()).when(flagDefinition).getExpectConstraints();
190     doReturn(CollectionUtil.emptyList()).when(flagDefinition).getMatchesConstraints();
191     doReturn(CollectionUtil.emptyList()).when(flagDefinition).getIndexHasKeyConstraints();
192 
193     doReturn(StaticContext.instance()).when(source).getStaticContext();
194 
195     FindingCollectingConstraintValidationHandler handler = new FindingCollectingConstraintValidationHandler();
196     DefaultConstraintValidator validator = new DefaultConstraintValidator(handler);
197     validator.validate(flag1, dynamicContext);
198     validator.validate(flag2, dynamicContext);
199     validator.finalizeValidation(dynamicContext);
200     assertAll(
201         () -> assertFalse(handler.isPassing(), "must pass"),
202         () -> assertThat("only 1 finding", handler.getFindings(), hasSize(1)),
203         () -> assertThat("finding is for a flag node", handler.getFindings(), hasItem(hasProperty("node", is(flag1)))));
204   }

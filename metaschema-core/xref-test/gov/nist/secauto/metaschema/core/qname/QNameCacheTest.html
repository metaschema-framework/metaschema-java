6   package gov.nist.secauto.metaschema.core.qname;
7   
8   import static org.junit.jupiter.api.Assertions.assertAll;
9   import static org.junit.jupiter.api.Assertions.assertEquals;
10  import static org.junit.jupiter.api.Assertions.assertNotNull;
11  
12  import org.junit.jupiter.params.ParameterizedTest;
13  import org.junit.jupiter.params.provider.Arguments;
14  import org.junit.jupiter.params.provider.MethodSource;
15  
16  import java.util.stream.Stream;
17  
18  import edu.umd.cs.findbugs.annotations.NonNull;
19  
20  class QNameCacheTest {
21    private static Stream<Arguments> provideValues() {
22      return Stream.of(
23          Arguments.of("", "name1"),
24          Arguments.of("https://example.com/ns", "name1"),
25          Arguments.of("https://example.com/ns", "name2"),
26          Arguments.of("", "name2"));
27    }
28  
29    @ParameterizedTest
30    @MethodSource("provideValues")
31    void test(@NonNull String namespace, @NonNull String localName) {
32      QNameCache cache = QNameCache.instance();
33  
34      IEnhancedQName qname = cache.of(namespace, localName);
35      IEnhancedQName lookup = cache.get(namespace, localName);
36  
37      assertAll(
38          () -> assertNotNull(lookup),
39          () -> assertEquals(qname, lookup, "Expected to retrieve the same QName record"),
40          () -> assertEquals(namespace, lookup == null ? null : lookup.getNamespace()),
41          () -> assertEquals(localName, lookup == null ? null : lookup.getLocalName()));
42    }
43  
44  }

6   package gov.nist.secauto.metaschema.core.datatype.adapter;
7   
8   import static org.junit.jupiter.api.Assertions.assertAll;
9   import static org.junit.jupiter.api.Assertions.assertEquals;
10  import static org.junit.jupiter.api.Assertions.assertThrows;
11  
12  import gov.nist.secauto.metaschema.core.datatype.object.AmbiguousDate;
13  
14  import org.junit.jupiter.params.ParameterizedTest;
15  import org.junit.jupiter.params.provider.Arguments;
16  import org.junit.jupiter.params.provider.MethodSource;
17  
18  import java.time.ZoneOffset;
19  import java.time.ZonedDateTime;
20  import java.util.stream.Stream;
21  
22  import edu.umd.cs.findbugs.annotations.NonNull;
23  
24  class DateAdapterTest {
25    private static final DateAdapter ADAPTER = new DateAdapter();
26  
27    private static Stream<Arguments> provideValues() {
28      return Stream.of(
29          // Cases without timezone (ambiguous)
30          Arguments.of("2018-01-01", true, ZonedDateTime.of(2018, 1, 1, 0, 0, 0, 0, ZoneOffset.UTC)),
31          Arguments.of("2020-01-01", true, ZonedDateTime.of(2020, 1, 1, 0, 0, 0, 0, ZoneOffset.UTC)),
32          Arguments.of("2018-01-01", true, ZonedDateTime.of(2018, 1, 1, 0, 0, 0, 0, ZoneOffset.UTC)),
33          Arguments.of("2000-02-29", true, ZonedDateTime.of(2000, 2, 29, 0, 0, 0, 0, ZoneOffset.UTC)),
34          Arguments.of("2020-02-29", true, ZonedDateTime.of(2020, 2, 29, 0, 0, 0, 0, ZoneOffset.UTC)),
35          // Cases with explicit timezone
36          Arguments.of("2020-06-23Z", false, ZonedDateTime.of(2020, 6, 23, 0, 0, 0, 0, ZoneOffset.UTC)),
37          Arguments.of("2020-06-23-04:00", false, ZonedDateTime.of(2020, 6, 23, 0, 0, 0, 0, ZoneOffset.of("-04:00"))));
38    }
39  
40    @ParameterizedTest
41    @MethodSource("provideValues")
42    void testSimpleDate(@NonNull String actual, boolean ambiguous, @NonNull ZonedDateTime expected) {
43      AmbiguousDate date = ADAPTER.parse(actual);
44      assertAll(
45          () -> assertEquals(ambiguous, !date.hasTimeZone()),
46          () -> assertEquals(expected, date.getValue()));
47    }
48  
49    private static Stream<Arguments> provideInvalidValues() {
50      return Stream.of(
51          // Cases with invalid date values
52          Arguments.of("2100-02-29"),
53          Arguments.of("2023-02-30Z"),
54          Arguments.of("2023-06-31-04:00"));
55    }
56  
57    @ParameterizedTest
58    @MethodSource("provideInvalidValues")
59    void testInvalidDates(@NonNull String actual) {
60      assertThrows(IllegalArgumentException.class, () -> {
61        ADAPTER.parse(actual);
62      });
63    }
64  }

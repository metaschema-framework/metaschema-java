11  import gov.nist.secauto.metaschema.core.datatype.markup.MarkupMultiline;
12  
13  import org.codehaus.stax2.XMLOutputFactory2;
14  import org.codehaus.stax2.XMLStreamWriter2;
15  import org.codehaus.stax2.ri.evt.MergedNsContext;
16  import org.codehaus.stax2.ri.evt.NamespaceEventImpl;
17  import org.junit.jupiter.api.Test;
18  
19  import java.util.List;
20  
21  import javax.xml.namespace.NamespaceContext;
22  import javax.xml.stream.XMLOutputFactory;
23  import javax.xml.stream.XMLStreamException;
24  
25  class MarkupXmlStreamWriterTest {
26    private static final String NAMESPACE = "http://www.w3.org/1999/xhtml";
27    private static final String NS_PREFIX = "";
28  
29    @Test
30    void testHTML() throws XMLStreamException {
31  
32      XMLOutputFactory2 factory = (XMLOutputFactory2) XMLOutputFactory.newInstance();
33      assert factory instanceof WstxOutputFactory;
34      factory.setProperty(WstxOutputProperties.P_OUTPUT_VALIDATE_STRUCTURE, false);
35      XMLStreamWriter2 xmlStreamWriter = (XMLStreamWriter2) factory.createXMLStreamWriter(System.out);
36      NamespaceContext nsContext = MergedNsContext.construct(xmlStreamWriter.getNamespaceContext(),
37          List.of(NamespaceEventImpl.constructNamespace(null, NS_PREFIX, NAMESPACE)));
38      xmlStreamWriter.setNamespaceContext(nsContext);
39  
40      String html = "<h1>Example</h1>\n"
41          + "<p><a href=\"link\">text</a><q>quote1</q></p>\n"
42          + "<table>\n"
43          + "<thead>\n"
44          + "<tr><th>Heading 1</th></tr>\n"
45          + "</thead>\n"
46          + "<tbody>\n"
47          + "<tr><td><q>data1</q> <insert type=\"param\" id-ref=\"insert\" /></td></tr>\n"
48          + "<tr><td><q>data2</q> <insert type=\"param\" id-ref=\"insert\" /></td></tr>\n"
49          + "</tbody>\n"
50          + "</table>\n"
51          + "<p>Some <q><em>more</em></q> <strong>text</strong> <img src=\"src\" alt=\"alt\" /></p>\n";
52  
53      MarkupMultiline ms = MarkupMultiline.fromHtml(html);
54      // System.out.println(AstCollectingVisitor.asString(ms.getDocument()));
55  
56      ms.writeXHtml(NAMESPACE, xmlStreamWriter);
57      xmlStreamWriter.close();
58    }

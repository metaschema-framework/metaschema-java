24  class DateTimeAdapterTest {
25    private static final DateTimeAdapter ADAPTER = new DateTimeAdapter();
26  
27    /**
28     * Provides test cases for date-time parsing.
29     * <p>
30     * Each argument contains:
31     * <ul>
32     * <li>input string to parse</li>
33     * <li>boolean indicating if the datetime is ambiguous (no timezone)</li>
34     * <li>expected ZonedDateTime result</li>
35     * </ul>
36     *
37     * @return Stream of test cases
38     */
39    private static Stream<Arguments> provideValues() {
40      return Stream.of(
41          // Cases without timezone (ambiguous)
42          Arguments.of(
43              "2018-01-01T00:00:00",
44              true,
45              ZonedDateTime.of(2018, 1, 1, 0, 0, 0, 0, ZoneOffset.UTC)),
46          Arguments.of(
47              "2019-09-28T23:20:50.5200",
48              true,
49              ZonedDateTime.of(2019, 9, 28, 23, 20, 50, toNanos(0.5200), ZoneOffset.UTC)),
50          Arguments.of(
51              "2019-12-02T16:39:57",
52              true,
53              ZonedDateTime.of(2019, 12, 2, 16, 39, 57, 0, ZoneOffset.UTC)),
54          Arguments.of(
55              "2019-12-31T23:59:59",
56              true,
57              ZonedDateTime.of(2019, 12, 31, 23, 59, 59, 0, ZoneOffset.UTC)),
58          // Cases with explicit timezone
59          Arguments.of(
60              "2019-09-28T23:20:50.52Z",
61              false,
62              ZonedDateTime.of(2019, 9, 28, 23, 20, 50, toNanos(0.52), ZoneOffset.UTC)),
63          Arguments.of(
64              "2019-09-28T23:20:50.0Z",
65              false,
66              ZonedDateTime.of(2019, 9, 28, 23, 20, 50, 0, ZoneOffset.UTC)),
67          Arguments.of(
68              "2019-12-02T16:39:57-08:00",
69              false,
70              ZonedDateTime.of(2019, 12, 2, 16, 39, 57, 0, ZoneOffset.of("-08:00"))),
71          Arguments.of(
72              "2019-12-02T16:39:57.100-08:00",
73              false,
74              ZonedDateTime.of(2019, 12, 2, 16, 39, 57, toNanos(0.100), ZoneOffset.of("-08:00"))),
75          Arguments.of(
76              "2019-12-31T23:59:59Z",
77              false,
78              ZonedDateTime.of(2019, 12, 31, 23, 59, 59, 0, ZoneOffset.UTC)));
79    }
80  
81    /**
82     * Converts a fractional second to nanoseconds.
83     *
84     * @param fraction
85     *          the fractional part of a second (0.0 to 0.999...)
86     * @return the equivalent nanoseconds
87     * @throws IllegalArgumentException
88     *           if fraction is negative or >= 1
89     */
90    private static int toNanos(double fraction) {
91      if (fraction < 0.0 || fraction >= 1.0) {
92        throw new IllegalArgumentException(String.format("Fraction '%.3f' must be between 0.0 and 0.999...", fraction));
93      }
94      return (int) Math.round(TimeUnit.SECONDS.toNanos(1) * fraction);
95    }
96  
97    @ParameterizedTest
98    @MethodSource("provideValues")
99    void testSimpleDateTime(@NonNull String actual, boolean ambiguous, @NonNull ZonedDateTime expected) {
100     AmbiguousDateTime date = ADAPTER.parse(actual);
101     assertAll(
102         () -> assertEquals(ambiguous, !date.hasTimeZone()),
103         () -> assertEquals(expected, date.getValue()));
104   }

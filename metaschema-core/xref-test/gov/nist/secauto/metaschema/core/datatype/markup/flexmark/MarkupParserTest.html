31  class MarkupParserTest {
32    private static final Logger LOGGER = LogManager.getLogger(MarkupParserTest.class);
33  
34    @Test
35    void test() throws XMLStreamException {
36      XMLInputFactory2 factory = (XMLInputFactory2) XMLInputFactory.newInstance();
37      assert factory instanceof WstxInputFactory;
38      factory.configureForXmlConformance();
39      factory.setProperty(XMLInputFactory.IS_COALESCING, true);
40  
41      String html = new StringBuilder()
42          .append("<node>\n")
43          .append("  <p> some text </p>\n")
44          .append("  <p><q>text</q></p>\n")
45          .append("  <p>some <b>text</b> <insert type=\"param\" id-ref=\"param-id\"/>.</p>\n")
46          .append("  <h1>Example</h1>\n")
47          .append("  <p><a href=\"link\">text</a></p>\n")
48          .append("  <ul>\n")
49          .append("    <li>a <strong>list item</strong></li>\n")
50          .append("    <li>another <i>list item</i></li>\n")
51          .append("  </ul>\n")
52          .append(" <table>\n")
53          .append(" <tr><th>Heading 1</th></tr>\n")
54          .append(" <tr><td><q>data1</q> <insert type=\"param\" id-ref=\"insert\" /></td></tr>\n")
55          .append(" </table>\n")
56          .append("  <p>Some <em>more</em> <strong>text</strong><img alt=\"alt\" src=\"src\"/></p>\n")
57          .append("</node>\n")
58          .toString();
59      XMLEventReader2 reader = (XMLEventReader2) factory.createXMLEventReader(new StringReader(html));
60      URI resource = ObjectUtils.notNull(URI.create("https://example.com/not-a-resource"));
61  
62      CharSequence startDocument = XmlEventUtil.toString(reader.nextEvent(), resource);
63      LOGGER.atDebug().log("StartDocument: {}", startDocument);
64  
65      CharSequence startElement = XmlEventUtil.toString(reader.nextEvent(), resource);
66      LOGGER.atDebug().log("StartElement: {}", startElement);
67  
68      assertDoesNotThrow(() -> {
69        MarkupMultiline markupString = XmlMarkupParser.instance().parseMarkupMultiline(reader, resource);
70        String ast = AstCollectingVisitor.asString(markupString.getDocument());
71  
72        // Verify specific elements are properly parsed
73        assertTrue(ast.contains("BulletListItem"), "List items should be present in AST");
74        assertTrue(ast.contains("TableBlock"), "Table should be present in AST");
75  
76        // Verify markup conversion
77        String markdown = markupString.toMarkdown();
78        assertTrue(markdown.contains("**list item**"), "Strong text should be converted to markdown");
79        assertTrue(markdown.contains("another *list item*"), "Italic text should be converted to markdown");
80        assertTrue(markdown.contains("{{ insert: param, param-id }}."),
81            "Insert placeholder should be converted to markdown");
82      });
83    }
84  
85    @Test
86    void emptyParagraphTest() throws XMLStreamException {
87      final String html = new StringBuilder()
88          .append("<node>\n")
89          .append("  <p/>\n")
90          .append("</node>\n")
91          .toString();
92  
93      XMLInputFactory2 factory = (XMLInputFactory2) XMLInputFactory.newInstance();
94      assert factory instanceof WstxInputFactory;
95      factory.configureForXmlConformance();
96      factory.setProperty(XMLInputFactory.IS_COALESCING, true);
97      XMLEventReader2 reader = (XMLEventReader2) factory.createXMLEventReader(new StringReader(html));
98      URI resource = ObjectUtils.notNull(URI.create("https://example.com/not-a-resource"));
99  
100     CharSequence startDocument = XmlEventUtil.toString(reader.nextEvent(), resource);
101     LOGGER.atDebug().log("StartDocument: {}", startDocument);
102 
103     CharSequence startElement = XmlEventUtil.toString(reader.nextEvent(), resource);
104     LOGGER.atDebug().log("StartElement: {}", startElement);
105 
106     assertDoesNotThrow(() -> {
107       MarkupMultiline ms = XmlMarkupParser.instance().parseMarkupMultiline(reader, resource);
108       String ast = AstCollectingVisitor.asString(ms.getDocument());
109       String xhtml = ms.toXHtml("");
110       String markdown = ms.toMarkdown();
111 
112       // Verify empty paragraph handling
113       assertTrue(!ast.contains("Paragraph"), "Empty paragraph should be present in AST");
114       assertTrue(xhtml.trim().isEmpty(), "Empty document should produce empty HTML");
115       assertTrue(markdown.trim().isEmpty(), "Empty paragraph should produce empty markdown");
116     });
117   }

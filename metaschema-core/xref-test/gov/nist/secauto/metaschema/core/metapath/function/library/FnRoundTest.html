6   package gov.nist.secauto.metaschema.core.metapath.function.library;
7   
8   import static gov.nist.secauto.metaschema.core.metapath.TestUtils.decimal;
9   import static gov.nist.secauto.metaschema.core.metapath.TestUtils.integer;
10  
11  import gov.nist.secauto.metaschema.core.metapath.ISequence;
12  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IIntegerItem;
13  import gov.nist.secauto.metaschema.core.metapath.item.atomic.INumericItem;
14  import gov.nist.secauto.metaschema.core.util.CollectionUtil;
15  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
16  
17  import org.junit.jupiter.api.Test;
18  import org.junit.jupiter.params.ParameterizedTest;
19  import org.junit.jupiter.params.provider.Arguments;
20  import org.junit.jupiter.params.provider.MethodSource;
21  
22  import java.util.List;
23  import java.util.stream.Stream;
24  
25  import edu.umd.cs.findbugs.annotations.NonNull;
26  
27  class FnRoundTest
28      extends FunctionTestBase {
29  
30    private static Stream<Arguments> provideValuesForRound() {
31      return Stream.of(
32          Arguments.of(decimal("2.5"), decimal("3")),
33          Arguments.of(decimal("2.4999"), decimal("2")),
34          Arguments.of(decimal("-2.5"), decimal("-2")));
35    }
36  
37    @ParameterizedTest
38    @MethodSource("provideValuesForRound")
39    void testRound(@NonNull INumericItem actual, @NonNull INumericItem expected) {
40      assertFunctionResult(
41          FnRound.SIGNATURE,
42          ISequence.of(expected),
43          CollectionUtil.singletonList(ISequence.of(actual)));
44    }
45  
46    @Test
47    void testRoundNoOp() {
48      assertFunctionResult(
49          FnRound.SIGNATURE,
50          ISequence.empty(),
51          CollectionUtil.singletonList(ISequence.empty()));
52    }
53  
54    private static Stream<Arguments> provideValuesForRoundWithPrecision() {
55      return Stream.of(
56          Arguments.of(decimal("1.125"), integer(2), decimal("1.13")),
57          Arguments.of(decimal("8452"), integer(-2), integer(8500)),
58          Arguments.of(decimal("3.1415e0"), integer(2), decimal("3.14e0")));
59    }
60  
61    @ParameterizedTest
62    @MethodSource("provideValuesForRoundWithPrecision")
63    void testRoundWithPrecision(
64        @NonNull INumericItem actual,
65        @NonNull IIntegerItem precision,
66        @NonNull INumericItem expected) {
67      assertFunctionResult(
68          FnRound.SIGNATURE_WITH_PRECISION,
69          ISequence.of(expected),
70          ObjectUtils.notNull(List.of(ISequence.of(actual), ISequence.of(precision))));
71    }
72  
73    @Test
74    void testRoundWithPrecisionNoOp() {
75      assertFunctionResult(
76          FnRound.SIGNATURE_WITH_PRECISION,
77          ISequence.empty(),
78          CollectionUtil.singletonList(ISequence.empty()));
79    }
80  }

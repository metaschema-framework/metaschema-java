8   import static org.junit.jupiter.api.Assertions.assertEquals;
9   
10  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
11  
12  import org.junit.jupiter.params.ParameterizedTest;
13  import org.junit.jupiter.params.provider.Arguments;
14  import org.junit.jupiter.params.provider.MethodSource;
15  
16  import java.util.UUID;
17  import java.util.stream.Stream;
18  
19  import edu.umd.cs.findbugs.annotations.NonNull;
20  
21  class IUuidItemTest {
22    private static Stream<Arguments> testCompare() { // NOPMD - false positive
23      UUID uuidRandom = ObjectUtils.notNull(UUID.randomUUID());
24      UUID uuid1 = ObjectUtils.notNull(UUID.fromString("4cfa2c52-9345-4012-8055-0bc9ac9b03fa"));
25      UUID uuid2 = ObjectUtils.notNull(UUID.fromString("25a6d916-f179-4550-ad2b-7e7cd9df35d2"));
26      String uuid2String = ObjectUtils.notNull(uuid2.toString());
27  
28      return Stream.of(
29          Arguments.of(IUuidItem.valueOf(uuidRandom), IUuidItem.valueOf(uuidRandom), IIntegerItem.ZERO),
30          Arguments.of(IUuidItem.valueOf(uuid1), IUuidItem.valueOf(uuid2), IIntegerItem.valueOf(2)),
31          Arguments.of(IUuidItem.valueOf(uuid2), IStringItem.valueOf(uuid2String), IIntegerItem.ZERO),
32          Arguments.of(IStringItem.valueOf(uuid2String), IUuidItem.valueOf(uuid2), IIntegerItem.ZERO));
33    }
34  
35    @ParameterizedTest
36    @MethodSource
37    void testCompare(@NonNull IAnyAtomicItem left, @NonNull IAnyAtomicItem right, @NonNull IIntegerItem expectedResult) {
38      IIntegerItem result = IIntegerItem.valueOf(left.compareTo(right));
39      assertEquals(expectedResult, result);
40    }

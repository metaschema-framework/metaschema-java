15  import org.junit.jupiter.params.ParameterizedTest;

18  
19  import java.util.stream.Stream;
20  
21  import edu.umd.cs.findbugs.annotations.NonNull;
22  
23  class RangeTest
24      extends ExpressionTestBase {
25    private static Stream<Arguments> provideValues() { // NOPMD - false positive
26      return Stream.of(
27          Arguments.of(
28              ISequence.of(integer(2), integer(3), integer(4), integer(5)),
29              "2 to 5"),
30          Arguments.of(
31              ISequence.empty(),
32              "() to 2"),
33          Arguments.of(
34              ISequence.empty(),
35              "2 to ()"),
36          Arguments.of(
37              ISequence.empty(),
38              "5 to 2"));
39    }
40  
41    @ParameterizedTest
42    @MethodSource("provideValues")
43    void testRange(@NonNull ISequence<?> expected, @NonNull String metapath) {
44      assertEquals(
45          expected,
46          MetapathExpression.compile(metapath).evaluateAs(null, MetapathExpression.ResultType.SEQUENCE,
47              newDynamicContext()));
48    }
49  }

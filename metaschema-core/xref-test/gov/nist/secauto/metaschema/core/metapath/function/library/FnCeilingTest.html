14  import org.junit.jupiter.api.Test;
15  import org.junit.jupiter.params.ParameterizedTest;

17  import org.junit.jupiter.params.provider.MethodSource;

19  import java.util.stream.Stream;
20  
21  import edu.umd.cs.findbugs.annotations.NonNull;
22  
23  class FnCeilingTest
24      extends FunctionTestBase {
25  
26    private static Stream<Arguments> provideValues() {
27      return Stream.of(
28          Arguments.of(decimal("11"), decimal("10.5")),
29          Arguments.of(decimal("-10"), decimal("-10.5")));
30    }
31  
32    @ParameterizedTest
33    @MethodSource("provideValues")
34    void testCeiling(@NonNull INumericItem expected, @NonNull INumericItem actual) {
35      assertFunctionResult(
36          FnCeiling.SIGNATURE,
37          ISequence.of(expected),
38          CollectionUtil.singletonList(ISequence.of(actual)));
39    }
40  
41    @Test
42    void testNoOp() {
43      assertFunctionResult(
44          FnCeiling.SIGNATURE,
45          ISequence.empty(),
46          CollectionUtil.singletonList(ISequence.empty()));
47    }
48  }

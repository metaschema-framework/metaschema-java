24  class FnHeadTest
25      extends ExpressionTestBase {
26    private static Stream<Arguments> provideValues() { // NOPMD - false positive
27      return Stream.of(
28          Arguments.of(
29              ISequence.of(integer(1)),
30              "head(1 to 5)"),
31          Arguments.of(
32              ISequence.of(string("a")),
33              "head(('a','b','c'))"),
34          Arguments.of(
35              ISequence.empty(),
36              "head(())"));
37    }
38  
39    @ParameterizedTest
40    @MethodSource("provideValues")
41    void test(@NonNull ISequence<?> expected, @NonNull String metapath) {
42      assertEquals(expected, MetapathExpression.compile(metapath).evaluateAs(null, MetapathExpression.ResultType.SEQUENCE,
43          newDynamicContext()));
44    }

6   package gov.nist.secauto.metaschema.core.metapath.cst.type;
7   
8   import static org.junit.jupiter.api.Assertions.assertEquals;
9   import static org.junit.jupiter.api.Assertions.assertThrows;
10  import static org.junit.jupiter.api.Assertions.assertTrue;
11  
12  import gov.nist.secauto.metaschema.core.metapath.ExpressionTestBase;
13  import gov.nist.secauto.metaschema.core.metapath.MetapathExpression;
14  import gov.nist.secauto.metaschema.core.metapath.StaticMetapathException;
15  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IAnyAtomicItem;
16  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
17  
18  import org.junit.jupiter.api.Test;
19  import org.junit.jupiter.params.ParameterizedTest;
20  import org.junit.jupiter.params.provider.Arguments;
21  import org.junit.jupiter.params.provider.MethodSource;
22  
23  import java.util.stream.Stream;
24  
25  import edu.umd.cs.findbugs.annotations.NonNull;
26  
27  class CastableTest
28      extends ExpressionTestBase {
29  
30    private static Stream<Arguments> provideValues() { // NOPMD - false positive
31      return CastTest.provideValues()
32          .map(args -> {
33            Object[] values = args.get();
34            return Arguments.of(values[0], values[1]);
35          });
36    }
37  
38    @ParameterizedTest
39    @MethodSource("provideValues")
40    void testCastable(@NonNull IAnyAtomicItem actual, @NonNull String singleType) {
41      MetapathExpression metapath = MetapathExpression.compile(". castable as " + singleType);
42      boolean result = ObjectUtils.notNull(metapath.evaluateAs(actual, MetapathExpression.ResultType.BOOLEAN));
43  
44      assertTrue(
45          result,
46          String.format("Expected `%s` to be castable.", singleType));
47    }
48  
49    @Test
50    void testInvalidTypePrefix() {
51      StaticMetapathException ex = assertThrows(StaticMetapathException.class, () -> {
52        MetapathExpression.compile("'a' castable as foo:bar");
53      });
54      assertEquals(StaticMetapathException.PREFIX_NOT_EXPANDABLE, ex.getCode());
55    }
56  
57    @Test
58    void testInvalidType() {
59      StaticMetapathException ex = assertThrows(StaticMetapathException.class, () -> {
60        MetapathExpression.compile("'a' castable as meta:bar");
61      });
62      assertEquals(StaticMetapathException.CAST_UNKNOWN_TYPE, ex.getCode());
63    }
64  
65    @Test
66    void testAnyAtomicType() {
67      StaticMetapathException ex = assertThrows(StaticMetapathException.class, () -> {
68        MetapathExpression.compile("'a' castable as meta:any-atomic-type");
69      });
70      assertEquals(StaticMetapathException.CAST_ANY_ATOMIC, ex.getCode());
71    }
72  }

108   private static IExpression parseExpression(@NonNull String path) {
109 
110     Metapath10Lexer lexer = new Metapath10Lexer(CharStreams.fromString(path));
111     CommonTokenStream tokens = new CommonTokenStream(lexer);
112     Metapath10 parser = new Metapath10(tokens);
113     parser.addErrorListener(new FailingErrorListener());
114 
115     ParseTree tree = parser.expr();
116     // ParseTreePrinter cstPrinter = new ParseTreePrinter(System.out);
117     // cstPrinter.print(tree, Arrays.asList(parser.getRuleNames()));
118 
119     return new BuildCSTVisitor(StaticContext.instance()).visit(tree);
120   }

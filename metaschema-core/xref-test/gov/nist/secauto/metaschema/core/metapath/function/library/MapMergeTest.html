28  class MapMergeTest
29      extends ExpressionTestBase {
30    private static Stream<Arguments> provideValues() { // NOPMD - false positive
31      return Stream.of(
32          Arguments.of(
33              map(),
34              "map:merge(())"),
35          Arguments.of(
36              map(entry(integer(0), string("no")), entry(integer(1), string("yes"))),
37              "map:merge((map:entry(0, \"no\"), map:entry(1, \"yes\")))"),
38          Arguments.of(
39              map(
40                  entry(integer(0), string("Sonntag")),
41                  entry(integer(1), string("Montag")),
42                  entry(integer(2), string("Dienstag")),
43                  entry(integer(3), string("Mittwoch")),
44                  entry(integer(4), string("Donnerstag")),
45                  entry(integer(5), string("Freitag")),
46                  entry(integer(6), string("Samstag")),
47                  entry(integer(7), string("Unbekannt"))),
48              "let $week :=  " + LookupTest.WEEKDAYS_GERMAN + " return map:merge(($week, map{7:\"Unbekannt\"}))"),
49          Arguments.of(
50              map(
51                  entry(integer(0), string("Sonntag")),
52                  entry(integer(1), string("Montag")),
53                  entry(integer(2), string("Dienstag")),
54                  entry(integer(3), string("Mittwoch")),
55                  entry(integer(4), string("Donnerstag")),
56                  entry(integer(5), string("Freitag")),
57                  entry(integer(6), string("Sonnabend"))),
58              "let $week :=  " + LookupTest.WEEKDAYS_GERMAN
59                  + " return map:merge(($week, map{6:\"Sonnabend\"}), map{\"duplicates\":\"use-last\"})"),
60          Arguments.of(
61              map(
62                  entry(integer(0), string("Sonntag")),
63                  entry(integer(1), string("Montag")),
64                  entry(integer(2), string("Dienstag")),
65                  entry(integer(3), string("Mittwoch")),
66                  entry(integer(4), string("Donnerstag")),
67                  entry(integer(5), string("Freitag")),
68                  entry(integer(6), string("Samstag"))),
69              "let $week :=  " + LookupTest.WEEKDAYS_GERMAN
70                  + " return map:merge(($week, map{6:\"Sonnabend\"}), map{\"duplicates\":\"use-first\"})"),
71          Arguments.of(
72              map(
73                  entry(integer(0), string("Sonntag")),
74                  entry(integer(1), string("Montag")),
75                  entry(integer(2), string("Dienstag")),
76                  entry(integer(3), string("Mittwoch")),
77                  entry(integer(4), string("Donnerstag")),
78                  entry(integer(5), string("Freitag")),
79                  entry(integer(6), sequence(string("Samstag"), string("Sonnabend")))),
80              "let $week :=  " + LookupTest.WEEKDAYS_GERMAN
81                  + " return map:merge(($week, map{6:\"Sonnabend\"}), map{\"duplicates\":\"combine\"})"));
82    }
83  
84    @ParameterizedTest
85    @MethodSource("provideValues")
86    void testExpression(@NonNull IItem expected, @NonNull String metapath) {
87  
88      IItem result = MetapathExpression.compile(metapath)
89          .evaluateAs(null, MetapathExpression.ResultType.NODE, newDynamicContext());
90      assertEquals(expected, result);
91    }
92  }

6   package gov.nist.secauto.metaschema.core.metapath.item.atomic;
7   
8   import static org.junit.jupiter.api.Assertions.assertArrayEquals;
9   import static org.junit.jupiter.api.Assertions.assertEquals;
10  
11  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
12  
13  import org.junit.jupiter.api.Assertions;
14  import org.junit.jupiter.api.Test;
15  
16  import java.nio.ByteBuffer;
17  
18  class IBase64BinaryItemTest {
19    private static final long MIN_LONG = -9_223_372_036_854_775_808L;
20    private static final long MAX_LONG = 9_223_372_036_854_775_807L;
21    private static final String BASE_64 = "gAAAAAAAAAB//////////w==";
22  
23    @Test
24    void testValueOf() {
25      IBase64BinaryItem item = IBase64BinaryItem.valueOf(ObjectUtils.notNull(
26          ByteBuffer.allocate(16).putLong(MIN_LONG).putLong(MAX_LONG)));
27      assertEquals(BASE_64, item.asString());
28    }
29  
30    @Test
31    void testCastSame() {
32      ByteBuffer buf
33          = ObjectUtils.notNull(ByteBuffer.allocate(16).putLong(MIN_LONG).putLong(MAX_LONG));
34      IBase64BinaryItem item = IBase64BinaryItem.valueOf(buf);
35      assertEquals(IBase64BinaryItem.cast(item), item);
36    }
37  
38    @Test
39    void testCastString() {
40      ByteBuffer buf
41          = ObjectUtils.notNull(ByteBuffer.allocate(16).putLong(MIN_LONG).putLong(MAX_LONG));
42      IBase64BinaryItem expected = IBase64BinaryItem.valueOf(buf);
43      IBase64BinaryItem actual = IBase64BinaryItem.cast(IStringItem.valueOf(BASE_64));
44      Assertions.assertAll(
45          // TODO: use equals method?
46          () -> assertArrayEquals(actual.asByteBuffer().array(), expected.asByteBuffer().array()),
47          () -> assertEquals(actual.asString(), expected.asString()));
48    }
49  }

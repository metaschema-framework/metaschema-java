19  import org.jmock.Expectations;

22  import org.jmock.junit5.JUnit5Mockery;
23  import org.junit.jupiter.api.Test;
24  import org.junit.jupiter.api.extension.RegisterExtension;
25  
26  import java.util.List;
27  
28  class ExpressionUtilsTest {
29  
30    @RegisterExtension
31    Mockery context = new JUnit5Mockery();
32  
33    @Mock
34    private IFlagNodeItem flagNodeItem1; // NOPMD - it's injected
35    @Mock
36    private IFlagNodeItem flagNodeItem2; // NOPMD - it's injected
37  
38    @Mock
39    private IExpression basicFlagExpr1; // NOPMD - it's injected
40    @Mock
41    private IExpression basicFlagExpr2; // NOPMD - it's injected
42    @Mock
43    private IExpression basicAssemblyExpr; // NOPMD - it's injected
44    @Mock
45    private IExpression basicFieldExpr; // NOPMD - it's injected
46  
47    @Test
48    void testTwoFlags() {
49      Class<INodeItem> baseType = INodeItem.class;
50  
51      context.checking(new Expectations() {
52        { // NOPMD - intentional
53          allowing(basicFlagExpr1).getStaticResultType();
54          will(returnValue(IFlagNodeItem.class));
55          allowing(basicFlagExpr2).getStaticResultType();
56          will(returnValue(IFlagNodeItem.class));
57        }
58      });
59      @SuppressWarnings("null")
60      Class<? extends INodeItem> result
61          = ExpressionUtils.analyzeStaticResultType(baseType, List.of(basicFlagExpr1, basicFlagExpr2));
62      assertEquals(IFlagNodeItem.class, result);
63    }
64  
65    @Test
66    void testFlagAndAssembly() {
67      Class<INodeItem> baseType = INodeItem.class;
68  
69      context.checking(new Expectations() {
70        { // NOPMD - intentional
71          allowing(basicFlagExpr1).getStaticResultType();
72          will(returnValue(IFlagNodeItem.class));
73          allowing(basicAssemblyExpr).getStaticResultType();
74          will(returnValue(IAssemblyNodeItem.class));
75        }
76      });
77      @SuppressWarnings("null")
78      Class<? extends INodeItem> result
79          = ExpressionUtils.analyzeStaticResultType(baseType, List.of(basicFlagExpr1, basicAssemblyExpr));
80      assertEquals(IDefinitionNodeItem.class, result);
81    }
82  
83    @Test
84    void testFieldAndAssembly() {
85      Class<INodeItem> baseType = INodeItem.class;
86  
87      context.checking(new Expectations() {
88        { // NOPMD - intentional
89          allowing(basicFieldExpr).getStaticResultType();
90          will(returnValue(IFieldNodeItem.class));
91          allowing(basicAssemblyExpr).getStaticResultType();
92          will(returnValue(IAssemblyNodeItem.class));
93        }
94      });
95      @SuppressWarnings("null")
96      Class<? extends INodeItem> result
97          = ExpressionUtils.analyzeStaticResultType(baseType, List.of(basicFieldExpr, basicAssemblyExpr));
98      assertEquals(IModelNodeItem.class, result);
99    }
100 }

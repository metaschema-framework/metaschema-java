6   package gov.nist.secauto.metaschema.databind.io;
7   
8   import org.eclipse.jdt.annotation.NotOwning;
9   
10  import java.net.URI;
11  
12  import edu.umd.cs.findbugs.annotations.NonNull;
13  
14  /**
15   * Provides objects used for parsing data associated with a specific format.
16   *
17   * @param <READER>
18   *          the format specific data reader
19   * @param <PROBLEM_HANDLER>
20   *          the format specific problem handler
21   */
22  public interface IParsingContext<READER, PROBLEM_HANDLER extends IProblemHandler> {
23    /**
24     * The parser used for reading data associated with the supported format.
25     *
26     * @return the parser
27     */
28    @NonNull
29    @NotOwning
30    READER getReader();
31  
32    /**
33     * Get the URI-based resource read by this parser.
34     *
35     * @return the resource URI
36     */
37    @NonNull
38    URI getSource();
39  
40    /**
41     * A handler that provides callbacks used to resolve parsing issues.
42     *
43     * @return the configured handler
44     */
45    @NonNull
46    PROBLEM_HANDLER getProblemHandler();
47  }

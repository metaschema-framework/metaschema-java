33  /**
34   * Supports building the model contents of a Metaschema assembly definition.
35   * <p>
36   * This class is not thread safe.
37   */
38  @SuppressWarnings("PMD.UseConcurrentHashMap")
39  public final class AssemblyModelGenerator
40      extends AbstractAbsoluteModelGenerator<
41          IBindingDefinitionModelAssembly,
42          DefaultAssemblyModelBuilder<
43              IModelInstanceAbsolute,
44              INamedModelInstanceAbsolute,
45              IFieldInstanceAbsolute,
46              IAssemblyInstanceAbsolute,
47              IChoiceInstance,
48              IChoiceGroupInstance>> {
49    // counters to track child positions
50    private int choicePosition; // 0
51    private int choiceGroupPosition; // 0
52  
53    /**
54     * Construct a new choice model container.
55     *
56     * @param binding
57     *          the choice model object bound to a Java class
58     * @param bindingInstance
59     *          the Metaschema module instance for the bound model object
60     * @param parent
61     *          the assembly definition containing this container
62     * @param nodeItemFactory
63     *          the node item factory used to generate child nodes
64     * @return the container
65     */
66    @SuppressWarnings("PMD.ShortMethodName")
67    public static IContainerModelAssemblySupport<
68        IModelInstanceAbsolute,
69        INamedModelInstanceAbsolute,
70        IFieldInstanceAbsolute,
71        IAssemblyInstanceAbsolute,
72        IChoiceInstance,
73        IChoiceGroupInstance> of(
74            @Nullable AssemblyModel binding,
75            @NonNull IBoundInstanceModelAssembly bindingInstance,
76            @NonNull IBindingDefinitionModelAssembly parent,
77            @NonNull INodeItemFactory nodeItemFactory) {
78      List<Object> instances;
79      return binding == null || (instances = binding.getInstances()) == null || instances.isEmpty()
80          ? IContainerModelAssemblySupport.empty()
81          : newInstance(
82              binding,
83              bindingInstance,
84              parent,
85              nodeItemFactory);
86    }
87  
88    private static IContainerModelAssemblySupport<
89        IModelInstanceAbsolute,
90        INamedModelInstanceAbsolute,
91        IFieldInstanceAbsolute,
92        IAssemblyInstanceAbsolute,
93        IChoiceInstance,
94        IChoiceGroupInstance> newInstance(
95            @NonNull AssemblyModel binding,
96            @NonNull IBoundInstanceModelAssembly bindingInstance,
97            @NonNull IBindingDefinitionModelAssembly parent,
98            @NonNull INodeItemFactory nodeItemFactory) {
99      AssemblyModelGenerator generator = new AssemblyModelGenerator(parent, nodeItemFactory);
100 
101     // TODO: make "instances" a constant
102     IBoundInstanceModelChoiceGroup instance = ObjectUtils.requireNonNull(
103         bindingInstance.getDefinition()
104             .getChoiceGroupInstanceByName(BindingConstants.METASCHEMA_CHOICE_GROUP_GROUP_AS_NAME));
105 
106     ObjectUtils.notNull(binding.getInstances()).forEach(obj -> {
107       assert obj != null;
108       IBoundInstanceModelGroupedAssembly objInstance
109           = (IBoundInstanceModelGroupedAssembly) instance.getItemInstance(obj);
110 
111       if (obj instanceof AssemblyReference) {
112         generator.addAssemblyInstance((AssemblyReference) obj, objInstance);
113       } else if (obj instanceof InlineDefineAssembly) {
114         generator.addAssemblyInstance((InlineDefineAssembly) obj, objInstance);
115       } else if (obj instanceof FieldReference) {
116         generator.addFieldInstance((FieldReference) obj, objInstance);
117       } else if (obj instanceof InlineDefineField) {
118         generator.addFieldInstance((InlineDefineField) obj, objInstance);
119       } else if (obj instanceof AssemblyModel.Choice) {
120         generator.addChoiceInstance((AssemblyModel.Choice) obj, objInstance);
121       } else if (obj instanceof AssemblyModel.ChoiceGroup) {
122         generator.addChoiceGroupInstance((AssemblyModel.ChoiceGroup) obj, objInstance);
123       } else {
124         throw new UnsupportedOperationException(String.format("Unknown model instance class: %s", obj.getClass()));
125       }
126     });
127 
128     return generator.getBuilder().buildAssembly();
129   }
130 
131   private AssemblyModelGenerator(
132       @NonNull IBindingDefinitionModelAssembly parent,
133       @NonNull INodeItemFactory nodeItemFactory) {
134     super(parent, nodeItemFactory, new DefaultAssemblyModelBuilder<>());
135   }
136 
137   private void addChoiceInstance(
138       @NonNull AssemblyModel.Choice obj,
139       @NonNull IBoundInstanceModelGroupedAssembly objInstance) {
140     getBuilder().append(new InstanceModelChoice(
141         obj,
142         objInstance,
143         choicePosition++,
144         getParent(),
145         getNodeItemFactory()));
146   }
147 
148   private void addChoiceGroupInstance(
149       @NonNull AssemblyModel.ChoiceGroup obj,
150       @NonNull IBoundInstanceModelGroupedAssembly objInstance) {
151     getBuilder().append(new InstanceModelChoiceGroup(
152         obj,
153         objInstance,
154         choiceGroupPosition++,
155         getParent(),
156         getNodeItemFactory()));
157   }
158 }

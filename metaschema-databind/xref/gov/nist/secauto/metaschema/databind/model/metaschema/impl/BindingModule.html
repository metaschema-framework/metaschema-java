62    /**
63     * Constructs a new Metaschema instance.
64     *
65     * @param resource
66     *          the resource from which the module was loaded
67     * @param rootDefinition
68     *          the underlying definition binding for the module
69     * @param binding
70     *          the module definition object bound to a Java object
71     * @param importedModules
72     *          the modules imported by this module
73     * @throws MetaschemaException
74     *           if a processing error occurs
75     */
76    @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
77    @SuppressFBWarnings(value = "CT_CONSTRUCTOR_THROW", justification = "Use of final fields")
78    public BindingModule( // NOPMD - unavoidable
79        @NonNull URI resource,
80        @NonNull IBoundDefinitionModelAssembly rootDefinition,
81        @NonNull METASCHEMA binding,
82        @NonNull List<? extends IBindingMetaschemaModule> importedModules) throws MetaschemaException {
83      super(importedModules);
84  
85      this.binding = binding;
86  
87      this.staticContext = ObjectUtils.notNull(Lazy.lazy(() -> {
88        StaticContext.Builder builder = StaticContext.builder()
89            .baseUri(resource)
90            .defaultModelNamespace(getXmlNamespace());
91  
92        getNamespaceBindings()
93            .forEach((prefix, ns) -> builder.namespace(
94                ObjectUtils.notNull(prefix), ObjectUtils.notNull(ns)));
95        return builder.build();
96      }));
97  
98      INodeItemFactory nodeItemFactory = INodeItemFactory.instance();
99      this.definitions = ObjectUtils.notNull(Lazy.lazy(() -> new Definitions(resource, rootDefinition, nodeItemFactory)));
100     this.documentNodeItem
101         = ObjectUtils.notNull(Lazy.lazy(() -> nodeItemFactory.newDocumentNodeItem(rootDefinition, resource, binding)));
102     this.moduleNodeItem
103         = ObjectUtils.notNull(Lazy.lazy(() -> nodeItemFactory.newModuleNodeItem(this)));
104   }

262   private final class Definitions {
263     @NonNull
264     private final Map<QName, IFlagDefinition> flagDefinitions;
265     @NonNull
266     private final Map<QName, IFieldDefinition> fieldDefinitions;
267     @NonNull
268     private final Map<QName, IAssemblyDefinition> assemblyDefinitions;
269     @NonNull
270     private final Map<QName, IAssemblyDefinition> rootAssemblyDefinitions;
271 
272     private Definitions(
273         @NonNull URI resource,
274         @NonNull IBoundDefinitionModelAssembly rootDefinition,
275         @NonNull INodeItemFactory nodeItemFactory) {
276 
277       this.flagDefinitions = new LinkedHashMap<>();
278       this.fieldDefinitions = new LinkedHashMap<>();
279       this.assemblyDefinitions = new LinkedHashMap<>();
280       this.rootAssemblyDefinitions = new LinkedHashMap<>();
281 
282       // create instance position counters
283       int globalFlagPosition = 0;
284       int globalFieldPosition = 0;
285       int globalAssemblyPosition = 0;
286 
287       IBoundInstanceModelChoiceGroup instance = ObjectUtils.requireNonNull(
288           rootDefinition.getChoiceGroupInstanceByName("definitions"));
289 
290       for (Object obj : binding.getDefinitions()) {
291         IBoundInstanceModelGroupedAssembly objInstance
292             = (IBoundInstanceModelGroupedAssembly) instance.getItemInstance(obj);
293 
294         if (obj instanceof METASCHEMA.DefineAssembly) {
295           IAssemblyDefinition definition = new DefinitionAssemblyGlobal(
296               (METASCHEMA.DefineAssembly) obj,
297               objInstance,
298               globalAssemblyPosition++,
299               BindingModule.this,
300               nodeItemFactory);
301           QName name = definition.getDefinitionQName();
302           assemblyDefinitions.put(name, definition);
303           if (definition.isRoot()) {
304             rootAssemblyDefinitions.put(name, definition);
305           }
306         } else if (obj instanceof METASCHEMA.DefineField) {
307           IFieldDefinition definition = new DefinitionFieldGlobal(
308               (METASCHEMA.DefineField) obj,
309               objInstance,
310               globalFieldPosition++,
311               BindingModule.this);
312           QName name = definition.getDefinitionQName();
313           fieldDefinitions.put(name, definition);
314         } else if (obj instanceof METASCHEMA.DefineFlag) {
315           IFlagDefinition definition = new DefinitionFlagGlobal(
316               (METASCHEMA.DefineFlag) obj,
317               objInstance,
318               globalFlagPosition++,
319               BindingModule.this);
320           QName name = definition.getDefinitionQName();
321           flagDefinitions.put(name, definition);
322         } else {
323           throw new IllegalStateException(
324               String.format("Unrecognized definition class '%s' in module '%s'.",
325                   obj.getClass(),
326                   resource.toASCIIString()));
327         }
328       }
329     }

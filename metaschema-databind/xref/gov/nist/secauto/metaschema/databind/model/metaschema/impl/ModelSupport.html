50    @NonNull
51    public static Map<IAttributable.Key, Set<String>> parseProperties(@NonNull List<Property> props) {
52      return CollectionUtil.unmodifiableMap(ObjectUtils.notNull(props.stream()
53          .collect(
54              Collectors.groupingBy(
55                  prop -> {
56                    String name = ObjectUtils.requireNonNull(prop.getName());
57                    URI namespace = prop.getNamespace();
58                    return namespace == null ? IAttributable.key(name)
59                        : IAttributable.key(name, ObjectUtils.notNull(namespace.toASCIIString()));
60                  },
61                  Collectors.mapping(
62                      prop -> ObjectUtils.requireNonNull(prop.getValue()),
63                      Collectors.toCollection(LinkedHashSet::new))))));
64    }

70    /**
71     * Translate a text scope value to the equivalent enumerated value.
72     *
73     * @param value
74     *          the text scope value
75     * @return the enumerated value
76     */
77    @SuppressWarnings("PMD.ImplicitSwitchFallThrough")
78    @NonNull
79    public static IDefinition.ModuleScope moduleScope(@NonNull String value) {
80      IDefinition.ModuleScope retval;
81      switch (value) {
82      case "local":
83        retval = IDefinition.ModuleScope.PRIVATE;
84        break;
85      case "global":
86      default:
87        retval = IDefinition.ModuleScope.PUBLIC;
88      }
89      return retval;
90    }

119   @NonNull
120   public static IDataTypeAdapter<?> dataType(@Nullable String dataType) {
121     IDataTypeAdapter<?> retval;
122     if (dataType == null) {
123       retval = MetaschemaDataTypeProvider.DEFAULT_DATA_TYPE;
124     } else {
125       retval = DataTypeService.getInstance().getJavaTypeAdapterByName(dataType);
126       if (retval == null) {
127         throw new IllegalStateException("Unrecognized data type: " + dataType);
128       }
129     }
130     return retval;
131   }

184   @NonNull
185   public static JsonGroupAsBehavior groupAsJsonBehavior(@Nullable String inJson) {
186     JsonGroupAsBehavior retval = IGroupable.DEFAULT_JSON_GROUP_AS_BEHAVIOR;
187     if (inJson != null) {
188       switch (inJson) {
189       case "ARRAY":
190         retval = JsonGroupAsBehavior.LIST;
191         break;
192       case "SINGLETON_OR_ARRAY":
193         retval = JsonGroupAsBehavior.SINGLETON_OR_LIST;
194         break;
195       case "BY_KEY":
196         retval = JsonGroupAsBehavior.KEYED;
197         break;
198       default:
199         retval = IGroupable.DEFAULT_JSON_GROUP_AS_BEHAVIOR;
200         break;
201       }
202     }
203     return retval;
204   }

221     }

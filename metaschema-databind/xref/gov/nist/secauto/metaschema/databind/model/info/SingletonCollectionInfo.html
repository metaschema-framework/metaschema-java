6   package gov.nist.secauto.metaschema.databind.model.info;
7   
8   import gov.nist.secauto.metaschema.core.model.IBoundObject;
9   import gov.nist.secauto.metaschema.core.util.CollectionUtil;
10  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
11  import gov.nist.secauto.metaschema.databind.io.BindingException;
12  import gov.nist.secauto.metaschema.databind.model.IBoundInstanceModel;
13  
14  import java.io.IOException;
15  import java.util.List;
16  
17  import edu.umd.cs.findbugs.annotations.NonNull;
18  import edu.umd.cs.findbugs.annotations.Nullable;
19  
20  class SingletonCollectionInfo<ITEM>
21      extends AbstractModelInstanceCollectionInfo<ITEM> {
22  
23    public SingletonCollectionInfo(@NonNull IBoundInstanceModel<ITEM> instance) {
24      super(instance);
25    }
26  
27    @SuppressWarnings("unchecked")
28    @Override
29    public List<ITEM> getItemsFromValue(Object value) {
30      return value == null ? CollectionUtil.emptyList() : CollectionUtil.singletonList((ITEM) value);
31    }
32  
33    @Override
34    public int size(Object value) {
35      return value == null ? 0 : 1;
36    }
37  
38    @Override
39    public boolean isEmpty(@Nullable Object value) {
40      return value == null;
41    }
42  
43    @SuppressWarnings("unchecked")
44    @Override
45    public Class<? extends ITEM> getItemType() {
46      return (Class<? extends ITEM>) getInstance().getItemType();
47    }
48  
49    @Override
50    public Object deepCopyItems(@NonNull IBoundObject fromObject, @NonNull IBoundObject toObject)
51        throws BindingException {
52      IBoundInstanceModel<ITEM> instance = getInstance();
53  
54      @SuppressWarnings("unchecked")
55      ITEM value = (ITEM) instance.getValue(fromObject);
56  
57      return value == null ? null : instance.deepCopyItem(ObjectUtils.requireNonNull(value), toObject);
58    }
59  
60    @SuppressWarnings("unchecked")
61    @Override
62    public ITEM emptyValue() {
63      return (ITEM) getInstance().getDefaultValue();
64    }
65  
66    @Override
67    public Object readItems(IModelInstanceReadHandler<ITEM> handler) throws IOException {
68      Object value = handler.readSingleton();
69      // REFACTOR: this can return a null value
70      return value == null ? emptyValue() : value;
71    }
72  
73    @SuppressWarnings("unchecked")
74    @Override
75    public void writeItems(IModelInstanceWriteHandler<ITEM> handler, Object value) throws IOException {
76      handler.writeSingleton((ITEM) value);
77    }
78  }

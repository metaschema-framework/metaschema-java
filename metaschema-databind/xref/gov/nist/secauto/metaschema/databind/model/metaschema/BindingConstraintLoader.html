63  /**
64   * Provides methods to load a constraint set expressed in XML.
65   * <p>
66   * Loaded constraint instances are cached to avoid the need to load them for
67   * every use. Any constraint set imported is also loaded and cached
68   * automatically.
69   */
70  public class BindingConstraintLoader
71      extends AbstractLoader<List<IConstraintSet>>
72      implements IConstraintLoader {
73  
74    @NonNull
75    private final IBoundLoader loader;
76  
77    public BindingConstraintLoader(@NonNull IBindingContext bindingContext) {
78      // ensure the bindings are registered
79      bindingContext.registerModule(MetaschemaModelModule.class);
80  
81      this.loader = bindingContext.newBoundLoader();
82      this.loader.enableFeature(DeserializationFeature.DESERIALIZE_VALIDATE_CONSTRAINTS);
83    }
84  
85    @Override
86    protected List<IConstraintSet> parseResource(@NonNull URI resource, @NonNull Deque<URI> visitedResources)
87        throws IOException {
88  
89      Object constraintsDocument = loader.load(resource);
90  
91      StaticContext.Builder builder = StaticContext.builder()
92          .baseUri(resource);
93  
94      builder.useWildcardWhenNamespaceNotDefaulted(true);
95  
96      List<IConstraintSet> retval;
97      if (constraintsDocument instanceof MetaschemaModuleConstraints) {
98        MetaschemaModuleConstraints obj = (MetaschemaModuleConstraints) constraintsDocument;
99  
100       // now check if this constraint set imports other constraint sets
101       List<MetaschemaModuleConstraints.Import> imports = CollectionUtil.listOrEmpty(obj.getImports());
102 
103       @NonNull Set<IConstraintSet> importedConstraints;
104       if (imports.isEmpty()) {
105         importedConstraints = CollectionUtil.emptySet();
106       } else {
107         try {
108           importedConstraints = new LinkedHashSet<>();
109           for (MetaschemaModuleConstraints.Import imported : imports) {
110             URI importedResource = imported.getHref();
111             importedResource = ObjectUtils.notNull(resource.resolve(importedResource));
112             importedConstraints.addAll(loadInternal(importedResource, visitedResources));
113           }
114         } catch (MetaschemaException ex) {
115           throw new IOException(ex);
116         }
117       }
118 
119       CollectionUtil.listOrEmpty(obj.getNamespaceBindings()).stream()
120           .forEach(binding -> builder.namespace(
121               ObjectUtils.notNull(binding.getPrefix()),
122               ObjectUtils.notNull(binding.getUri())));
123       ISource source = ISource.externalSource(builder.build());
124 
125       // now create this constraint set
126       retval = CollectionUtil.singletonList(new DefaultConstraintSet(
127           resource,
128           parseScopedConstraints(obj, source),
129           new LinkedHashSet<>(importedConstraints)));
130     } else if (constraintsDocument instanceof MetaschemaMetaConstraints) {
131       MetaschemaMetaConstraints obj = (MetaschemaMetaConstraints) constraintsDocument;
132 
133       // now check if this constraint set imports other constraint sets
134       List<MetaschemaMetaConstraints.Import> imports = CollectionUtil.listOrEmpty(obj.getImports());
135 
136       retval = new LinkedList<>();
137       if (!imports.isEmpty()) {
138         try {
139           for (MetaschemaMetaConstraints.Import imported : imports) {
140             URI importedResource = imported.getHref();
141             importedResource = ObjectUtils.notNull(resource.resolve(importedResource));
142             retval.addAll(loadInternal(importedResource, visitedResources));
143           }
144         } catch (MetaschemaException ex) {
145           throw new IOException(ex);
146         }
147       }
148 
149       CollectionUtil.listOrEmpty(obj.getNamespaceBindings()).stream()
150           .forEach(binding -> builder.namespace(
151               ObjectUtils.notNull(binding.getPrefix()),
152               ObjectUtils.notNull(binding.getUri())));
153 
154       ISource source = ISource.externalSource(builder.build());
155 
156       List<ITargetedConstraints> targetedConstraints = ObjectUtils.notNull(CollectionUtil.listOrEmpty(obj.getContexts())
157           .stream()
158           .flatMap(context -> parseContext(ObjectUtils.notNull(context), null, source)
159               .getTargetedConstraints().stream())
160           .collect(Collectors.toList()));
161       retval.add(new MetaConstraintSet(targetedConstraints));
162 
163       retval = CollectionUtil.unmodifiableList(retval);
164     } else {
165       throw new UnsupportedOperationException(String.format("Unsupported constraint content '%s'.", resource));
166     }
167     return retval;
168   }

170   /**
171    * Parse individual constraint definitions from the provided XMLBeans object.
172    *
173    * @param obj
174    *          the XMLBeans object
175    * @param source
176    *          the source of the constraint content
177    * @return the scoped constraint definitions
178    */
179   @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops") // intentional
180   @NonNull
181   protected List<IScopedContraints> parseScopedConstraints(
182       @NonNull MetaschemaModuleConstraints obj,
183       @NonNull ISource source) {
184     List<IScopedContraints> scopedConstraints = new LinkedList<>();
185 
186     for (MetaschemaModuleConstraints.Scope scope : CollectionUtil.listOrEmpty(obj.getScopes())) {
187       assert scope != null;
188 
189       List<ITargetedConstraints> targetedConstraints = new LinkedList<>();
190       try {
191         for (IValueConstraintsBase constraintsObj : CollectionUtil.listOrEmpty(scope.getConstraints())) {
192           if (constraintsObj instanceof MetaschemaModuleConstraints.Scope.Assembly) {
193             targetedConstraints.add(handleScopedAssembly(
194                 (MetaschemaModuleConstraints.Scope.Assembly) constraintsObj,
195                 source));
196           } else if (constraintsObj instanceof MetaschemaModuleConstraints.Scope.Field) {
197             targetedConstraints.add(handleScopedField(
198                 (MetaschemaModuleConstraints.Scope.Field) constraintsObj,
199                 source));
200           } else if (constraintsObj instanceof MetaschemaModuleConstraints.Scope.Flag) {
201             targetedConstraints.add(handleScopedFlag(
202                 (MetaschemaModuleConstraints.Scope.Flag) constraintsObj,
203                 source));
204           }
205         }
206       } catch (MetapathException | XmlValueNotSupportedException ex) {
207         if (ex.getCause() instanceof MetapathException) {
208           throw new MetapathException(
209               String.format("Unable to compile a Metapath in '%s'. %s",
210                   source.getSource(),
211                   ex.getLocalizedMessage()),
212               ex);
213         }
214         throw ex;
215       }
216 
217       URI namespace = ObjectUtils.requireNonNull(scope.getMetaschemaNamespace());
218       String shortName = ObjectUtils.requireNonNull(scope.getMetaschemaShortName());
219 
220       scopedConstraints.add(new DefaultScopedContraints(
221           namespace,
222           shortName,
223           CollectionUtil.unmodifiableList(targetedConstraints)));
224     }
225     return CollectionUtil.unmodifiableList(scopedConstraints);
226   }

228   private static AssemblyTargetedConstraints handleScopedAssembly(
229       @NonNull MetaschemaModuleConstraints.Scope.Assembly obj,
230       @NonNull ISource source) {
231     IModelConstrained constraints = new AssemblyConstraintSet();
232     ConstraintBindingSupport.parse(constraints, obj, source);
233     return new AssemblyTargetedConstraints(
234         ObjectUtils.requireNonNull(obj.getTarget()),
235         constraints);
236   }

260   private Context parseContext(
261       @NonNull MetapathContext contextObj,
262       @Nullable Context parent,
263       @NonNull ISource source) {
264 
265     List<String> metapaths;
266     if (parent == null) {
267       metapaths = ObjectUtils.notNull(CollectionUtil.listOrEmpty(contextObj.getMetapaths()).stream()
268           .map(MetaschemaMetapath::getTarget)
269           .collect(Collectors.toList()));
270     } else {
271       List<String> parentMetapaths = parent.getMetapaths().stream()
272           .collect(Collectors.toList());
273       metapaths = ObjectUtils.notNull(CollectionUtil.listOrEmpty(contextObj.getMetapaths()).stream()
274           .map(MetaschemaMetapath::getTarget)
275           .flatMap(childPath -> parentMetapaths.stream()
276               .map(parentPath -> parentPath + '/' + childPath))
277           .collect(Collectors.toList()));
278     }
279 
280     AssemblyConstraints contextConstraints = contextObj.getConstraints();
281     IModelConstrained constraints = new AssemblyConstraintSet();
282     if (contextConstraints != null) {
283       ConstraintBindingSupport.parse(constraints, contextConstraints, source);
284     }
285     Context context = new Context(metapaths, constraints);
286 
287     List<Context> childContexts = ObjectUtils.notNull(CollectionUtil.listOrEmpty(contextObj.getContexts()).stream()
288         .map(childObj -> parseContext(ObjectUtils.notNull(childObj), context, source))
289         .collect(Collectors.toList()));
290 
291     context.addAll(childContexts);
292 
293     return context;
294   }

296   private static class Context {
297     @NonNull
298     private final List<String> metapaths;
299     @NonNull
300     private final List<Context> childContexts = new LinkedList<>();
301     @NonNull
302     private final Lazy<List<ITargetedConstraints>> targetedConstraints;
303 
304     public Context(
305         @NonNull List<String> metapaths,
306         @NonNull IModelConstrained constraints) {
307       this.metapaths = metapaths;
308       this.targetedConstraints = ObjectUtils.notNull(Lazy.lazy(() -> {
309 
310         Stream<ITargetedConstraints> paths = getMetapaths().stream()
311             .map(metapath -> new MetaTargetedContraints(ObjectUtils.notNull(metapath), constraints));
312         Stream<ITargetedConstraints> childPaths = childContexts.stream()
313             .flatMap(child -> child.getTargetedConstraints().stream());
314 
315         return Stream.concat(paths, childPaths)
316             .collect(Collectors.toUnmodifiableList());
317       }));
318     }
319 
320     @NonNull
321     public List<ITargetedConstraints> getTargetedConstraints() {
322       return ObjectUtils.notNull(targetedConstraints.get());
323     }
324 
325     public void addAll(@NonNull Collection<Context> childContexts) {
326       childContexts.addAll(childContexts);
327     }
328 
329     @NonNull
330     public List<String> getMetapaths() {
331       return metapaths;
332     }
333   }

335   private static class MetaTargetedContraints
336       extends AbstractTargetedConstraints<IModelConstrained>
337       implements IFeatureModelConstrained {
338 
339     protected MetaTargetedContraints(
340         @NonNull String target,
341         @NonNull IModelConstrained constraints) {
342       super(target, constraints);
343     }
344 
345     /**
346      * Apply the constraints to the provided {@code definition}.
347      * <p>
348      * This will be called when a definition is found that matches the target
349      * expression.
350      *
351      * @param definition
352      *          the definition to apply the constraints to.
353      */
354     protected void applyTo(@NonNull IDefinition definition) {
355       getLetExpressions().values().forEach(definition::addLetExpression);
356       getAllowedValuesConstraints().forEach(definition::addConstraint);
357       getMatchesConstraints().forEach(definition::addConstraint);
358       getIndexHasKeyConstraints().forEach(definition::addConstraint);
359       getExpectConstraints().forEach(definition::addConstraint);
360     }
361 
362     protected void applyTo(@NonNull IAssemblyDefinition definition) {
363       applyTo((IDefinition) definition);
364       getIndexConstraints().forEach(definition::addConstraint);
365       getUniqueConstraints().forEach(definition::addConstraint);
366       getHasCardinalityConstraints().forEach(definition::addConstraint);
367     }
368 
369     @Override
370     public void target(IFlagDefinition definition) {
371       applyTo(definition);
372     }
373 
374     @Override
375     public void target(IFieldDefinition definition) {
376       applyTo(definition);
377     }
378 
379     @Override
380     public void target(IAssemblyDefinition definition) {
381       applyTo(definition);
382     }
383   }

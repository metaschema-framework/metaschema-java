6   package gov.nist.secauto.metaschema.databind.codegen.typeinfo;
7   
8   import gov.nist.secauto.metaschema.core.model.IModule;
9   import gov.nist.secauto.metaschema.databind.codegen.IGeneratedClass;
10  import gov.nist.secauto.metaschema.databind.codegen.IGeneratedDefinitionClass;
11  import gov.nist.secauto.metaschema.databind.codegen.IGeneratedModuleClass;
12  import gov.nist.secauto.metaschema.databind.codegen.typeinfo.def.IModelDefinitionTypeInfo;
13  
14  import java.io.IOException;
15  import java.net.URI;
16  import java.nio.file.Path;
17  import java.util.Collection;
18  
19  import edu.umd.cs.findbugs.annotations.NonNull;
20  
21  public interface IMetaschemaClassFactory {
22    /**
23     * Get a new instance of the default class generation factory that uses the
24     * provided {@code typeResolver}.
25     *
26     * @param typeResolver
27     *          the resolver used to generate type information for Metasschema
28     *          constructs
29     * @return the new class factory
30     */
31    @NonNull
32    static IMetaschemaClassFactory newInstance(@NonNull ITypeResolver typeResolver) {
33      return DefaultMetaschemaClassFactory.newInstance(typeResolver);
34    }
35  
36    /**
37     * Get the type resolver used to generate type information for Metasschema
38     * constructs represented as Java classes, fields, and methods.
39     *
40     * @return the type resolver
41     */
42    @NonNull
43    ITypeResolver getTypeResolver();
44  
45    /**
46     * Generate a class in the provided {@code targetDirectory} that represents the
47     * provided Module {@code module}.
48     *
49     * @param module
50     *          the Module module to generate the class for
51     * @param targetDirectory
52     *          the directory to generate the Java class in
53     * @return information about the generated class
54     * @throws IOException
55     *           if an error occurred while generating the Java class
56     */
57    @NonNull
58    IGeneratedModuleClass generateClass(
59        @NonNull IModule module,
60        @NonNull Path targetDirectory) throws IOException;
61  
62    /**
63     * Generate a class in the provided {@code targetDirectory} that represents the
64     * provided Module definition's {@code typeInfo}.
65     *
66     * @param typeInfo
67     *          the type information for the class to generate
68     * @param targetDirectory
69     *          the directory to generate the Java class in
70     * @return the generated class details
71     * @throws IOException
72     *           if an error occurred while generating the Java class
73     */
74    @NonNull
75    IGeneratedDefinitionClass generateClass(
76        @NonNull IModelDefinitionTypeInfo typeInfo,
77        @NonNull Path targetDirectory) throws IOException;
78  
79    /**
80     * Generate a package-info.java class in the provided {@code targetDirectory}
81     * that represents a collection of Module modules.
82     *
83     * @param javaPackage
84     *          the Java package name to use
85     * @param xmlNamespace
86     *          the default XML namespace for all bound Module information elements
87     *          in the generated package
88     * @param metaschemaProductions
89     *          a collection of previously generated Module modules and definition
90     *          classes
91     * @param targetDirectory
92     *          the directory to generate the Java class in
93     * @return the generated class details
94     * @throws IOException
95     *           if an error occurred while generating the Java class
96     */
97    @NonNull
98    IGeneratedClass generatePackageInfoClass(
99        @NonNull String javaPackage,
100       @NonNull URI xmlNamespace,
101       @NonNull Collection<IGeneratedModuleClass> metaschemaProductions,
102       @NonNull Path targetDirectory) throws IOException;
103 }

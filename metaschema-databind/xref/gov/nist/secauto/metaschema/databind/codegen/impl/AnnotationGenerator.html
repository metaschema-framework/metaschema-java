75    /**
76     * Get the default vale of the given member of an annotation.
77     *
78     * @param annotation
79     *          the annotation to analyze
80     * @param member
81     *          the annotation member to analyze
82     * @return the default value for the annotation member or {@code null} if there
83     *         is not default value
84     */
85    public static Object getDefaultValue(Class<?> annotation, String member) {
86      Method method;
87      try {
88        method = annotation.getDeclaredMethod(member);
89      } catch (NoSuchMethodException ex) {
90        throw new IllegalArgumentException(ex);
91      }
92      Object retval;
93      try {
94        retval = method.getDefaultValue();
95      } catch (TypeNotPresentException ex) {
96        retval = null; // NOPMD readability
97      }
98      return retval;
99    }

101   private static void buildConstraint(Class<?> annotationType, AnnotationSpec.Builder annotation,
102       IConstraint constraint) {
103     String id = constraint.getId();
104     if (id != null) {
105       annotation.addMember("id", "$S", id);
106     }
107 
108     String formalName = constraint.getFormalName();
109     if (formalName != null) {
110       annotation.addMember("formalName", "$S", formalName);
111     }
112 
113     MarkupLine description = constraint.getDescription();
114     if (description != null) {
115       annotation.addMember("description", "$S", description.toMarkdown());
116     }
117 
118     annotation.addMember("level", "$T.$L", IConstraint.Level.class, constraint.getLevel());
119 
120     String target = constraint.getTarget();
121     if (!target.equals(getDefaultValue(annotationType, "target"))) {
122       annotation.addMember("target", "$S", target);
123     }
124   }

126   public static void buildValueConstraints(
127       @NonNull AnnotationSpec.Builder builder,
128       @NonNull IFlagDefinition definition) {
129 
130     Map<QName, ? extends ILet> lets = definition.getLetExpressions();
131     if (!lets.isEmpty() || !definition.getConstraints().isEmpty()) {
132       AnnotationSpec.Builder annotation = AnnotationSpec.builder(ValueConstraints.class);
133       assert annotation != null;
134 
135       applyLetAssignments(annotation, lets);
136       applyAllowedValuesConstraints(annotation, definition.getAllowedValuesConstraints());
137       applyIndexHasKeyConstraints(annotation, definition.getIndexHasKeyConstraints());
138       applyMatchesConstraints(annotation, definition.getMatchesConstraints());
139       applyExpectConstraints(annotation, definition.getExpectConstraints());
140 
141       builder.addMember("valueConstraints", "$L", annotation.build());
142     }
143   }

145   public static void buildValueConstraints(
146       @NonNull AnnotationSpec.Builder builder,
147       @NonNull IModelDefinition definition) {
148 
149     Map<QName, ? extends ILet> lets = definition.getLetExpressions();
150     List<? extends IAllowedValuesConstraint> allowedValues = definition.getAllowedValuesConstraints();
151     List<? extends IIndexHasKeyConstraint> indexHasKey = definition.getIndexHasKeyConstraints();
152     List<? extends IMatchesConstraint> matches = definition.getMatchesConstraints();
153     List<? extends IExpectConstraint> expects = definition.getExpectConstraints();
154 
155     if (!lets.isEmpty() || !allowedValues.isEmpty() || !indexHasKey.isEmpty() || !matches.isEmpty()
156         || !expects.isEmpty()) {
157       AnnotationSpec.Builder annotation = AnnotationSpec.builder(ValueConstraints.class);
158       assert annotation != null;
159 
160       applyLetAssignments(annotation, lets);
161       applyAllowedValuesConstraints(annotation, allowedValues);
162       applyIndexHasKeyConstraints(annotation, indexHasKey);
163       applyMatchesConstraints(annotation, matches);
164       applyExpectConstraints(annotation, expects);
165 
166       builder.addMember("valueConstraints", "$L", annotation.build());
167     }
168   }

170   public static void buildAssemblyConstraints(
171       @NonNull AnnotationSpec.Builder builder,
172       @NonNull IAssemblyDefinition definition) {
173 
174     List<? extends IIndexConstraint> index = definition.getIndexConstraints();
175     List<? extends IUniqueConstraint> unique = definition.getUniqueConstraints();
176     List<? extends ICardinalityConstraint> cardinality = definition.getHasCardinalityConstraints();
177 
178     if (!index.isEmpty() || !unique.isEmpty() || !cardinality.isEmpty()) {
179       AnnotationSpec.Builder annotation = ObjectUtils.notNull(AnnotationSpec.builder(AssemblyConstraints.class));
180 
181       applyIndexConstraints(annotation, index);
182       applyUniqueConstraints(annotation, unique);
183       applyHasCardinalityConstraints(definition, annotation, cardinality);
184 
185       builder.addMember("modelConstraints", "$L", annotation.build());
186     }
187   }

208   private static void applyAllowedValuesConstraints(
209       @NonNull AnnotationSpec.Builder annotation,
210       @NonNull List<? extends IAllowedValuesConstraint> constraints) {
211     for (IAllowedValuesConstraint constraint : constraints) {
212       AnnotationSpec.Builder constraintAnnotation = AnnotationSpec.builder(AllowedValues.class);
213       buildConstraint(AllowedValues.class, constraintAnnotation, constraint);
214 
215       boolean isAllowedOther = constraint.isAllowedOther();
216       if (isAllowedOther != (boolean) getDefaultValue(AllowedValues.class, "allowOthers")) {
217         constraintAnnotation.addMember("allowOthers", "$L", isAllowedOther);
218       }
219 
220       for (IAllowedValue value : constraint.getAllowedValues().values()) {
221         AnnotationSpec.Builder valueAnnotation = AnnotationSpec.builder(AllowedValue.class);
222 
223         valueAnnotation.addMember("value", "$S", value.getValue());
224         valueAnnotation.addMember("description", "$S", value.getDescription().toMarkdown());
225 
226         constraintAnnotation.addMember("values", "$L", valueAnnotation.build());
227       }
228 
229       MarkupMultiline remarks = constraint.getRemarks();
230       if (remarks != null) {
231         constraintAnnotation.addMember("remarks", "$S", remarks.toMarkdown());
232       }
233       annotation.addMember("allowedValues", "$L", constraintAnnotation.build());
234     }
235   }

257   private static void buildKeyFields(
258       @NonNull Builder constraintAnnotation,
259       @NonNull List<? extends IKeyField> keyFields) {
260     for (IKeyField key : keyFields) {
261       AnnotationSpec.Builder keyAnnotation = AnnotationSpec.builder(KeyField.class);
262 
263       String target = key.getTarget();
264       if (!target.equals(getDefaultValue(KeyField.class, "target"))) {
265         keyAnnotation.addMember("target", "$S", target);
266       }
267 
268       Pattern pattern = key.getPattern();
269       if (pattern != null) {
270         keyAnnotation.addMember("pattern", "$S", pattern.pattern());
271       }
272 
273       MarkupMultiline remarks = key.getRemarks();
274       if (remarks != null) {
275         keyAnnotation.addMember("remarks", "$S", remarks.toMarkdown());
276       }
277 
278       constraintAnnotation.addMember("keyFields", "$L", keyAnnotation.build());
279     }

282   private static void applyMatchesConstraints(
283       @NonNull AnnotationSpec.Builder annotation,
284       @NonNull List<? extends IMatchesConstraint> constraints) {
285     for (IMatchesConstraint constraint : constraints) {
286       AnnotationSpec.Builder constraintAnnotation = AnnotationSpec.builder(Matches.class);
287       buildConstraint(Matches.class, constraintAnnotation, constraint);
288 
289       Pattern pattern = constraint.getPattern();
290       if (pattern != null) {
291         constraintAnnotation.addMember("pattern", "$S", pattern.pattern());
292       }
293 
294       IDataTypeAdapter<?> dataType = constraint.getDataType();
295       if (dataType != null) {
296         constraintAnnotation.addMember("typeAdapter", "$T.class", dataType.getClass());
297       }
298 
299       MarkupMultiline remarks = constraint.getRemarks();
300       if (remarks != null) {
301         constraintAnnotation.addMember("remarks", "$S", remarks.toMarkdown());
302       }
303       annotation.addMember("matches", "$L", constraintAnnotation.build());
304     }

370   @SuppressWarnings({
371       "PMD.GuardLogStatement" // guarded in outer calls
372   })
373   private static void checkCardinalities(
374       @NonNull IAssemblyDefinition definition,
375       @NonNull ICardinalityConstraint constraint,
376       @NonNull ISequence<? extends IDefinitionNodeItem<?, ?>> instanceSet,
377       @NonNull LogBuilder logBuilder) {
378 
379     LogBuilder warn = LOGGER.atWarn();
380     for (IDefinitionNodeItem<?, ?> item : instanceSet.getValue()) {
381       INamedInstance instance = item.getInstance();
382       if (instance instanceof INamedModelInstanceAbsolute) {
383         INamedModelInstanceAbsolute modelInstance = (INamedModelInstanceAbsolute) instance;
384 
385         checkMinOccurs(definition, constraint, modelInstance, logBuilder);
386         checkMaxOccurs(definition, constraint, modelInstance, logBuilder);
387       } else {
388         warn.log(String.format(
389             "Definition '%s' has min-occurs=%d cardinality constraint targeting '%s' that is not a model instance",
390             definition.getName(), constraint.getMinOccurs(), constraint.getTarget()));
391       }
392     }
393   }

395   @SuppressWarnings({
396       "PMD.GuardLogStatement" // guarded in outer calls
397   })
398   private static void checkMinOccurs(
399       @NonNull IAssemblyDefinition definition,
400       @NonNull ICardinalityConstraint constraint,
401       @NonNull INamedModelInstanceAbsolute modelInstance,
402       @NonNull LogBuilder logBuilder) {
403     Integer minOccurs = constraint.getMinOccurs();
404     if (minOccurs != null) {
405       if (minOccurs == modelInstance.getMinOccurs()) {
406         logBuilder.log(String.format(
407             "Definition '%s' has min-occurs=%d cardinality constraint targeting '%s' that is redundant with a"
408                 + " targeted instance named '%s' that requires min-occurs=%d",
409             definition.getName(), minOccurs, constraint.getTarget(),
410             modelInstance.getName(),
411             modelInstance.getMinOccurs()));
412       } else if (minOccurs < modelInstance.getMinOccurs()) {
413         logBuilder.log(String.format(
414             "Definition '%s' has min-occurs=%d cardinality constraint targeting '%s' that conflicts with a"
415                 + " targeted instance named '%s' that requires min-occurs=%d",
416             definition.getName(), minOccurs, constraint.getTarget(),
417             modelInstance.getName(),
418             modelInstance.getMinOccurs()));
419       }
420     }
421   }

451   private static void applyHasCardinalityConstraints(
452       @NonNull IAssemblyDefinition definition,
453       @NonNull AnnotationSpec.Builder annotation,
454       @NonNull List<? extends ICardinalityConstraint> constraints) {
455 
456     DynamicContext dynamicContext = new DynamicContext();
457     dynamicContext.disablePredicateEvaluation();
458 
459     for (ICardinalityConstraint constraint : constraints) {
460 
461       IAssemblyNodeItem definitionNodeItem
462           = INodeItemFactory.instance().newAssemblyNodeItem(definition);
463 
464       ISequence<? extends IDefinitionNodeItem<?, ?>> instanceSet
465           = constraint.matchTargets(definitionNodeItem, dynamicContext);
466 
467       if (LOGGER.isWarnEnabled()) {
468         checkCardinalities(definition, constraint, instanceSet, ObjectUtils.notNull(LOGGER.atWarn()));
469       }
470 
471       AnnotationSpec.Builder constraintAnnotation = AnnotationSpec.builder(HasCardinality.class);
472 
473       buildConstraint(HasCardinality.class, constraintAnnotation, constraint);
474 
475       Integer minOccurs = constraint.getMinOccurs();
476       if (minOccurs != null && !minOccurs.equals(getDefaultValue(HasCardinality.class, "minOccurs"))) {
477         constraintAnnotation.addMember("minOccurs", "$L", minOccurs);
478       }
479 
480       Integer maxOccurs = constraint.getMaxOccurs();
481       if (maxOccurs != null && !maxOccurs.equals(getDefaultValue(HasCardinality.class, "maxOccurs"))) {
482         constraintAnnotation.addMember("maxOccurs", "$L", maxOccurs);
483       }
484 
485       annotation.addMember("cardinality", "$L", constraintAnnotation.build());
486 
487       MarkupMultiline remarks = constraint.getRemarks();
488       if (remarks != null) {
489         constraintAnnotation.addMember("remarks", "$S", remarks.toMarkdown());
490       }
491     }

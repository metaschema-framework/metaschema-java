42  class DefaultTypeResolver implements ITypeResolver {
43    private static final Logger LOGGER = LogManager.getLogger(DefaultTypeResolver.class);
44  
45    private final Map<String, Set<String>> packageToClassNamesMap = new ConcurrentHashMap<>();
46    private final Map<IModelDefinition, ClassName> definitionToTypeMap = new ConcurrentHashMap<>();
47    private final Map<IModule, ClassName> moduleToTypeMap = new ConcurrentHashMap<>();
48    private final Map<IAssemblyDefinition, IAssemblyDefinitionTypeInfo> assemblyDefinitionToTypeInfoMap
49        = new ConcurrentHashMap<>();
50    private final Map<IFieldDefinition, IFieldDefinitionTypeInfo> fieldDefinitionToTypeInfoMap
51        = new ConcurrentHashMap<>();
52    private final Map<IDefinitionTypeInfo, Set<String>> typeInfoToPropertyNameMap = new ConcurrentHashMap<>();

54    @NonNull
55    private final IBindingConfiguration bindingConfiguration;
56  
57    public DefaultTypeResolver(@NonNull IBindingConfiguration bindingConfiguration) {
58      this.bindingConfiguration = bindingConfiguration;
59    }
60  
61    protected IBindingConfiguration getBindingConfiguration() {
62      return bindingConfiguration;
63    }

65    @Override
66    public IAssemblyDefinitionTypeInfo getTypeInfo(@NonNull IAssemblyDefinition definition) {
67      return ObjectUtils.notNull(assemblyDefinitionToTypeInfoMap.computeIfAbsent(
68          definition,
69          def -> IAssemblyDefinitionTypeInfo.newTypeInfo(ObjectUtils.notNull(def),
70              this)));
71    }
72  
73    @Override
74    public IFieldDefinitionTypeInfo getTypeInfo(@NonNull IFieldDefinition definition) {
75      return ObjectUtils.notNull(fieldDefinitionToTypeInfoMap.computeIfAbsent(
76          definition,
77          def -> IFieldDefinitionTypeInfo.newTypeInfo(ObjectUtils.notNull(def),
78              this)));
79    }
80  
81    @Override
82    public IModelDefinitionTypeInfo getTypeInfo(@NonNull IModelDefinition definition) {
83      IModelDefinitionTypeInfo retval;
84      if (definition instanceof IAssemblyDefinition) {
85        retval = getTypeInfo((IAssemblyDefinition) definition);
86      } else if (definition instanceof IFieldDefinition) {
87        retval = getTypeInfo((IFieldDefinition) definition);
88      } else {
89        throw new IllegalStateException(String.format("Unknown type '%s'",
90            definition.getClass().getName()));
91      }
92      return retval;
93    }
94  
95    @Override
96    public IGroupedNamedModelInstanceTypeInfo getTypeInfo(
97        @NonNull INamedModelInstanceGrouped modelInstance,
98        @NonNull IChoiceGroupTypeInfo choiceGroupTypeInfo) {
99      IGroupedNamedModelInstanceTypeInfo retval;
100     if (modelInstance instanceof IAssemblyInstanceGrouped) {
101       retval = getTypeInfo((IAssemblyInstanceGrouped) modelInstance, choiceGroupTypeInfo);
102     } else if (modelInstance instanceof IFieldInstanceGrouped) {
103       retval = getTypeInfo((IFieldInstanceGrouped) modelInstance, choiceGroupTypeInfo);
104     } else {
105       throw new IllegalStateException(String.format("Unknown type '%s'",
106           modelInstance.getClass().getName()));
107     }
108     return retval;
109   }

125   @NonNull
126   private ClassName getFlagContainerClassName(
127       @NonNull IModelDefinition definition,
128       @NonNull String packageName,
129       @NonNull String suggestedClassName) {
130     ClassName retval;
131     if (definition.isInline()) {
132       // this is a local definition, which means a child class needs to be generated
133       INamedInstance inlineInstance = definition.getInlineInstance();
134       IModelDefinition parentDefinition = inlineInstance.getContainingDefinition();
135       ClassName parentClassName = getClassName(parentDefinition);
136       retval = getSubclassName(parentClassName, suggestedClassName, definition);
137     } else {
138       String className = generateClassName(packageName, suggestedClassName, definition);
139       retval = ObjectUtils.notNull(ClassName.get(packageName, className));
140     }
141     return retval;
142   }
143 
144   @Override
145   public ClassName getSubclassName(
146       @NonNull ClassName parentClass,
147       @NonNull String suggestedClassName,
148       @NonNull IModelDefinition definition) {
149     String name = generateClassName(
150         ObjectUtils.notNull(parentClass.canonicalName()),
151         ClassUtils.toClassName(suggestedClassName),
152         definition);
153     return ObjectUtils.notNull(parentClass.nestedClass(name));
154   }

156   @Override
157   @NonNull
158   public ClassName getClassName(@NonNull IModelDefinition definition) {
159     return ObjectUtils.notNull(definitionToTypeMap.computeIfAbsent(
160         definition,
161         def -> {
162           String packageName = getBindingConfiguration().getPackageNameForModule(def.getContainingModule());
163           String suggestedClassName = getBindingConfiguration().getClassName(definition);
164           return getFlagContainerClassName(def, packageName, suggestedClassName);
165         }));
166   }
167 
168   @Override
169   public ClassName getClassName(@NonNull INamedModelInstanceTypeInfo typeInfo) {
170     return getClassName(typeInfo.getInstance().getDefinition());
171   }

173   @Override
174   public ClassName getClassName(IChoiceGroupInstance instance) {
175     // TODO: Support some form of binding override for a common interface type
176     return ObjectUtils.notNull(ClassName.get(Object.class));
177   }

179   @Override
180   public ClassName getClassName(IModule module) {
181     return ObjectUtils.notNull(moduleToTypeMap.computeIfAbsent(
182         module,
183         mod -> {
184           assert mod != null;
185           String packageName = getBindingConfiguration().getPackageNameForModule(mod);
186           String className = getBindingConfiguration().getClassName(mod);
187           String classNameBase = className;
188           int index = 1;
189           while (isClassNameClash(packageName, className)) {
190             className = classNameBase + Integer.toString(index);
191           }
192           addClassName(packageName, className);
193           return ClassName.get(packageName, className);
194         }));
195   }

197   @NonNull
198   protected Set<String> getClassNamesFor(@NonNull String packageOrTypeName) {
199     return ObjectUtils.notNull(packageToClassNamesMap.computeIfAbsent(
200         packageOrTypeName,
201         pkg -> Collections.synchronizedSet(new LinkedHashSet<>())));
202   }
203 
204   protected boolean isClassNameClash(@NonNull String packageOrTypeName, @NonNull String className) {
205     return getClassNamesFor(packageOrTypeName).contains(className);
206   }
207 
208   protected boolean addClassName(@NonNull String packageOrTypeName, @NonNull String className) {
209     return getClassNamesFor(packageOrTypeName).add(className);
210   }
211 
212   private String generateClassName(
213       @NonNull String packageOrTypeName,
214       @NonNull String suggestedClassName,
215       @NonNull IModelDefinition definition) {
216     @NonNull String retval = suggestedClassName;
217     Set<String> classNames = getClassNamesFor(packageOrTypeName);
218     synchronized (classNames) {
219       boolean clash = false;
220       if (classNames.contains(suggestedClassName)) {
221         clash = true;
222         // first try to append the metaschema's short name
223         String metaschemaShortName = definition.getContainingModule().getShortName();
224         retval = ClassUtils.toClassName(suggestedClassName + StringUtils.capitalize(metaschemaShortName));
225       }
226 
227       String classNameBase = retval;
228       int index = 1;
229       while (classNames.contains(retval)) {
230         retval = classNameBase + Integer.toString(index++);
231       }
232       classNames.add(retval);
233 
234       if (clash && LOGGER.isWarnEnabled()) {
235         LOGGER.warn(String.format(
236             "Class name '%s', based on '%s' in '%s', clashes with another bound class. Using '%s' instead.",
237             suggestedClassName,
238             definition.getName(),
239             definition.getContainingModule().getLocation(),
240             retval));
241       }
242     }
243     return retval;
244   }

246   @Override
247   public ClassName getBaseClassName(IModelDefinition definition) {
248     String className = bindingConfiguration.getQualifiedBaseClassName(definition);
249     ClassName retval = null;
250     if (className != null) {
251       retval = ClassName.bestGuess(className);
252     }
253     return retval;
254   }
255 
256   @Override
257   public List<ClassName> getSuperinterfaces(IModelDefinition definition) {
258     List<String> classNames = bindingConfiguration.getQualifiedSuperinterfaceClassNames(definition);
259     return ObjectUtils.notNull(classNames.stream()
260         .map(ClassName::bestGuess)
261         .collect(Collectors.toUnmodifiableList()));
262   }
263 
264   @Override
265   public String getPackageName(@NonNull IModule module) {
266     return bindingConfiguration.getPackageNameForModule(module);
267   }
268 
269   @Override
270   @NonNull
271   public String getPropertyName(IDefinitionTypeInfo parent, String name) {
272     synchronized (typeInfoToPropertyNameMap) {
273       Set<String> propertyNames = typeInfoToPropertyNameMap.get(parent);
274       if (propertyNames == null) {
275         propertyNames = new HashSet<>();
276       }
277 
278       String retval = name;
279       int index = 0;
280       while (propertyNames.contains(retval)) {
281         // append an integer value to make the name unique
282         retval = ClassUtils.toPropertyName(name + Integer.toString(++index));
283       }
284       propertyNames.add(retval);
285       return retval;
286     }
287   }

6   package gov.nist.secauto.metaschema.databind.codegen.typeinfo;
7   
8   import com.squareup.javapoet.AnnotationSpec;
9   import com.squareup.javapoet.ClassName;
10  import com.squareup.javapoet.TypeSpec;
11  
12  import gov.nist.secauto.metaschema.core.model.IModelDefinition;
13  import gov.nist.secauto.metaschema.core.model.INamedModelInstanceGrouped;
14  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
15  
16  import org.apache.commons.lang3.StringUtils;
17  
18  import java.lang.annotation.Annotation;
19  import java.util.LinkedHashSet;
20  import java.util.Set;
21  
22  import javax.lang.model.element.Modifier;
23  
24  import edu.umd.cs.findbugs.annotations.NonNull;
25  
26  public abstract class AbstractGroupedNamedModelInstanceTypeInfo<I extends INamedModelInstanceGrouped>
27      implements IGroupedNamedModelInstanceTypeInfo {
28    @NonNull
29    private final I instance;
30    @NonNull
31    private final IChoiceGroupTypeInfo parentTypeInfo;
32  
33    protected AbstractGroupedNamedModelInstanceTypeInfo(
34        @NonNull I instance,
35        @NonNull IChoiceGroupTypeInfo parentTypeInfo) {
36      this.instance = instance;
37      this.parentTypeInfo = parentTypeInfo;
38    }
39  
40    protected abstract Class<? extends Annotation> getBindingAnnotation();
41  
42    protected abstract void applyInstanceAnnotation(
43        @NonNull AnnotationSpec.Builder instanceAnnotation,
44        @NonNull AnnotationSpec.Builder choiceGroupAnnotation);
45  
46    @NonNull
47    protected I getInstance() {
48      return instance;
49    }
50  
51    protected IChoiceGroupTypeInfo getChoiceGroupTypeInfo() {
52      return parentTypeInfo;
53    }
54  
55    @Override
56    public Set<IModelDefinition> generateMemberAnnotation(
57        @NonNull AnnotationSpec.Builder choiceGroupAnnotation,
58        @NonNull TypeSpec.Builder typeBuilder,
59        boolean requireExtension) {
60  
61      AnnotationSpec.Builder memberAnnotation = ObjectUtils.notNull(AnnotationSpec.builder(getBindingAnnotation()));
62  
63      TypeInfoUtils.buildCommonBindingAnnotationValues(getInstance(), memberAnnotation);
64  
65      Set<IModelDefinition> retval = new LinkedHashSet<>();
66  
67      I instance = getInstance();
68      IModelDefinition definition = getInstance().getDefinition();
69  
70      IChoiceGroupTypeInfo choiceGroupTypeInfo = getChoiceGroupTypeInfo();
71      ITypeResolver typeResolver = choiceGroupTypeInfo.getParentTypeInfo().getTypeResolver();
72  
73      ClassName itemTypeName;
74      if (definition.isInline()) {
75        // these definitions will be generated as standalone child classes
76        itemTypeName = typeResolver.getClassName(definition);
77        retval.add(definition);
78      } else if (requireExtension) {
79        // these definitions will be generated as an extension of a global class
80        ClassName extendedClassName = typeResolver.getClassName(definition);
81        itemTypeName = typeResolver.getSubclassName(
82            choiceGroupTypeInfo.getParentTypeInfo().getClassName(),
83            ObjectUtils.notNull(StringUtils.capitalize(instance.getEffectiveDisciminatorValue())),
84            definition);
85  
86        TypeSpec.Builder subClass = TypeSpec.classBuilder(itemTypeName);
87        subClass.superclass(extendedClassName);
88        subClass.addModifiers(Modifier.PUBLIC, Modifier.STATIC); // , Modifier.FINAL);
89        // subClass.addField(
90        // FieldSpec.builder(String.class, "DISCRIMINATOR", Modifier.PUBLIC,
91        // Modifier.STATIC, Modifier.FINAL)
92        // .initializer("\"" + instance.getEffectiveDisciminatorValue() + "\"")
93        // .build());
94        typeBuilder.addType(subClass.build());
95      } else {
96        // reference the global class
97        itemTypeName = typeResolver.getClassName(definition);
98      }
99  
100     memberAnnotation.addMember("binding", "$T.class", itemTypeName);
101 
102     applyInstanceAnnotation(memberAnnotation, choiceGroupAnnotation);
103 
104     return retval;
105   }
106 
107 }

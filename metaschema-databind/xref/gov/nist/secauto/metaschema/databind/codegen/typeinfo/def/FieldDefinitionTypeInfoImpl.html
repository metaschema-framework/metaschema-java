18  import org.apache.logging.log4j.LogManager;

21  import java.util.LinkedHashMap;
22  import java.util.Map;
23  import java.util.function.Function;
24  import java.util.stream.Collectors;
25  import java.util.stream.Stream;
26  
27  import edu.umd.cs.findbugs.annotations.NonNull;
28  import nl.talsmasoftware.lazy4j.Lazy;
29  
30  class FieldDefinitionTypeInfoImpl
31      extends AbstractModelDefinitionTypeInfo<IFieldDefinition>
32      implements IFieldDefinitionTypeInfo {
33    private static final Logger LOGGER = LogManager.getLogger(FieldDefinitionTypeInfoImpl.class);
34  
35    @NonNull
36    private final Lazy<Map<String, IPropertyTypeInfo>> propertyNameToTypeInfoMap;
37    @NonNull
38    private final Lazy<Map<IInstance, IInstanceTypeInfo>> instanceToTypeInfoMap;
39  
40    public FieldDefinitionTypeInfoImpl(@NonNull IFieldDefinition definition, @NonNull ITypeResolver typeResolver) {
41      super(definition, typeResolver);
42      this.instanceToTypeInfoMap = ObjectUtils.notNull(Lazy.lazy(() -> getFlagInstanceTypeInfos().stream()
43          .collect(CustomCollectors.toMap(
44              IFlagInstanceTypeInfo::getInstance,
45              CustomCollectors.identity(),
46              (key, v1, v2) -> {
47                if (LOGGER.isErrorEnabled()) {
48                  LOGGER.error(String.format("Unexpected duplicate property name '%s'", key));
49                }
50                return ObjectUtils.notNull(v2);
51              },
52              LinkedHashMap::new))));
53      this.propertyNameToTypeInfoMap = ObjectUtils.notNull(Lazy.lazy(() -> Stream.concat(
54          getInstanceTypeInfoMap().values().stream(),
55          Stream.of((IPropertyTypeInfo) IFieldValueTypeInfo.newTypeInfo(this)))
56          .collect(Collectors.toMap(
57              IPropertyTypeInfo::getPropertyName,
58              Function.identity(),
59              (v1, v2) -> v2,
60              LinkedHashMap::new))));
61    }
62  
63    @Override
64    protected Map<String, IPropertyTypeInfo> getPropertyTypeInfoMap() {
65      return ObjectUtils.notNull(propertyNameToTypeInfoMap.get());
66    }
67  
68    @Override
69    protected Map<IInstance, IInstanceTypeInfo> getInstanceTypeInfoMap() {
70      return ObjectUtils.notNull(instanceToTypeInfoMap.get());
71    }
72  }

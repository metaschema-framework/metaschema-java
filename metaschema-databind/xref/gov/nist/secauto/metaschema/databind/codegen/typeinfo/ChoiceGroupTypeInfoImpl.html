31  
32  public class ChoiceGroupTypeInfoImpl
33      extends AbstractModelInstanceTypeInfo<IChoiceGroupInstance>
34      implements IChoiceGroupTypeInfo {
35  
36    /**
37     * Create a type information object describing a choice group instance.
38     *
39     * @param instance
40     *          the choice group instance to generate type information for
41     * @param parent
42     *          the type information for the parent assembly definition containing
43     *          the choice group
44     */
45    public ChoiceGroupTypeInfoImpl(
46        @NonNull IChoiceGroupInstance instance,
47        @NonNull IAssemblyDefinitionTypeInfo parent) {
48      super(instance, parent);
49    }
50  
51    @Override
52    public TypeName getJavaItemType() {
53      return getParentTypeInfo().getTypeResolver().getClassName(getInstance());
54    }
55  
56    @Override
57    protected AnnotationSpec.Builder newBindingAnnotation() {
58      return ObjectUtils.notNull(AnnotationSpec.builder(BoundChoiceGroup.class));
59    }
60  
61    @SuppressWarnings({ "PMD.UseConcurrentHashMap", "PMD.NPathComplexity", "PMD.CyclomaticComplexity" })
62    @Override
63    public Set<IModelDefinition> buildBindingAnnotation(
64        TypeSpec.Builder typeBuilder,
65        FieldSpec.Builder fieldBuilder,
66        AnnotationSpec.Builder annotation) {
67      IChoiceGroupInstance choiceGroup = getInstance();
68  
69      String discriminator = choiceGroup.getJsonDiscriminatorProperty();
70      if (!IChoiceGroupInstance.DEFAULT_JSON_DISCRIMINATOR_PROPERTY_NAME.equals(discriminator)) {
71        annotation.addMember("discriminator", "$S", discriminator);
72      }
73  
74      int minOccurs = choiceGroup.getMinOccurs();
75      if (minOccurs != IGroupable.DEFAULT_GROUP_AS_MIN_OCCURS) {
76        annotation.addMember("minOccurs", "$L", minOccurs);
77      }
78  
79      int maxOccurs = choiceGroup.getMaxOccurs();
80      if (maxOccurs != IGroupable.DEFAULT_GROUP_AS_MAX_OCCURS) {
81        annotation.addMember("maxOccurs", "$L", maxOccurs);
82      }
83  
84      if (maxOccurs == -1 || maxOccurs > 1) {
85        // requires a group-as
86        annotation.addMember("groupAs", "$L", generateGroupAsAnnotation().build());
87      }
88  
89      String jsonKeyName = choiceGroup.getJsonKeyFlagInstanceName();
90      if (jsonKeyName != null) {
91        annotation.addMember("jsonKey", "$S", jsonKeyName);
92      }
93  
94      Set<IModelDefinition> retval = new LinkedHashSet<>();
95  
96      IAssemblyDefinitionTypeInfo parentTypeInfo = getParentTypeInfo();
97      ITypeResolver typeResolver = parentTypeInfo.getTypeResolver();
98  
99      Map<ClassName, List<INamedModelInstanceGrouped>> referencedDefinitions = new LinkedHashMap<>();
100     Collection<? extends INamedModelInstanceGrouped> modelInstances = getInstance().getNamedModelInstances();
101     for (INamedModelInstanceGrouped modelInstance : modelInstances) {
102       ClassName className = typeResolver.getClassName(modelInstance.getDefinition());
103       List<INamedModelInstanceGrouped> instances = referencedDefinitions.get(className);
104       if (instances == null) {
105         instances = new LinkedList<>(); // NOPMD needed
106         referencedDefinitions.put(className, instances);
107       }
108       instances.add(modelInstance);
109     }
110 
111     for (INamedModelInstanceGrouped modelInstance : modelInstances) {
112       assert modelInstance != null;
113       IGroupedNamedModelInstanceTypeInfo instanceTypeInfo = typeResolver.getTypeInfo(modelInstance, this);
114 
115       ClassName className = typeResolver.getClassName(modelInstance.getDefinition());
116       retval.addAll(instanceTypeInfo.generateMemberAnnotation(
117           annotation,
118           typeBuilder,
119           referencedDefinitions.get(className).size() > 1));
120     }
121     return retval;
122   }
123 
124 }

6   package gov.nist.secauto.metaschema.databind.codegen.typeinfo;
7   
8   import com.squareup.javapoet.AnnotationSpec;
9   import com.squareup.javapoet.ClassName;
10  import com.squareup.javapoet.FieldSpec;
11  import com.squareup.javapoet.ParameterizedTypeName;
12  import com.squareup.javapoet.TypeName;
13  import com.squareup.javapoet.TypeSpec;
14  
15  import gov.nist.secauto.metaschema.core.model.IGroupable;
16  import gov.nist.secauto.metaschema.core.model.IModelDefinition;
17  import gov.nist.secauto.metaschema.core.model.IModelInstanceAbsolute;
18  import gov.nist.secauto.metaschema.core.model.JsonGroupAsBehavior;
19  import gov.nist.secauto.metaschema.core.model.XmlGroupAsBehavior;
20  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
21  import gov.nist.secauto.metaschema.databind.codegen.typeinfo.def.IAssemblyDefinitionTypeInfo;
22  import gov.nist.secauto.metaschema.databind.model.annotations.GroupAs;
23  
24  import java.util.LinkedHashSet;
25  import java.util.List;
26  import java.util.Map;
27  import java.util.Set;
28  
29  import edu.umd.cs.findbugs.annotations.NonNull;
30  
31  abstract class AbstractModelInstanceTypeInfo<INSTANCE extends IModelInstanceAbsolute>
32      extends AbstractInstanceTypeInfo<INSTANCE, IAssemblyDefinitionTypeInfo>
33      implements IModelInstanceTypeInfo {
34  
35    protected AbstractModelInstanceTypeInfo(
36        @NonNull INSTANCE instance,
37        @NonNull IAssemblyDefinitionTypeInfo parentDefinition) {
38      super(instance, parentDefinition);
39    }
40  
41    @Override
42    public String getBaseName() {
43      INSTANCE instance = getInstance();
44      String baseName = getInstance().getGroupAsName();
45      if (baseName == null) {
46        throw new IllegalStateException(String.format(
47            "Unable to derive the property name, due to missing group as name, for '%s' in the module '%s'.",
48            instance.toCoordinates(),
49            instance.getContainingModule().getLocation()));
50      }
51      return baseName;
52    }
53  
54    @Override
55    public @NonNull
56    TypeName getJavaFieldType() {
57      TypeName item = getJavaItemType();
58  
59      @NonNull
60      TypeName retval;
61      IModelInstanceAbsolute instance = getInstance();
62      int maxOccurance = instance.getMaxOccurs();
63      if (maxOccurance == -1 || maxOccurance > 1) {
64        if (JsonGroupAsBehavior.KEYED.equals(instance.getJsonGroupAsBehavior())) {
65          retval = ObjectUtils.notNull(
66              ParameterizedTypeName.get(ClassName.get(Map.class), ClassName.get(String.class), item));
67        } else {
68          retval = ObjectUtils.notNull(ParameterizedTypeName.get(ClassName.get(List.class), item));
69        }
70      } else {
71        retval = item;
72      }
73  
74      return retval;
75    }
76  
77    @NonNull
78    protected abstract AnnotationSpec.Builder newBindingAnnotation();
79  
80    @Override
81    public Set<IModelDefinition> buildField(
82        TypeSpec.Builder typeBuilder,
83        FieldSpec.Builder fieldBuilder) {
84      Set<IModelDefinition> retval = new LinkedHashSet<>(super.buildField(typeBuilder, fieldBuilder));
85  
86      AnnotationSpec.Builder annotation = newBindingAnnotation();
87  
88      retval.addAll(buildBindingAnnotation(typeBuilder, fieldBuilder, annotation));
89  
90      fieldBuilder.addAnnotation(annotation.build());
91  
92      return retval;
93    }
94  
95    @NonNull
96    protected AnnotationSpec.Builder generateGroupAsAnnotation() {
97      AnnotationSpec.Builder groupAsAnnoation = AnnotationSpec.builder(GroupAs.class);
98  
99      IModelInstanceAbsolute modelInstance = getInstance();
100 
101     groupAsAnnoation.addMember("name", "$S",
102         ObjectUtils.requireNonNull(modelInstance.getGroupAsName(), "The grouping name must be non-null"));
103 
104     // TODO: handle group-as namespace as a prefix
105 
106     JsonGroupAsBehavior jsonGroupAsBehavior = modelInstance.getJsonGroupAsBehavior();
107     assert jsonGroupAsBehavior != null;
108     if (!IGroupable.DEFAULT_JSON_GROUP_AS_BEHAVIOR.equals(jsonGroupAsBehavior)) {
109       groupAsAnnoation.addMember("inJson", "$T.$L",
110           JsonGroupAsBehavior.class, jsonGroupAsBehavior.toString());
111     }
112 
113     XmlGroupAsBehavior xmlGroupAsBehavior = modelInstance.getXmlGroupAsBehavior();
114     assert xmlGroupAsBehavior != null;
115     if (!IGroupable.DEFAULT_XML_GROUP_AS_BEHAVIOR.equals(xmlGroupAsBehavior)) {
116       groupAsAnnoation.addMember("inXml", "$T.$L",
117           XmlGroupAsBehavior.class, xmlGroupAsBehavior.toString());
118     }
119     return groupAsAnnoation;
120   }
121 }
